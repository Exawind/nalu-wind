349template <typename T1, typename T2>
350void
351copy_kokkos_unordered_map(
352 const Kokkos::UnorderedMap<T1, T2>& src, Kokkos::UnorderedMap<T1, T2>& dest)
353{
354 if (src.capacity() > dest.capacity()) {
355 dest = Kokkos::UnorderedMap<T1, T2>(src.capacity());
356 }
357
358 unsigned capacity = src.capacity();
359 unsigned fail_count = 0;
360 for (unsigned i = 0; i < capacity; ++i) {
361 if (src.valid_at(i)) {
362 auto insert_result = dest.insert(src.key_at(i));
363 fail_count += insert_result.failed() ? 1 : 0;
364 }
365 }
366 STK_ThrowRequire(fail_count == 0);
367}
