24 momentum_dt_lhs(
25  const CVFEMOperators<poly_order, Scalar>& ops,

32  const auto& weight = ops.mat_.nodalWeights;
33 
34  for (int n = 0; n < n1D; ++n) {
35  for (int m = 0; m < n1D; ++m) {
36  for (int l = 0; l < n1D; ++l) {
37  const int rowIndices[3] = {
38  idx<n1D>(XH, n, m, l), idx<n1D>(YH, n, m, l), idx<n1D>(ZH, n, m, l)};
39 
40  for (int k = 0; k < n1D; ++k) {
41  const Scalar gammaWnk = gamma1_div_dt * weight(n, k);
42  for (int j = 0; j < n1D; ++j) {
43  auto gammaWnkWmj = gammaWnk * weight(m, j);
44  for (int i = 0; i < n1D; ++i) {
45  const Scalar lhsfac =
46  gammaWnkWmj * weight(l, i) * metric(k, j, i) * rho_p1(k, j, i);
47  lhs(rowIndices[XH], idx<n1D>(XH, k, j, i)) += lhsfac;
48  lhs(rowIndices[YH], idx<n1D>(YH, k, j, i)) += lhsfac;
49  lhs(rowIndices[ZH], idx<n1D>(ZH, k, j, i)) += lhsfac;
50  }

88 template <int poly_order, typename Scalar>
89 void
90 momentum_dt_rhs(
91  const CVFEMOperators<poly_order, Scalar>& ops,
92  const nodal_scalar_view<poly_order, Scalar>& metric,
93  double gamma_div_dt[3],
94  const nodal_vector_view<poly_order, Scalar>& Gp,
95  const nodal_scalar_view<poly_order, Scalar>& rhom1,
96  const nodal_scalar_view<poly_order, Scalar>& rhop0,
97  const nodal_scalar_view<poly_order, Scalar>& rhop1,
98  const nodal_vector_view<poly_order, Scalar>& velm1,
99  const nodal_vector_view<poly_order, Scalar>& velp0,
100  const nodal_vector_view<poly_order, Scalar>& velp1,
101  nodal_vector_view<poly_order, Scalar>& rhs)
102 {
103  constexpr int n1D = poly_order + 1;
104 
105  nodal_vector_workview<poly_order, Scalar> work_drhoudt(0);
106  auto& drhoudt = work_drhoudt.view();
107 
108  for (int k = 0; k < n1D; ++k) {
109  for (int j = 0; j < n1D; ++j) {
110  for (int i = 0; i < n1D; ++i) {
111  const Scalar local_vol = metric(k, j, i);
112  const Scalar facp1 = gamma_div_dt[0] * rhop1(k, j, i);
113  const Scalar facp0 = gamma_div_dt[1] * rhop0(k, j, i);
114  const Scalar facm1 = gamma_div_dt[2] * rhom1(k, j, i);
115  for (int d = 0; d < 3; ++d) {
116  drhoudt(k, j, i, d) =
117  -(facp1 * velp1(k, j, i, d) + facp0 * velp0(k, j, i, d) +
118  facm1 * velm1(k, j, i, d) + Gp(k, j, i, d)) *
119  local_vol;
120  }
121  }
122  }
123  }
124  ops.volume(drhoudt, rhs);
125 }
