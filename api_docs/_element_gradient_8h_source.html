26template <int p, typename BoxArray, typename InpArray>
27KOKKOS_FUNCTION
28 typename std::enable_if<InpArray::rank == 4, ArrayND<ftype[3][3]>>::type
29 gradient_nodal(const BoxArray& box, const InpArray& u, int k, int j, int i)
30{
31 ArrayND<ftype[3][3]> gu_hat;
32 for (int dj = 0; dj < 3; ++dj) {
33 gu_hat(dj, 0) = 0;
34 gu_hat(dj, 1) = 0;
35 gu_hat(dj, 2) = 0;
36 for (int q = 0; q < p + 1; ++q) {
37 static constexpr auto D = Coeffs<p>::D;
38 gu_hat(dj, 0) += D(i, q) * u(k, j, q, dj);
39 gu_hat(dj, 1) += D(j, q) * u(k, q, i, dj);
40 gu_hat(dj, 2) += D(k, q) * u(q, j, i, dj);
41 }
42 }
43
44 ArrayND<ftype[3][3]> gu;
45 inv_transform_t(geom::linear_hex_jacobian<p>(box, k, j, i), gu_hat, gu);
46 return gu;
47}

75KOKKOS_FORCEINLINE_FUNCTION
76 typename std::enable_if<UArrayType::rank == 4, ArrayND<ftype[3][3]>>::type
77 gradient_scs(
78 const BoxArrayType& box,
79 const UArrayType& u,
80 const UHatArrayType& uhat,
81 int l,
82 int s,
83 int r)
84{
85 enum { XH = 0, YH = 1, ZH = 2 };
86 constexpr int dir_0 = dir;
87 constexpr int dir_1 = (dir == XH) ? YH : XH;
88 constexpr int dir_2 = (dir == ZH) ? YH : ZH;
89
90 ArrayND<ftype[3][3]> gu_hat;
91 for (int d = 0; d < 3; ++d) {
92 ftype acc = 0;
93 for (int q = 0; q < p + 1; ++q) {
94 static constexpr auto deriv = Coeffs<p>::Dt;
95 acc += deriv(l, q) * shuffled_access<dir>(u, s, r, q, d);
96 }
97 gu_hat(d, dir_0) = acc;
98
99 acc = 0;
100 for (int q = 0; q < p + 1; ++q) {
101 static constexpr auto deriv = Coeffs<p>::D;
102 acc += deriv(r, q) * uhat(s, q, d);
103 }
104 gu_hat(d, dir_1) = acc;
105
106 acc = 0;
107 for (int q = 0; q < p + 1; ++q) {
108 static constexpr auto deriv = Coeffs<p>::D;
109 acc += deriv(s, q) * uhat(q, r, d);
110 }
111 gu_hat(d, dir_2) = acc;
112 }
113 ArrayND<ftype[3][3]> gu;
114 inv_transform_t(
115 geom::linear_hex_jacobian_scs<p, dir>(box, l, s, r), gu_hat, gu);
