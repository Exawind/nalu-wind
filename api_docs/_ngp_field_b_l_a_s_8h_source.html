27template <typename Mesh, typename FieldType, typename ScalarType>
28inline void
29field_axpby(
30 const Mesh& ngpMesh,
31 const stk::mesh::Selector& sel,
32 const ScalarType alpha,
33 const FieldType& xField,
34 const ScalarType beta,
35 FieldType& yField,
36 const unsigned numComponents = 1,
37 const stk::topology::rank_t rank = stk::topology::NODE_RANK)
38{
39 static_assert(
40 std::is_same<typename FieldType::value_type, ScalarType>::value,
41 "Mismatch on field data types");
42
43 using Traits = NGPMeshTraits<Mesh>;
44 using MeshIndex = typename Traits::MeshIndex;
45
46 yField.sync_to_device();
47
48 nalu_ngp::run_entity_algorithm(
49 "ngp_field_axpby", ngpMesh, rank, sel, KOKKOS_LAMBDA(const MeshIndex& mi) {
50 for (unsigned d = 0; d < numComponents; ++d)
51 yField.get(mi, d) =
52 alpha * xField.get(mi, d) + beta * yField.get(mi, d);
53 });
54
55 // Indicate modification on device for future synchronization
56 yField.modify_on_device();
57}

60inline void
61field_copy(
62 const Mesh& ngpMesh,
63 const stk::mesh::Selector& sel,
64 FieldType& dest,
65 const FieldType& src,
66 const unsigned start,
67 const unsigned end,
68 const stk::topology::rank_t rank = stk::topology::NODE_RANK)
69{
70 using Traits = NGPMeshTraits<Mesh>;
71 using MeshIndex = typename Traits::MeshIndex;
72
73 nalu_ngp::run_entity_algorithm(
74 "ngp_field_copy", ngpMesh, rank, sel, KOKKOS_LAMBDA(const MeshIndex& mi) {
75 for (unsigned d = start; d < end; ++d)
76 dest.get(mi, d) = src.get(mi, d);
77 });
78
79 // Indicate modification on device for future synchronization
80 dest.modify_on_device();

96template <typename MeshInfoType>
97inline void
98field_copy(
99 const MeshInfoType& meshInfo,
100 stk::mesh::FieldBase& dest,
101 const stk::mesh::FieldBase& src,
102 const unsigned numComponents = 1,
103 const stk::topology::rank_t rank = stk::topology::NODE_RANK)
104{
105 const stk::mesh::Selector sel = stk::mesh::selectField(dest);
106 const auto& fieldMgr = meshInfo.ngp_field_manager();
107 const auto& srcField =
108 fieldMgr.template get_field<double>(src.mesh_meta_data_ordinal());
109 auto& destField =
110 fieldMgr.template get_field<double>(dest.mesh_meta_data_ordinal());
111 field_copy(
112 meshInfo.ngp_mesh(), sel, destField, srcField, 0, numComponents, rank);
113}
