52template <class FROM, class TO>
53void
54LinInterp<FROM, TO>::filter_to_nearest(
55 EntityKeyMap& RangeToDomain, const MeshA& FromElem, MeshB& ToPoints)
56{
57
58 const stk::mesh::BulkData& fromBulkData = FromElem.fromBulkData_;
59 stk::mesh::BulkData& toBulkData = ToPoints.toBulkData_;
60
61 const VectorFieldType* fromcoordinates = FromElem.fromcoordinates_;
62 const VectorFieldType* tocoordinates = ToPoints.tocoordinates_;
63
64 // FixMe: Check nDim against Dimension to make sure they are equal
65 const unsigned nDim = FromElem.fromMetaData_.spatial_dimension();
66
67 typedef typename EntityKeyMap::iterator iterator;
68 typedef typename EntityKeyMap::const_iterator const_iterator;
69
70 // some simple user diagnostics to let the user know if something bad is
71 // happening
72 double maxBestX = -std::numeric_limits<double>::max();
73 size_t maxCandidateBoundingBox = 0;
74
75 for (const_iterator current_key = RangeToDomain.begin();
76 current_key != RangeToDomain.end();) {
77
78 double bestX_ = std::numeric_limits<double>::max();
79
80 const stk::mesh::EntityKey thePt = current_key->first;
81 stk::mesh::Entity theNode = toBulkData.get_entity(thePt);
82 // load nodal coordinates from node
83 const double* tocoords = stk::mesh::field_data(*tocoordinates, theNode);
84
85 std::pair<iterator, iterator> keys =
86 RangeToDomain.equal_range(current_key->first);
87 iterator nearest = keys.second;
88
89 size_t candidateBoundingBoxSize = 0;
90 for (iterator ii = keys.first; ii != keys.second; ++ii) {
91 candidateBoundingBoxSize++;
92
93 const stk::mesh::EntityKey theBox = ii->second;
94 stk::mesh::Entity theElem = fromBulkData.get_entity(theBox);
95
96 // extract master element from the bucket in which the element resides
97 const stk::mesh::Bucket& theBucket = fromBulkData.bucket(theElem);
98 const stk::topology& theElemTopo = theBucket.topology();
99 MasterElement* meSCS =
100 sierra::nalu::MasterElementRepo::get_surface_master_element_on_host(
101 theElemTopo);
102
103 // load nodal coordinates from element
104 stk::mesh::Entity const* elem_node_rels =
105 fromBulkData.begin_nodes(theElem);
106 const int num_nodes = fromBulkData.num_nodes(theElem);
107
108 const int nodesPerElement = meSCS->nodesPerElement_;
109 std::vector<double> theElementCoords(nDim * nodesPerElement);
110
111 for (int ni = 0; ni < num_nodes; ++ni) {
112 stk::mesh::Entity node = elem_node_rels[ni];
113
114 // load up vectors
115 const double* fromcoords =
116 stk::mesh::field_data(*fromcoordinates, node);
117 for (unsigned j = 0; j < nDim; ++j) {
118 const int offSet = j * nodesPerElement + ni;
119 theElementCoords[offSet] = fromcoords[j];
120 }
121 }
122
123 std::vector<double> isoParCoords(nDim);
124 const double nearestDistance = meSCS->isInElement(
125 &theElementCoords[0], &(tocoords[0]), &(isoParCoords[0]));
126 if (nearestDistance < bestX_) {
127 bestX_ = nearestDistance;
128 ToPoints.TransferInfo_[thePt] = isoParCoords;
129 nearest = ii;
130 maxBestX = std::max(maxBestX, bestX_);
131 }
132 }
133 maxCandidateBoundingBox =
134 std::max(maxCandidateBoundingBox, candidateBoundingBoxSize);
135
136 current_key = keys.second;
137 if (nearest != keys.first)
138 RangeToDomain.erase(keys.first, nearest);
139 if (nearest != keys.second)
140 RangeToDomain.erase(++nearest, keys.second);
141 }
142
143 // parallel sum and output diagnostics
144 double g_maxBestX = 0.0;
145 size_t g_maxCandidateBoundngBox = 0;
146 stk::all_reduce_max(FromElem.comm(), &maxBestX, &g_maxBestX, 1);
147 stk::all_reduce_max(
148 FromElem.comm(), &maxCandidateBoundingBox, &g_maxCandidateBoundngBox, 1);
149
150 NaluEnv::self().naluOutputP0() << std::endl;
151 NaluEnv::self().naluOutputP0()
152 << "XFER::LinInterp::fine_search() Overview:" << std::endl;
153 NaluEnv::self().naluOutputP0()
154 << " Maximum normalized distance found is: " << g_maxBestX
155 << " (should be unity or less)" << std::endl;
156 NaluEnv::self().naluOutputP0() << " Maximum number of candidate bounding "
157 "boxes found for a single point is: "
158 << g_maxCandidateBoundngBox << std::endl;
159 NaluEnv::self().naluOutputP0()
160 << " Should max normalized distance and/or candidate bounding box size be "
161 "too large, please check setup"
162 << std::endl;
163}

165template <class FROM, class TO>
166void
167LinInterp<FROM, TO>::apply(
168 MeshB& ToPoints, const MeshA& FromElem, const EntityKeyMap& RangeToDomain)
169{
170
171 const stk::mesh::BulkData& fromBulkData = FromElem.fromBulkData_;
172 stk::mesh::BulkData& toBulkData = ToPoints.toBulkData_;
173
174 typename EntityKeyMap::const_iterator ii;
175 for (ii = RangeToDomain.begin(); ii != RangeToDomain.end(); ++ii) {
176
177 const stk::mesh::EntityKey thePt = ii->first;
178 const stk::mesh::EntityKey theBox = ii->second;
179
180 if (1 != ToPoints.TransferInfo_.count(thePt)) {
181 if (0 == ToPoints.TransferInfo_.count(thePt))
182 throw std::runtime_error("Key not found in database");
183 else
184 throw std::runtime_error("Too many Keys found in database");
185 }
186 const std::vector<double>& isoParCoords_ = ToPoints.TransferInfo_[thePt];
187 stk::mesh::Entity theNode = toBulkData.get_entity(thePt);
188 stk::mesh::Entity theElem = fromBulkData.get_entity(theBox);
189
190 const stk::mesh::Bucket& theBucket = fromBulkData.bucket(theElem);
191 const stk::topology& theElemTopo = theBucket.topology();
192 MasterElement* meSCS =
193 sierra::nalu::MasterElementRepo::get_surface_master_element_on_host(
194 theElemTopo);
195
196 stk::mesh::Entity const* elem_node_rels = fromBulkData.begin_nodes(theElem);
197 const int num_nodes = fromBulkData.num_nodes(theElem);
198 const int nodesPerElement = meSCS->nodesPerElement_;
199
200 for (unsigned n = 0; n != FromElem.fromFieldVec_.size(); ++n) {
201
202 // extract field
203 const stk::mesh::FieldBase* toFieldBaseField = ToPoints.toFieldVec_[n];
204
205 // FixMe: integers are problematic for now...
206 const size_t sizeOfField =
207 field_bytes_per_entity(*toFieldBaseField, theNode) / sizeof(double);
208 std::vector<double> Coeff(nodesPerElement * sizeOfField);
209
210 // now load the elemental values for future interpolation; fill in
211 // connected nodes
212 for (int ni = 0; ni < num_nodes; ++ni) {
213 stk::mesh::Entity node = elem_node_rels[ni];
214
215 const stk::mesh::FieldBase* fromFieldBaseField =
216 FromElem.fromFieldVec_[n];
217 const double* theField =
218 (double*)stk::mesh::field_data(*fromFieldBaseField, node);
219
220 for (size_t j = 0; j < sizeOfField; ++j) {
221 const int offSet = j * nodesPerElement + ni;
222 Coeff[offSet] = theField[j];
223 }
224 }
225
226 double* toField =
227 (double*)stk::mesh::field_data(*toFieldBaseField, theNode);
228 if (!toField)
229 throw std::runtime_error("Receiving field undefined on mesh object.");
230 meSCS->interpolatePoint(
231 sizeOfField, &isoParCoords_[0], &Coeff[0], toField);
232 }
233 }
234}
