24 template <typename range_type, typename helper_type>
25 void
26 scale_lift_force(
27  ActuatorBulk& actBulk,
28  const ActuatorMeta& actMeta,
29  range_type& rangePolicy,
30  helper_type& helper,
31  const int offset,
32  const int nPoints)
33 {
34  // suppress compiler warnings for unused variables when compiling w/o openfast
35  (void)offset;
36  (void)nPoints;
37 
38  switch (actMeta.actuatorType_) {
39  case (ActuatorType::ActLineSimpleNGP): {
40  auto actBulkSimple = dynamic_cast<ActuatorBulkSimple&>(actBulk);
41  auto actMetaSimple = dynamic_cast<const ActuatorMetaSimple&>(actMeta);
42  auto G = helper.get_local_view(actBulkSimple.liftForceDistribution_);
43  auto rho = helper.get_local_view(actBulkSimple.density_);
44 
45  const int turbId = actBulkSimple.localTurbineId_;
46  double dR = actMetaSimple.dR_.h_view(turbId);
47 
48  Kokkos::parallel_for(
49  "scale G", rangePolicy, ACTUATOR_LAMBDA(int i) {
50  const double denom = rho(i) * dR;
51  for (int j = 0; j < 3; ++j) {
52  G(i, j) /= denom;
53  }
54  });
55  break;
56  }
57  case (ActuatorType::ActLineFASTNGP):
58  case (ActuatorType::ActDiskFASTNGP): {
59 #ifndef NALU_USES_OPENFAST
60  STK_ThrowErrorMsg("Actuator methods require OpenFAST");
61 #if !defined(KOKKOS_ENABLE_GPU)
62  break;
63 #endif
64 #else
65  auto G = helper.get_local_view(actBulk.liftForceDistribution_);
66  auto point = helper.get_local_view(actBulk.pointCentroid_);
67  Kokkos::parallel_for(
68  "scale G FAST outputs", rangePolicy, ACTUATOR_LAMBDA(int i) {
69  double dr = 0;
70  if (i == offset) {
71  for (int j = 0; j < 3; ++j) {
72  dr += std::pow(point(i, j) - point(i + 1, j), 2.0);
73  }
74  } else if (i == offset + nPoints - 1) {
75  for (int j = 0; j < 3; ++j) {
76  dr += std::pow(point(i, j) - point(i - 1, j), 2.0);
77  }
78  } else {
79  for (int j = 0; j < 3; ++j) {
80  dr += std::pow(point(i - 1, j) - point(i + 1, j), 2.0);
81  }
82  }
83  // dr is computed using central difference
84  dr = 0.5 * std::sqrt(dr);
85  for (int j = 0; j < 3; ++j) {
86  G(i, j) /= dr;
87  }
88  });
89  break;
90 #endif
91  }
92  default: {
93  throw std::runtime_error("Unsupported actuator type supplied to the fllc");
94  }
95  }
96 }
