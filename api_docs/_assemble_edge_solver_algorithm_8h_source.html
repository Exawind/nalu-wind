47 template <typename LambdaFunction>
48 void run_algorithm(stk::mesh::BulkData& bulk, LambdaFunction lambdaFunc)
49 {
50 const auto& meta = bulk.mesh_meta_data();
51 const auto& ngpMesh = realm_.ngp_mesh();
52
53 const int bytes_per_team = 0;
54 const int bytes_per_thread = calc_shmem_bytes_per_thread_edge(rhsSize_);
55
56 stk::mesh::Selector sel = meta.locally_owned_part() &
57 stk::mesh::selectUnion(partVec_) &
58 !(realm_.get_inactive_selector());
59
60 const auto& buckets = stk::mesh::get_bucket_ids(bulk, entityRank_, sel);
61 auto team_exec =
62 get_device_team_policy(buckets.size(), bytes_per_team, bytes_per_thread);
63
64 // Create local copies of class data for device capture
65 const auto entityRank = entityRank_;
66 const auto rhsSize = rhsSize_;
67
68 auto coeffApplier = coeff_applier();
69
70 const auto nodesPerEntity = nodesPerEntity_;
71
72 Kokkos::parallel_for(
73 team_exec, KOKKOS_LAMBDA(const DeviceTeamHandleType& team) {
74 auto bktId = buckets.device_get(team.league_rank());
75 auto& b = ngpMesh.get_bucket(entityRank, bktId);
76
77 ShmemDataType smdata(team, rhsSize);
78
79 const size_t bktLen = b.size();
80 Kokkos::parallel_for(
81 Kokkos::TeamThreadRange(team, bktLen), [&](const size_t& bktIndex) {
82 auto edge = b[bktIndex];
83 const auto edgeIndex = ngpMesh.fast_mesh_index(edge);
84 smdata.ngpElemNodes = ngpMesh.get_nodes(entityRank, edgeIndex);
85
86 const auto nodeL = ngpMesh.fast_mesh_index(smdata.ngpElemNodes[0]);
87 const auto nodeR = ngpMesh.fast_mesh_index(smdata.ngpElemNodes[1]);
88
89 set_vals(smdata.rhs, 0.0);
90 set_vals(smdata.lhs, 0.0);
91
92 lambdaFunc(smdata, edgeIndex, nodeL, nodeR);
93
94 coeffApplier(
95 nodesPerEntity, smdata.ngpElemNodes, smdata.scratchIds,
96 smdata.sortPermutation, smdata.rhs, smdata.lhs, __FILE__);
97 });
98 });
99 coeffApplier.free_coeff_applier();
100 }
