73struct EntityIdVectorHash
74{
75 std::size_t operator()(const stk::mesh::EntityIdVector& ids) const
76 {
77 if (ids.empty()) {
78 return 0;
79 }
80
81 constexpr uint32_t shift = 0x9e3779b9;
82 auto hash = std::hash<stk::mesh::EntityId>()(ids[0]);
83 for (unsigned j = 1; j < ids.size(); ++j) {
84 hash ^=
85 std::hash<std::size_t>()(ids[j]) + shift + (hash << 6) + (hash >> 2);
86 }
87 return hash;
88 }
89};
