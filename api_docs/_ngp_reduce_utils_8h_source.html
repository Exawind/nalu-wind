27template <typename ScalarType, int N>
28struct NgpReduceArray
29{
30 ScalarType array_[N];
31
32 KOKKOS_INLINE_FUNCTION
33 NgpReduceArray() {}
34
35 KOKKOS_INLINE_FUNCTION
36 NgpReduceArray(ScalarType val)
37 {
38 for (int i = 0; i < N; ++i)
39 array_[i] = val;
40 }
41
42 KOKKOS_INLINE_FUNCTION
43 NgpReduceArray(const NgpReduceArray& rhs)
44 {
45 for (int i = 0; i < N; ++i)
46 array_[i] = rhs.array_[i];
47 }
