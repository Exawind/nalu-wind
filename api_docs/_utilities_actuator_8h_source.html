86 template <typename T>
87 inline void
88 reduce_view_on_host(T view)
89 {
90  STK_ThrowAssert(view.size() > 0);
91  STK_ThrowAssert(view.data());
92  MPI_Datatype mpi_type;
93  if (std::is_same<typename T::value_type, double>::value) {
94  mpi_type = MPI_DOUBLE;
95  } else if (std::is_same<typename T::value_type, int>::value) {
96  mpi_type = MPI_INT;
97  } else if (std::is_same<typename T::value_type, bool>::value) {
98  mpi_type = MPI_C_BOOL;
99  } else if (std::is_same<typename T::value_type, uint64_t>::value) {
100  mpi_type = MPI_LONG;
101  } else {
102  STK_ThrowErrorMsg("unsupported type to reduce view on host");
103  }
104 
105  MPI_Allreduce(
106  MPI_IN_PLACE, view.data(), view.size(), mpi_type, MPI_SUM,
107  NaluEnv::self().parallel_comm());
108 }
