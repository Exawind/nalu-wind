25 template <typename Scalar>
26 KOKKOS_FORCEINLINE_FUNCTION Scalar
27 determinant(const LocalArray<Scalar[3][3]>& jac)
28 {
29  return jac(XH, XH) * (jac(YH, YH) * jac(ZH, ZH) - jac(YH, ZH) * jac(ZH, YH)) -
30  jac(XH, YH) * (jac(YH, XH) * jac(ZH, ZH) - jac(YH, ZH) * jac(ZH, XH)) +
31  jac(XH, ZH) * (jac(YH, XH) * jac(ZH, YH) - jac(YH, YH) * jac(ZH, XH));
32 }

50 template <typename Scalar>
51 KOKKOS_FORCEINLINE_FUNCTION LocalArray<Scalar[3][3]>
52 invert_matrix(const LocalArray<Scalar[3][3]>& mat)
53 {
54  LocalArray<Scalar[3][3]> inv;
55  auto inv_detj = 1. / determinant(mat);
56  inv(XH, XH) =
57  inv_detj * (mat(YH, YH) * mat(ZH, ZH) - mat(ZH, YH) * mat(YH, ZH));
58  inv(XH, YH) =
59  inv_detj * (mat(YH, ZH) * mat(ZH, XH) - mat(ZH, ZH) * mat(YH, XH));
60  inv(XH, ZH) =
61  inv_detj * (mat(YH, XH) * mat(ZH, YH) - mat(ZH, XH) * mat(YH, YH));
62  inv(YH, XH) =
63  inv_detj * (mat(XH, ZH) * mat(ZH, YH) - mat(ZH, ZH) * mat(XH, YH));
64  inv(YH, YH) =
65  inv_detj * (mat(XH, XH) * mat(ZH, ZH) - mat(ZH, XH) * mat(XH, ZH));
66  inv(YH, ZH) =
67  inv_detj * (mat(XH, YH) * mat(ZH, XH) - mat(ZH, YH) * mat(XH, XH));
68  inv(ZH, XH) =
69  inv_detj * (mat(XH, YH) * mat(YH, ZH) - mat(YH, YH) * mat(XH, ZH));
70  inv(ZH, YH) =
71  inv_detj * (mat(XH, ZH) * mat(YH, XH) - mat(YH, ZH) * mat(XH, XH));
72  inv(ZH, ZH) =
73  inv_detj * (mat(XH, XH) * mat(YH, YH) - mat(YH, XH) * mat(XH, YH));
74 
75  return inv;
76 }

97  const LocalArray<InpScalar[3][3]>& A,
98  const Kokkos::Array<OutScalar, 3>& x,
99  Kokkos::Array<OutScalar, 3>& y)

102  y[YH] = A(YH, XH) * x[XH] + A(YH, YH) * x[YH] + A(YH, ZH) * x[ZH];
103  y[ZH] = A(ZH, XH) * x[XH] + A(ZH, YH) * x[YH] + A(ZH, ZH) * x[ZH];
104 }

173 template <typename Scalar>
174 KOKKOS_FORCEINLINE_FUNCTION LocalArray<Scalar[3][3]>
175 square(const LocalArray<Scalar[3][3]>& a)
176 {
177  LocalArray<ftype[3][3]> b;
178  for (int dj = 0; dj < 3; ++dj) {
179  for (int di = 0; di < 3; ++di) {
180  b(dj, di) =
181  a(dj, 0) * a(0, di) + a(dj, 1) * a(1, di) + a(dj, 2) * a(2, di);
182  }
183  }
184  return b;
