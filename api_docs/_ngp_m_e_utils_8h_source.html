35template <typename DataReqType>
36MasterElement*
37get_me_instance(const DataReqType& dataReq, const METype meType)
38{
39 MasterElement* me = nullptr;
40
41 switch (meType) {
42 case METype::SCV:
43 me = dataReq.get_cvfem_volume_me();
44 break;
45
46 case METype::SCS:
47 me = dataReq.get_cvfem_surface_me();
48 break;
49
50 case METype::FACE:
51 me = dataReq.get_cvfem_face_me();
52 break;
53
54 case METype::FEM:
55 me = dataReq.get_fem_volume_me();
56 break;
57 }
58
59 return me;
60}

62template <typename DataReqType>
63int
64nodes_per_entity(const DataReqType& dataReq, const METype meType)
65{
66 auto* me = get_me_instance(dataReq, meType);
67
68 // Return immediately if ME is not registered
69 if (me == nullptr)
70 return 0;
71
72 Kokkos::View<int*, sierra::nalu::MemSpace> npe("npe", 1);
73 Kokkos::parallel_for(
74 "get_nodes_per_element", DeviceRangePolicy(0, 1),
75 KOKKOS_LAMBDA(const int i) { npe(i) = me->nodesPerElement_; });
76 Kokkos::View<int*, sierra::nalu::MemSpace>::HostMirror npe_host("npe", 1);
77 Kokkos::deep_copy(npe_host, npe);
78 return npe_host(0);
79}

100template <typename DataReqType>
101int
102nodes_per_entity(const DataReqType& dataReq)
103{
104 auto* meFC = dataReq.get_cvfem_face_me();
105 auto* meSCS = dataReq.get_cvfem_surface_me();
106 auto* meSCV = dataReq.get_cvfem_volume_me();
107 auto* meFEM = dataReq.get_fem_volume_me();
108
109 int npe = (meSCS != nullptr) ? nodes_per_entity(dataReq, METype::SCS)
110 : (meSCV != nullptr) ? nodes_per_entity(dataReq, METype::SCV)
111 : (meFEM != nullptr) ? nodes_per_entity(dataReq, METype::FEM)
112 : (meFC != nullptr) ? nodes_per_entity(dataReq, METype::FACE)
113 : 0;
114
115 return npe;
116}
