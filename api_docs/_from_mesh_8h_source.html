62  std::vector<const stk::mesh::FieldBase*> get_fields(
63  const stk::mesh::MetaData& fromMetaData, const PairNames& VarPairName)
64  {
65  // will want to check that all is well with field registration
66  bool allFieldsAreFine = true;
67  std::vector<const stk::mesh::FieldBase*> fromFieldVec;
68  // provide field names
69  for (PairNames::const_iterator i = VarPairName.begin();
70  i != VarPairName.end(); ++i) {
71  const std::string& fieldName = i->first;
72  const stk::mesh::FieldBase* fromfield =
73  stk::mesh::get_field_by_name(fieldName, fromMetaData);
74  if (NULL == fromfield) {
75  allFieldsAreFine = false;
76  NaluEnv::self().naluOutputP0()
77  << "Xfer::FromMesh:Error field: " << fieldName
78  << " has not been registered anywhere within the FromRealm: "
79  << fromRealm_.name() << std::endl;
80  } else {
81  // always push back; check for errors below
82  fromFieldVec.push_back(fromfield);
83 
84  // check that the field is defined on **all** parts
85  stk::mesh::Selector fieldSelector = stk::mesh::selectField(*fromfield);
86  for (size_t k = 0; k < fromPartVec_.size(); ++k) {
87 
88  stk::mesh::BucketVector const& partBuckets =
89  fromBulkData_.get_buckets(
90  stk::topology::NODE_RANK, stk::mesh::Selector(*fromPartVec_[k]));
91 
92  bool fieldIsFine = true;
93  for (stk::mesh::BucketVector::const_iterator ib = partBuckets.begin();
94  ib != partBuckets.end(); ++ib) {
95  stk::mesh::Bucket& b = **ib;
96  fieldIsFine &= fieldSelector(b);
97  }
98 
99  // local check to make sure that the field is somewhere (delay the
100  // throw)
101  if (!fieldIsFine) {
102  NaluEnv::self().naluOutputP0()
103  << "Xfer::FromMesh:Error field: " << fromfield->name()
104  << " is not registered on part: " << fromPartVec_[k]->name()
105  << std::endl;
106  allFieldsAreFine = false;
107  }
108  }
109  }
110  }
111 
112  // final error check; only return when all is well
113  if (allFieldsAreFine) {
114  return fromFieldVec;
115  } else {
116  throw std::runtime_error("Xfer::FromMesh:Error field registration on "
117  "desired parts of the mesh is not complete");
118  }
119  }

157  void bounding_boxes(std::vector<BoundingBox>& v) const
158  {
159  const unsigned nDim = fromMetaData_.spatial_dimension();
160 
161  Point min_corner, max_corner;
162 
163  stk::mesh::Selector s_locally_owned_union =
164  fromMetaData_.locally_owned_part() & stk::mesh::selectUnion(fromPartVec_);
165 
166  // determine entity rank for the part served up; should be homogeneous
167  stk::mesh::EntityRank partEntityRank =
168  fromPartVec_[0]->primary_entity_rank();
169 
170  stk::mesh::BucketVector const& entity_buckets =
171  fromBulkData_.get_buckets(partEntityRank, s_locally_owned_union);
172  for (stk::mesh::BucketVector::const_iterator ib = entity_buckets.begin();
173  ib != entity_buckets.end(); ++ib) {
174  stk::mesh::Bucket& b = **ib;
175 
176  const stk::mesh::Bucket::size_type length = b.size();
177 
178  for (stk::mesh::Bucket::size_type k = 0; k < length; ++k) {
179 
180  // get entity
181  stk::mesh::Entity theEntity = b[k];
182 
183  // initialize max and min
184  for (unsigned j = 0; j < nDim; ++j) {
185  min_corner[j] = +1.0e16;
186  max_corner[j] = -1.0e16;
187  }
188 
189  stk::mesh::Entity const* entity_node_rels =
190  fromBulkData_.begin_nodes(theEntity);
191  int num_entity_nodes = fromBulkData_.num_nodes(theEntity);
192  for (int ni = 0; ni < num_entity_nodes; ++ni) {
193  stk::mesh::Entity node = entity_node_rels[ni];
194 
195  double* coords = stk::mesh::field_data(*fromcoordinates_, node);
196  for (unsigned j = 0; j < nDim; ++j) {
197  min_corner[j] = std::min(min_corner[j], coords[j]);
198  max_corner[j] = std::max(max_corner[j], coords[j]);
199  }
200  }
201 
202  // setup ident
203  FromMesh::EntityProc theIdent(
204  fromBulkData_.entity_key(theEntity), fromBulkData_.parallel_rank());
205 
206  v.push_back(BoundingBox(Box(min_corner, max_corner), theIdent));
207  }
208  }
209  std::sort(v.begin(), v.end(), BoundingBoxCompare());
210  }

212  void update_ghosting(const EntityProcVec& entity_keys)
213  {
214  ghosting_map_.resize(entity_keys.size());
215  for (size_t i = 0; i < entity_keys.size(); ++i) {
216  // convert from EntityProc based on EntityKey to EntityProc based on raw
217  // Entity.
218  const EntityProc& key_proc = entity_keys[i];
219  const stk::mesh::EntityKey key = key_proc.id();
220  const unsigned proc = key_proc.proc();
221  const stk::mesh::Entity e = entity(key);
222  const stk::mesh::EntityProc ep(e, proc);
223  ghosting_map_[i] = ep;
224  }
225 
226  unsigned s = !ghosting_map_.empty();
227  stk::all_reduce(comm_, stk::ReduceSum<1>(&s));
228 
229  if (s) {
230  std::sort(ghosting_map_.begin(), ghosting_map_.end());
231  stk::mesh::EntityProcVec::iterator del =
232  std::unique(ghosting_map_.begin(), ghosting_map_.end());
233  ghosting_map_.resize(std::distance(ghosting_map_.begin(), del));
234 
235  std::string theGhostName = "nalu_transfer_ghosting";
236  for (unsigned i = 0; i != fromFieldVec_.size(); ++i)
237  theGhostName += "_" + fromFieldVec_[i]->name();
238  ghosting_ = &fromBulkData_.create_ghosting(theGhostName);
239  fromBulkData_.change_ghosting(*ghosting_, ghosting_map_);
240  mesh_modified_ = true;
241  }
242  }
