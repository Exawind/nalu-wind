62 std::vector<const stk::mesh::FieldBase*> get_fields(
63 const stk::mesh::MetaData& fromMetaData, const PairNames& VarPairName)
64 {
65 // will want to check that all is well with field registration
66 bool allFieldsAreFine = true;
67 std::vector<const stk::mesh::FieldBase*> fromFieldVec;
68 // provide field names
69 for (PairNames::const_iterator i = VarPairName.begin();
70 i != VarPairName.end(); ++i) {
71 const std::string& fieldName = i->first;
72 const stk::mesh::FieldBase* fromfield =
73 stk::mesh::get_field_by_name(fieldName, fromMetaData);
74 if (NULL == fromfield) {
75 allFieldsAreFine = false;
76 NaluEnv::self().naluOutputP0()
77 << "Xfer::FromMesh:Error field: " << fieldName
78 << " has not been registered anywhere within the FromRealm: "
79 << fromRealm_.name() << std::endl;
80 } else {
81 // always push back; check for errors below
82 fromFieldVec.push_back(fromfield);
83
84 // check that the field is defined on **all** parts
85 stk::mesh::Selector fieldSelector = stk::mesh::selectField(*fromfield);
86 for (size_t k = 0; k < fromPartVec_.size(); ++k) {
87
88 stk::mesh::BucketVector const& partBuckets =
89 fromBulkData_.get_buckets(
90 stk::topology::NODE_RANK, stk::mesh::Selector(*fromPartVec_[k]));
91
92 bool fieldIsFine = true;
93 for (stk::mesh::BucketVector::const_iterator ib = partBuckets.begin();
94 ib != partBuckets.end(); ++ib) {
95 stk::mesh::Bucket& b = **ib;
96 fieldIsFine &= fieldSelector(b);
97 }
98
99 // local check to make sure that the field is somewhere (delay the
100 // throw)
101 if (!fieldIsFine) {
102 NaluEnv::self().naluOutputP0()
103 << "Xfer::FromMesh:Error field: " << fromfield->name()
104 << " is not registered on part: " << fromPartVec_[k]->name()
105 << std::endl;
106 allFieldsAreFine = false;
107 }
108 }
109 }
110 }
111
112 // final error check; only return when all is well
113 if (allFieldsAreFine) {
114 return fromFieldVec;
115 } else {
116 throw std::runtime_error(
117 "Xfer::FromMesh:Error field registration on "
118 "desired parts of the mesh is not complete");
119 }
120 }

158 void bounding_boxes(std::vector<BoundingBox>& v) const
159 {
160 const unsigned nDim = fromMetaData_.spatial_dimension();
161
162 Point min_corner, max_corner;
163
164 stk::mesh::Selector s_locally_owned_union =
165 fromMetaData_.locally_owned_part() & stk::mesh::selectUnion(fromPartVec_);
166
167 // determine entity rank for the part served up; should be homogeneous
168 stk::mesh::EntityRank partEntityRank =
169 fromPartVec_[0]->primary_entity_rank();
170
171 stk::mesh::BucketVector const& entity_buckets =
172 fromBulkData_.get_buckets(partEntityRank, s_locally_owned_union);
173 for (stk::mesh::BucketVector::const_iterator ib = entity_buckets.begin();
174 ib != entity_buckets.end(); ++ib) {
175 stk::mesh::Bucket& b = **ib;
176
177 const stk::mesh::Bucket::size_type length = b.size();
178
179 for (stk::mesh::Bucket::size_type k = 0; k < length; ++k) {
180
181 // get entity
182 stk::mesh::Entity theEntity = b[k];
183
184 // initialize max and min
185 for (unsigned j = 0; j < nDim; ++j) {
186 min_corner[j] = +1.0e16;
187 max_corner[j] = -1.0e16;
188 }
189
190 stk::mesh::Entity const* entity_node_rels =
191 fromBulkData_.begin_nodes(theEntity);
192 int num_entity_nodes = fromBulkData_.num_nodes(theEntity);
193 for (int ni = 0; ni < num_entity_nodes; ++ni) {
194 stk::mesh::Entity node = entity_node_rels[ni];
195
196 double* coords = stk::mesh::field_data(*fromcoordinates_, node);
197 for (unsigned j = 0; j < nDim; ++j) {
198 min_corner[j] = std::min(min_corner[j], coords[j]);
199 max_corner[j] = std::max(max_corner[j], coords[j]);
200 }
201 }
202
203 // setup ident
204 FromMesh::EntityProc theIdent(
205 fromBulkData_.entity_key(theEntity), fromBulkData_.parallel_rank());
206
207 v.push_back(BoundingBox(Box(min_corner, max_corner), theIdent));
208 }
209 }
210 std::sort(v.begin(), v.end(), BoundingBoxCompare());
211 }

213 void update_ghosting(const EntityProcVec& entity_keys)
214 {
215 ghosting_map_.resize(entity_keys.size());
216 for (size_t i = 0; i < entity_keys.size(); ++i) {
217 // convert from EntityProc based on EntityKey to EntityProc based on raw
218 // Entity.
219 const EntityProc& key_proc = entity_keys[i];
220 const stk::mesh::EntityKey key = key_proc.id();
221 const unsigned proc = key_proc.proc();
222 const stk::mesh::Entity e = entity(key);
223 const stk::mesh::EntityProc ep(e, proc);
224 ghosting_map_[i] = ep;
225 }
226
227 unsigned s = !ghosting_map_.empty();
228 stk::all_reduce(comm_, stk::ReduceSum<1>(&s));
229
230 if (s) {
231 std::sort(ghosting_map_.begin(), ghosting_map_.end());
232 stk::mesh::EntityProcVec::iterator del =
233 std::unique(ghosting_map_.begin(), ghosting_map_.end());
234 ghosting_map_.resize(std::distance(ghosting_map_.begin(), del));
235
236 std::string theGhostName = "nalu_transfer_ghosting";
237 for (unsigned i = 0; i != fromFieldVec_.size(); ++i)
238 theGhostName += "_" + fromFieldVec_[i]->name();
239 ghosting_ = &fromBulkData_.create_ghosting(theGhostName);
240 fromBulkData_.change_ghosting(*ghosting_, ghosting_map_);
241 mesh_modified_ = true;
242 }
243 }

245 void update_values()
246 {
247 if (ghosting_) {
248 std::vector<const stk::mesh::FieldBase*> fields(
249 fromFieldVec_.begin(), fromFieldVec_.end());
250 if (mesh_modified_) {
251 // Copy coordinates to the newly ghosted nodes
252 mesh_modified_ = false;
253 fields.push_back(fromcoordinates_);
254 }
255 stk::mesh::communicate_field_data(*ghosting_, fields);
256 stk::mesh::copy_owned_to_shared(fromBulkData_, fields);
257 }
