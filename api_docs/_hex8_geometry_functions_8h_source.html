34 template <typename ViewType>
35 KOKKOS_FUNCTION void
36 quad_area_by_triangulation(
37  int ics, const typename ViewType::value_type areacoords[4][3], ViewType& area)
38 {
47  using ftype = typename ViewType::value_type;
48 
49  area(ics, 0) = 0.0;
50  area(ics, 1) = 0.0;
51  area(ics, 2) = 0.0;
52 
53  const ftype xmid[3] = {
54  0.25 * (areacoords[0][0] + areacoords[1][0] + areacoords[2][0] +
55  areacoords[3][0]),
56  0.25 * (areacoords[0][1] + areacoords[1][1] + areacoords[2][1] +
57  areacoords[3][1]),
58  0.25 * (areacoords[0][2] + areacoords[1][2] + areacoords[2][2] +
59  areacoords[3][2])};
60 
61  ftype r1[3] = {
62  areacoords[0][0] - xmid[0], areacoords[0][1] - xmid[1],
63  areacoords[0][2] - xmid[2]};
64  for (int itriangle = 0; itriangle < 4; ++itriangle) {
65  const int t_index = (itriangle + 1) % 4;
66  const ftype r2[3] = {
67  areacoords[t_index][0] - xmid[0], areacoords[t_index][1] - xmid[1],
68  areacoords[t_index][2] - xmid[2]};
69 
70  area(ics, 0) += r1[1] * r2[2] - r2[1] * r1[2];
71  area(ics, 1) += r1[2] * r2[0] - r2[2] * r1[0];
72  area(ics, 2) += r1[0] * r2[1] - r2[0] * r1[1];
73 
74  r1[0] = r2[0];
75  r1[1] = r2[1];
76  r1[2] = r2[2];
77  }
78  area(ics, 0) *= 0.5;
79  area(ics, 1) *= 0.5;
80  area(ics, 2) *= 0.5;
81 }

83 template <typename RealType>
84 KOKKOS_FUNCTION RealType
85 hex_volume_grandy(RealType scvcoords[8][3])
86 {
94  constexpr int nTri = 24;
95  constexpr int dim = 3;
96 
97  constexpr int nNodes = 8;
98  constexpr int nFaces = 6;
99  constexpr int npv = nNodes + nFaces;
100 
101  RealType coordv[npv][dim];
102 
103  // copy coordinates
104  for (int n = 0; n < nNodes; ++n) {
105  coordv[n][0] = scvcoords[n][0];
106  coordv[n][1] = scvcoords[n][1];
107  coordv[n][2] = scvcoords[n][2];
108  }
109 
110  constexpr int nodesPerFace = 4;
111  constexpr int face_nodes[nFaces][nodesPerFace] = {{0, 3, 2, 1}, {4, 5, 6, 7},
112  {0, 1, 5, 4}, {2, 3, 7, 6},
113  {1, 2, 6, 5}, {0, 4, 3, 7}};
114 
115  // append face midpoint coordinates
116  for (int k = 0; k < nFaces; ++k) {
117  const int coordIndex = k + nNodes;
118  for (int d = 0; d < dim; ++d) {
119  coordv[coordIndex][d] =
120  0.25 * (coordv[face_nodes[k][0]][d] + coordv[face_nodes[k][1]][d] +
121  coordv[face_nodes[k][2]][d] + coordv[face_nodes[k][3]][d]);
122  }
123  }
124 
125  constexpr int triangular_facets[nTri][3] = {
126  {0, 8, 1}, {8, 2, 1}, {3, 2, 8}, {3, 8, 0}, {6, 9, 5}, {7, 9, 6},
127  {4, 9, 7}, {4, 5, 9}, {10, 0, 1}, {5, 10, 1}, {4, 10, 5}, {4, 0, 10},
128  {7, 6, 11}, {6, 2, 11}, {2, 3, 11}, {3, 7, 11}, {6, 12, 2}, {5, 12, 6},
129  {5, 1, 12}, {1, 2, 12}, {0, 4, 13}, {4, 7, 13}, {7, 3, 13}, {3, 0, 13}};
130 
131  RealType volume = 0.0;
132  for (int k = 0; k < nTri; ++k) {
133  const int p = triangular_facets[k][0];
134  const int q = triangular_facets[k][1];
135  const int r = triangular_facets[k][2];
136 
137  const RealType triFaceMid[3] = {
138  coordv[p][0] + coordv[q][0] + coordv[r][0],
139  coordv[p][1] + coordv[q][1] + coordv[r][1],
140  coordv[p][2] + coordv[q][2] + coordv[r][2]};
141 
142  enum { XC = 0, YC = 1, ZC = 2 };
143  RealType dxv[3];
144 
145  dxv[0] = (coordv[q][YC] - coordv[p][YC]) * (coordv[r][ZC] - coordv[p][ZC]) -
146  (coordv[r][YC] - coordv[p][YC]) * (coordv[q][ZC] - coordv[p][ZC]);
147 
148  dxv[1] = (coordv[r][XC] - coordv[p][XC]) * (coordv[q][ZC] - coordv[p][ZC]) -
149  (coordv[q][XC] - coordv[p][XC]) * (coordv[r][ZC] - coordv[p][ZC]);
150 
151  dxv[2] = (coordv[q][XC] - coordv[p][XC]) * (coordv[r][YC] - coordv[p][YC]) -
152  (coordv[r][XC] - coordv[p][XC]) * (coordv[q][YC] - coordv[p][YC]);
153 
154  volume +=
155  triFaceMid[0] * dxv[0] + triFaceMid[1] * dxv[1] + triFaceMid[2] * dxv[2];
156  }
157  volume /= RealType(18.0);
158  return volume;
159 }

203  }
204 
205  for (int k = 0; k < 3; ++k) {

207  scvcoords[6][k] + scvcoords[7][k]);
208  }
209 
210  for (int k = 0; k < 3; ++k) {
211  coordv[12][k] = 0.25 * (scvcoords[1][k] + scvcoords[2][k] +
212  scvcoords[6][k] + scvcoords[5][k]);
213  }
214 
215  for (int k = 0; k < 3; ++k) {
216  coordv[13][k] = 0.25 * (scvcoords[0][k] + scvcoords[3][k] +
217  scvcoords[7][k] + scvcoords[4][k]);
218  }

220  constexpr int triangular_facets[nTri][3] = {
221  {0, 8, 1}, {8, 2, 1}, {3, 2, 8}, {3, 8, 0}, {6, 9, 5}, {7, 9, 6},
222  {4, 9, 7}, {4, 5, 9}, {10, 0, 1}, {5, 10, 1}, {4, 10, 5}, {4, 0, 10},
223  {7, 6, 11}, {6, 2, 11}, {2, 3, 11}, {3, 7, 11}, {6, 12, 2}, {5, 12, 6},
224  {5, 1, 12}, {1, 2, 12}, {0, 4, 13}, {4, 7, 13}, {7, 3, 13}, {3, 0, 13}};
225 
226  RealType volume = 0.0;
227  for (int k = 0; k < nTri; ++k) {
228  const int p = triangular_facets[k][0];
229  const int q = triangular_facets[k][1];
230  const int r = triangular_facets[k][2];
231 
232  const RealType triFaceMid[3] = {
233  coordv[p][0] + coordv[q][0] + coordv[r][0],
234  coordv[p][1] + coordv[q][1] + coordv[r][1],
235  coordv[p][2] + coordv[q][2] + coordv[r][2]};
236 
237  enum { XC = 0, YC = 1, ZC = 2 };
238  RealType dxv[3];
239 
240  dxv[0] = (coordv[q][YC] - coordv[p][YC]) * (coordv[r][ZC] - coordv[p][ZC]) -
241  (coordv[r][YC] - coordv[p][YC]) * (coordv[q][ZC] - coordv[p][ZC]);
242 
243  dxv[1] = (coordv[r][XC] - coordv[p][XC]) * (coordv[q][ZC] - coordv[p][ZC]) -
244  (coordv[q][XC] - coordv[p][XC]) * (coordv[r][ZC] - coordv[p][ZC]);
245 
246  dxv[2] = (coordv[q][XC] - coordv[p][XC]) * (coordv[r][YC] - coordv[p][YC]) -
247  (coordv[r][XC] - coordv[p][XC]) * (coordv[q][YC] - coordv[p][YC]);
248 
249  volume +=
250  triFaceMid[0] * dxv[0] + triFaceMid[1] * dxv[1] + triFaceMid[2] * dxv[2];
251  }
252  volume /= RealType(18.0);
253  return volume;
254 }

256 template <typename CoordViewType>
257 KOKKOS_FUNCTION void
258 subdivide_hex_8(
259  CoordViewType coords, typename CoordViewType::value_type coordv[27][3])
260 {
265  constexpr int numBaseNodes = 8;
266 
267  for (int n = 0; n < numBaseNodes; ++n) {
268  coordv[n][0] = coords(n, 0);
269  coordv[n][1] = coords(n, 1);
270  coordv[n][2] = coords(n, 2);
271  }
272 
273  // Face-by-face ordering for the subdivided hex. This is different than what
274  // is done for a multilinear Hex27 element, which has equivalent nodal
275  // locations.
276  for (int d = 0; d < 3; ++d) {
277  // Face 0
278  coordv[8][d] = 0.5 * (coords(0, d) + coords(1, d)); // edge 1
279  coordv[9][d] = 0.5 * (coords(1, d) + coords(2, d)); // edge 2
280  coordv[10][d] = 0.5 * (coords(2, d) + coords(3, d)); // edge 3
281  coordv[11][d] = 0.5 * (coords(3, d) + coords(0, d)); // edge 4
282 
283  coordv[12][d] =
284  0.25 * (coords(0, d) + coords(1, d) + coords(2, d) + coords(3, d));
285 
286  // Face 1
287  coordv[13][d] = 0.5 * (coords(4, d) + coords(5, d)); // edge 5
288  coordv[14][d] = 0.5 * (coords(5, d) + coords(6, d)); // edge 6
289  coordv[15][d] = 0.5 * (coords(6, d) + coords(7, d)); // edge 7
290  coordv[16][d] = 0.5 * (coords(7, d) + coords(4, d)); // edge 8
291 
292  coordv[17][d] =
293  0.25 * (coords(4, d) + coords(5, d) + coords(6, d) + coords(7, d));
294 
295  // Face 2
296  coordv[18][d] = 0.5 * (coords(1, d) + coords(5, d)); // edge 9
297  coordv[19][d] = 0.5 * (coords(0, d) + coords(4, d)); // edge 10
298 
299  coordv[20][d] =
300  0.25 * (coords(0, d) + coords(1, d) + coords(4, d) + coords(5, d));
301 
302  // Face 3
303  coordv[21][d] = 0.5 * (coords(3, d) + coords(7, d)); // edge 11
304  coordv[22][d] = 0.5 * (coords(2, d) + coords(6, d)); // edge 12
305 
306  coordv[23][d] =
307  0.25 * (coords(2, d) + coords(3, d) + coords(6, d) + coords(7, d));
308 
309  // Face 4
310  coordv[24][d] =
311  0.25 * (coords(1, d) + coords(2, d) + coords(5, d) + coords(6, d));
312 
313  // Face 5
314  coordv[25][d] =
315  0.25 * (coords(0, d) + coords(3, d) + coords(4, d) + coords(7, d));
316 
317  // Volume centroid
318  coordv[26][d] = 0.;
319  for (int n = 0; n < 8; ++n) {
320  coordv[26][d] += coords(n, d);
321  }
322  coordv[26][d] *= 0.125;
323  }
324 }

336  interpj[1] * interpk[0] * base_box[d][2] -

339  interpj[0] * interpk[1] * base_box[d][5] +
340  interpj[1] * interpk[1] * base_box[d][6] -
341  interpj[1] * interpk[1] * base_box[d][7]) *
342  0.5;
343 }

412  interpi[1] * interpj[1] * base_box[di][2] -

418  0.5;

459 template <int dir, typename Scalar>
460 KOKKOS_FORCEINLINE_FUNCTION void
461 areav_from_jacobian_t(const Scalar jact[3][3], Scalar area[3])
462 {
463  constexpr int orth_comp_1 = (dir == XH) ? ZH : (dir == YH) ? XH : YH;
464  constexpr int orth_comp_2 = (dir == XH) ? YH : (dir == YH) ? ZH : XH;
465  area[XH] = jact[YH][orth_comp_1] * jact[ZH][orth_comp_2] -
466  jact[ZH][orth_comp_1] * jact[YH][orth_comp_2];
467  area[YH] = jact[ZH][orth_comp_1] * jact[XH][orth_comp_2] -
468  jact[XH][orth_comp_1] * jact[ZH][orth_comp_2];
469  area[ZH] = jact[XH][orth_comp_1] * jact[YH][orth_comp_2] -
470  jact[YH][orth_comp_1] * jact[XH][orth_comp_2];
471 }

473 template <typename Scalar>
474 KOKKOS_FORCEINLINE_FUNCTION void
475 hex_areav_x(
476  const Scalar base_box[3][8],
477  const Scalar interpi[2],
478  const Scalar interpj[2],
479  const Scalar interpk[2],
480  Scalar area[3])

483  const auto dx_ds2 = jacobian_component_yh(0, base_box, interpi, interpk);
484 
485  const auto dy_ds1 = jacobian_component_zh(1, base_box, interpi, interpj);
486  const auto dy_ds2 = jacobian_component_yh(1, base_box, interpi, interpk);

489  const auto dz_ds2 = jacobian_component_yh(2, base_box, interpi, interpk);

491  area[0] = dy_ds1 * dz_ds2 - dz_ds1 * dy_ds2;
492  area[1] = dz_ds1 * dx_ds2 - dx_ds1 * dz_ds2;
493  area[2] = dx_ds1 * dy_ds2 - dy_ds1 * dx_ds2;
