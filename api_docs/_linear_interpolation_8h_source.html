45template <typename T>
46inline typename InterpTraits<T>::index_type
47check_bounds(const Array1D<T>& xinp, const T& x)
48{
49 auto sz = xinp.size();
50
51 if (sz < 2) {
52 throw std::runtime_error(
53 "Interpolation table contains less than 2 entries.");
54 }
55
56 if (x < xinp[0]) {
57 return std::make_pair(OutOfBounds::LOWLIM, 0);
58 } else if (x > xinp[sz - 1]) {
59 return std::make_pair(OutOfBounds::UPLIM, sz - 1);
60 } else {
61 return std::make_pair(OutOfBounds::VALID, 0);
62 }
63}

73template <typename T>
74inline typename InterpTraits<T>::index_type
75find_index(const Array1D<T>& xinp, const T& x)
76{
77 auto idx = check_bounds(xinp, x);
78 if (idx.first == OutOfBounds::UPLIM || idx.first == OutOfBounds::LOWLIM)
79 return idx;
80
81 auto sz = xinp.size();
82 for (size_t i = 1; i < sz; i++) {
83 if (x <= xinp[i]) {
84 idx.second = i - 1;
85 break;
86 }
87 }
88 return idx;
89}

94template <typename T>
95void
96linear_interp(
97 const Array1D<T>& xinp,
98 const Array1D<T>& yinp,
99 const T& xout,
100 T& yout,
101 OutOfBounds::OobAction oob = OutOfBounds::CLAMP)
102{
103 auto idx = find_index(xinp, xout);
104
105 switch (idx.first) {
106 case OutOfBounds::LOWLIM:
107 case OutOfBounds::UPLIM: {
108 switch (oob) {
109 case OutOfBounds::ERROR:
110 throw std::runtime_error("Out of bounds error in interpolation");
111// Avoid nvcc unreachable statement warnings
112#ifndef __CUDACC__
113 break;
114#endif
115
116 case OutOfBounds::WARN:
117 std::cout
118 << std::endl
119 << "WARNING: Out of bound values encountered during interpolation"
120 << std::endl;
121 // Repeated code to avoid fallthrough warnings
122 yout = yinp[idx.second];
123 break;
124
125 case OutOfBounds::CLAMP:
126 yout = yinp[idx.second];
127 break;
128
129 case OutOfBounds::EXTRAPOLATE: {
130 auto ii = idx.second;
131 if ((ii + 1) == xinp.size())
132 --ii;
133
134 yout = yinp[ii] + (yinp[ii + 1] - yinp[ii]) / (xinp[ii + 1] - xinp[ii]) *
135 (xout - xinp[ii]);
136 break;
137 }
138 }
139 break;
140 }
141 case OutOfBounds::VALID:
142 auto j = idx.second;
143 T fac = (xout - xinp[j]) / (xinp[j + 1] - xinp[j]);
144 yout = (static_cast<T>(1.0) - fac) * yinp[j] + fac * yinp[j + 1];
145 break;
146 }
147}
