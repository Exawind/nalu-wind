117template <typename T>
118class NGPKernel : public Kernel
119{
120public:
121 KOKKOS_DEFAULTED_FUNCTION
122 NGPKernel() = default;
123
124 // Implementation note
125 //
126 // The destructor does not free the deviceCopy_ instance. This is done to
127 // eliminate the warnings issued when compiling with nvcc for GPU builds.
128 // Instead the `deviceCopy_` is freed by explicitly calling `free_on_device`
129 // from sierra::nalu::Algorithm::~Algorithm() before freeing the host pointers
130 // stored in `activeKernels_`
131 KOKKOS_DEFAULTED_FUNCTION virtual ~NGPKernel() = default;
132
133 virtual Kernel* create_on_device() final
134 {
135 free_on_device();
136 deviceCopy_ = nalu_ngp::create<T>(*dynamic_cast<T*>(this));
137 return deviceCopy_;
138 }
139
140 virtual void free_on_device() final
141 {
142 if (deviceCopy_ != nullptr) {
143 nalu_ngp::destroy<T>(dynamic_cast<T*>(deviceCopy_));
144 deviceCopy_ = nullptr;
145 }
146 }
147
148 T* device_copy() const { return deviceCopy_; }
149
150protected:
151 T* deviceCopy_{nullptr};
152};
