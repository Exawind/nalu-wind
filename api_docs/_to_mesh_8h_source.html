57 typedef std::vector<std::pair<std::string, std::string>> PairNames;
58 std::vector<const stk::mesh::FieldBase*> get_fields(
59 const stk::mesh::MetaData& toMetaData, const PairNames& VarPairName)
60 {
61 // will want to check that all is well with field registration
62 bool allFieldsAreFine = true;
63 std::vector<const stk::mesh::FieldBase*> toFieldVec;
64 // provide field names
65 for (PairNames::const_iterator i = VarPairName.begin();
66 i != VarPairName.end(); ++i) {
67 const std::string& fieldName = i->second;
68 const stk::mesh::FieldBase* tofield =
69 stk::mesh::get_field_by_name(fieldName, toMetaData);
70 if (NULL == tofield) {
71 allFieldsAreFine = false;
72 NaluEnv::self().naluOutputP0()
73 << "Xfer::ToMesh:Error field: " << fieldName
74 << " has not been registered anywhere within the ToRealm: "
75 << toRealm_.name() << std::endl;
76 } else {
77 // always push back; check for errors below
78 toFieldVec.push_back(tofield);
79
80 // check that the field is defined on **all** parts
81 stk::mesh::Selector fieldSelector = stk::mesh::selectField(*tofield);
82 for (size_t k = 0; k < toPartVec_.size(); ++k) {
83
84 const stk::mesh::BucketVector& partBuckets = toBulkData_.get_buckets(
85 stk::topology::NODE_RANK, stk::mesh::Selector(*toPartVec_[k]));
86
87 bool fieldIsFine = true;
88 for (stk::mesh::BucketVector::const_iterator ib = partBuckets.begin();
89 ib != partBuckets.end(); ++ib) {
90 stk::mesh::Bucket& b = **ib;
91 fieldIsFine &= fieldSelector(b);
92 }
93
94 // local check to make sure that the field is somewhere (delay the
95 // throw)
96 if (!fieldIsFine) {
97 NaluEnv::self().naluOutputP0()
98 << "Xfer::ToMesh:Error field: " << tofield->name()
99 << " is not registered on part: " << toPartVec_[k]->name()
100 << std::endl;
101 allFieldsAreFine = false;
102 }
103 }
104 }
105 }
106
107 // final error check; only return when all is well
108 if (allFieldsAreFine) {
109 return toFieldVec;
110 } else {
111 throw std::runtime_error(
112 "Xfer::ToMesh:Error field registration on "
113 "desired parts of the mesh is not complete");
114 }
115 }

152 void bounding_boxes(std::vector<BoundingBox>& v) const
153 {
154
155 const unsigned spatial_dimension = toMetaData_.spatial_dimension();
156
157 Point center;
158
159 stk::mesh::Selector s_locally_owned_union =
160 toMetaData_.locally_owned_part() & stk::mesh::selectUnion(toPartVec_);
161
162 stk::mesh::BucketVector const& node_buckets =
163 toBulkData_.get_buckets(stk::topology::NODE_RANK, s_locally_owned_union);
164 for (stk::mesh::BucketVector::const_iterator ib = node_buckets.begin();
165 ib != node_buckets.end(); ++ib) {
166 stk::mesh::Bucket& b = **ib;
167
168 const stk::mesh::Bucket::size_type length = b.size();
169
170 for (stk::mesh::Bucket::size_type k = 0; k < length; ++k) {
171
172 // get node
173 stk::mesh::Entity node = b[k];
174
175 double* coord = stk::mesh::field_data(*tocoordinates_, node);
176 for (unsigned i = 0; i < spatial_dimension; ++i) {
177 center[i] = coord[i];
178 }
179
180 // setup ident
181 ToMesh::EntityProc theIdent(
182 toBulkData_.entity_key(node), toBulkData_.parallel_rank());
183
184 ToMesh::BoundingBox theBox(Sphere(center, radius_), theIdent);
185 v.push_back(theBox);
186 }
187 }
188 std::sort(v.begin(), v.end(), BoundingBoxCompare());
189 }
