23 void
24 diffusion_xhat_contrib(
25  const CVFEMOperators<poly_order, Scalar>& ops,
26  const scs_vector_view<poly_order, Scalar>& metric,
27  matrix_view<poly_order, Scalar>& lhs,
28  bool reduced_sens)
29 {
30  constexpr int n1D = poly_order + 1;
31  const auto& mat = ops.mat_;
32  const auto& nodalWeights =
33  (reduced_sens) ? mat.lumpedNodalWeights : mat.nodalWeights;
34 
35  for (int n = 0; n < n1D; ++n) {
36  for (int m = 0; m < n1D; ++m) {
37  constexpr int l_minus = 0;
38 
39  auto rowIndexMinus = idx<n1D>(n, m, l_minus);
40  for (int k = 0; k < n1D; ++k) {
41  const Scalar Wnk = nodalWeights(n, k);
42 
43  for (int j = 0; j < n1D; ++j) {
44  const Scalar WnkWmj = Wnk * nodalWeights(m, j);
45  const Scalar orth = WnkWmj * metric(XH, k, j, l_minus, XH);
46 
47  Scalar non_orth_y = 0.0;
48  Scalar non_orth_z = 0.0;
49  for (int q = 0; q < n1D; ++q) {
50  non_orth_y += nodalWeights(n, k) * nodalWeights(m, q) *
51  mat.nodalDeriv(q, j) * metric(XH, k, q, l_minus, YH);
52  non_orth_z += nodalWeights(n, q) * nodalWeights(m, j) *
53  mat.nodalDeriv(q, k) * metric(XH, q, j, l_minus, ZH);
54  }
55 
56  for (int i = 0; i < n1D; ++i) {
57  lhs(rowIndexMinus, idx<n1D>(k, j, i)) +=
58  orth * mat.scsDeriv(l_minus, i) +
59  mat.scsInterp(l_minus, i) * (non_orth_y + non_orth_z);
60  }
61  }
62  }
63 
64  for (int l = 1; l < n1D - 1; ++l) {
65  auto rowIndex = idx<n1D>(n, m, l);
66  for (int k = 0; k < n1D; ++k) {
67  const Scalar Wnk = nodalWeights(n, k);
68  for (int j = 0; j < n1D; ++j) {
69  const Scalar WnkWmj = Wnk * nodalWeights(m, j);
70  const Scalar orthm1 = WnkWmj * metric(XH, k, j, l - 1, XH);
71  const Scalar orthp0 = WnkWmj * metric(XH, k, j, l + 0, XH);
72 
73  Scalar non_orth_y_m1 = 0.0;
74  Scalar non_orth_y_p0 = 0.0;
75  Scalar non_orth_z_m1 = 0.0;
76  Scalar non_orth_z_p0 = 0.0;
77 
78  for (int q = 0; q < n1D; ++q) {
79  const Scalar wy =
80  nodalWeights(n, k) * nodalWeights(m, q) * mat.nodalDeriv(q, j);
81  non_orth_y_m1 += wy * metric(XH, k, q, l - 1, YH);
82  non_orth_y_p0 += wy * metric(XH, k, q, l + 0, YH);
83 
84  const Scalar wz =
85  nodalWeights(n, q) * nodalWeights(m, j) * mat.nodalDeriv(q, k);
86  non_orth_z_m1 += wz * metric(XH, q, j, l - 1, ZH);
87  non_orth_z_p0 += wz * metric(XH, q, j, l + 0, ZH);
88  }
89 
90  for (int i = 0; i < n1D; ++i) {
91  const Scalar integrated_flux_m =
92  orthm1 * mat.scsDeriv(l - 1, i) +
93  mat.scsInterp(l - 1, i) * (non_orth_y_m1 + non_orth_z_m1);
94  const Scalar integrated_flux_p =
95  orthp0 * mat.scsDeriv(l + 0, i) +
96  mat.scsInterp(l + 0, i) * (non_orth_y_p0 + non_orth_z_p0);
97  lhs(rowIndex, idx<n1D>(k, j, i)) +=
98  integrated_flux_p - integrated_flux_m;
99  }
100  }
101  }
102  }
103 
104  constexpr int l_plus = n1D - 1;
105  auto rowIndexPlus = idx<n1D>(n, m, l_plus);
106  for (int k = 0; k < n1D; ++k) {
107  const Scalar Wnk = nodalWeights(n, k);
108  for (int j = 0; j < n1D; ++j) {
109  const Scalar WnkWmj = Wnk * nodalWeights(m, j);
110  const Scalar orth = WnkWmj * metric(XH, k, j, l_plus - 1, XH);
111 
112  Scalar non_orth_y = 0.0;
113  Scalar non_orth_z = 0.0;
114  for (int q = 0; q < n1D; ++q) {
115  non_orth_y += nodalWeights(n, k) * nodalWeights(m, q) *
116  mat.nodalDeriv(q, j) *
117  metric(XH, k, q, l_plus - 1, YH);
118  non_orth_z += nodalWeights(n, q) * nodalWeights(m, j) *
119  mat.nodalDeriv(q, k) *
120  metric(XH, q, j, l_plus - 1, ZH);
121  }
122 
123  for (int i = 0; i < n1D; ++i) {
124  lhs(rowIndexPlus, idx<n1D>(k, j, i)) -=
125  orth * mat.scsDeriv(l_plus - 1, i) +
126  mat.scsInterp(l_plus - 1, i) * (non_orth_y + non_orth_z);
127  }
128  }
129  }
130  }
131  }

367 template <int poly_order, typename Scalar>
368 void
369 scalar_diffusion_rhs(
370  CVFEMOperators<poly_order, Scalar>& ops,
371  const scs_vector_view<poly_order, Scalar>& metric,
372  const nodal_scalar_view<poly_order, Scalar>& scalar,
373  nodal_scalar_view<poly_order, Scalar>& rhs)
374 {
375  constexpr int n1D = poly_order + 1;
376  constexpr int nscs = poly_order;
377 
378  nodal_vector_workview<poly_order, Scalar> l_grad_phi;
379  auto& grad_phi_scs = l_grad_phi.view();
380 
381  nodal_scalar_workview<poly_order, Scalar> work_integrand;
382  auto& integrand = work_integrand.view();
383 
384  ops.scs_xhat_grad(scalar, grad_phi_scs);
385  for (int k = 0; k < n1D; ++k) {
386  for (int j = 0; j < n1D; ++j) {
387  for (int i = 0; i < nscs; ++i) {
388  integrand(k, j, i) =
389  metric(XH, k, j, i, XH) * grad_phi_scs(k, j, i, XH) +
390  metric(XH, k, j, i, YH) * grad_phi_scs(k, j, i, YH) +
391  metric(XH, k, j, i, ZH) * grad_phi_scs(k, j, i, ZH);
392  }
393  }
394  }
395  ops.integrate_and_diff_xhat(integrand, rhs);
396 
397  ops.scs_yhat_grad(scalar, grad_phi_scs);
398  for (int k = 0; k < n1D; ++k) {
399  for (int j = 0; j < nscs; ++j) {
400  for (int i = 0; i < n1D; ++i) {
401  integrand(k, j, i) =
402  metric(YH, k, j, i, XH) * grad_phi_scs(k, j, i, XH) +
403  metric(YH, k, j, i, YH) * grad_phi_scs(k, j, i, YH) +
404  metric(YH, k, j, i, ZH) * grad_phi_scs(k, j, i, ZH);
405  }
406  }
407  }
408  ops.integrate_and_diff_yhat(integrand, rhs);
409 
410  ops.scs_zhat_grad(scalar, grad_phi_scs);
411  for (int k = 0; k < nscs; ++k) {
412  for (int j = 0; j < n1D; ++j) {
413  for (int i = 0; i < n1D; ++i) {
414  integrand(k, j, i) =
415  metric(ZH, k, j, i, XH) * grad_phi_scs(k, j, i, XH) +
416  metric(ZH, k, j, i, YH) * grad_phi_scs(k, j, i, YH) +
417  metric(ZH, k, j, i, ZH) * grad_phi_scs(k, j, i, ZH);
418  }
419  }
420  }

422 }
