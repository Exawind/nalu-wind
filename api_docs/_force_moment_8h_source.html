19void KOKKOS_INLINE_FUNCTION
20computeEffForceMoment(
21 double* forceCFD, double* xyzCFD, double* forceMomOF, double* xyzOF)
22{
23
24 const int ndim = 3; // I don't see this ever being used in other situations
25 for (int j = 0; j < ndim; j++)
26 forceMomOF[j] += forceCFD[j];
27 forceMomOF[3] +=
28 (xyzCFD[1] - xyzOF[1]) * forceCFD[2] - (xyzCFD[2] - xyzOF[2]) * forceCFD[1];
29 forceMomOF[4] +=
30 (xyzCFD[2] - xyzOF[2]) * forceCFD[0] - (xyzCFD[0] - xyzOF[0]) * forceCFD[2];
31 forceMomOF[5] +=
32 (xyzCFD[0] - xyzOF[0]) * forceCFD[1] - (xyzCFD[1] - xyzOF[1]) * forceCFD[0];
33}

37void KOKKOS_INLINE_FUNCTION
38splitForceMoment(

40 double interpFac,
41 double* leftForceMoment,
42 double* rightForceMoment)
43{
44 for (int i = 0; i < 6; i++) {
45 leftForceMoment[i] += (1.0 - interpFac) * totForceMoment[i];
