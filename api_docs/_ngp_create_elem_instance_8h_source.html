69template <typename BaseType, template <typename> class T, typename... Args>
70BaseType*
71create_elem_algorithm(const stk::topology topo, Args&&... args)
72{
73 switch (topo.value()) {
74 case stk::topology::HEX_8:
75 return new T<AlgTraitsHex8>(std::forward<Args>(args)...);
76 case stk::topology::TET_4:
77 return new T<AlgTraitsTet4>(std::forward<Args>(args)...);
78 case stk::topology::PYRAMID_5:
79 return new T<AlgTraitsPyr5>(std::forward<Args>(args)...);
80 case stk::topology::WEDGE_6:
81 return new T<AlgTraitsWed6>(std::forward<Args>(args)...);
82 case stk::topology::QUAD_4_2D:
83 return new T<AlgTraitsQuad4_2D>(std::forward<Args>(args)...);
84 case stk::topology::TRI_3_2D:
85 return new T<AlgTraitsTri3_2D>(std::forward<Args>(args)...);
86 default:
87 throw std::runtime_error(
88 "NGP elem algorithm not implemented for " + topo.name());
89 }
90}

114template <typename BaseType, template <typename> class T, typename... Args>
115BaseType*
116create_face_elem_algorithm(
117 const stk::topology faceTopo, const stk::topology elemTopo, Args&&... args)
118{
119 switch (faceTopo) {
120 case stk::topology::QUAD_4:
121 switch (elemTopo) {
122 case stk::topology::HEX_8:
123 return new T<AlgTraitsQuad4Hex8>(std::forward<Args>(args)...);
124 case stk::topology::PYRAMID_5:
125 return new T<AlgTraitsQuad4Pyr5>(std::forward<Args>(args)...);
126 case stk::topology::WEDGE_6:
127 return new T<AlgTraitsQuad4Wed6>(std::forward<Args>(args)...);
128
129 default:
130 throw std::runtime_error(
131 "NGP face_elem algorithm not implemented for QUAD_4 and " +
132 elemTopo.name() + " pair.");
133 }
134
135 case stk::topology::TRI_3:
136 switch (elemTopo) {
137 case stk::topology::TET_4:
138 return new T<AlgTraitsTri3Tet4>(std::forward<Args>(args)...);
139 case stk::topology::PYRAMID_5:
140 return new T<AlgTraitsTri3Pyr5>(std::forward<Args>(args)...);
141 case stk::topology::WEDGE_6:
142 return new T<AlgTraitsTri3Wed6>(std::forward<Args>(args)...);
143 default:
144 throw std::runtime_error(
145 "NGP face_elem algorithm is not implemented for TRI_3 and " +
146 elemTopo.name() + " pair.");
147 }
148
149 case stk::topology::LINE_2:
150 switch (elemTopo) {
151 case stk::topology::QUAD_4_2D:
152 return new T<AlgTraitsEdge2DQuad42D>(std::forward<Args>(args)...);
153 case stk::topology::TRI_3_2D:
154 return new T<AlgTraitsEdge2DTri32D>(std::forward<Args>(args)...);
155 default:
156 throw std::runtime_error(
157 "NGP face_elem algorithm is not implemented for EDGE_2D and " +
158 elemTopo.name() + " pair.");
159 }
160
161 default:
162 throw std::runtime_error(
163 "NGP face_elem algorithm not implemented " + faceTopo.name());
164 }
165}
