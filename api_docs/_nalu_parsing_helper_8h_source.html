28  static void emit(YAML::Emitter& emout, const YAML::Node& node)
29  {
30  // recursive depth first
31  YAML::NodeType::value type = node.Type();
32  std::string out;
33  switch (type) {
34  case YAML::NodeType::Scalar:
35  out = node.as<std::string>();
36  emout << out;
37  break;
38  case YAML::NodeType::Sequence:
39  emout << YAML::BeginSeq;
40  for (unsigned int i = 0; i < node.size(); ++i) {
41  const YAML::Node subnode = node[i];
42  emit(emout, subnode);
43  }
44  emout << YAML::EndSeq;
45  break;
46  case YAML::NodeType::Map:
47  emout << YAML::BeginMap;
48  for (YAML::const_iterator i = node.begin(); i != node.end(); ++i) {
49  const YAML::Node key = i->first;
50  const YAML::Node value = i->second;
51  out = key.as<std::string>();
52  emout << YAML::Key << out;
53  emout << YAML::Value;
54  emit(emout, value);
55  }
56  emout << YAML::EndMap;
57  break;
58  case YAML::NodeType::Null:
59  emout << " (empty) ";
60  break;
61  default:
62  std::cerr << "Warning: emit: unknown/unsupported node type" << std::endl;
63  break;
64  }
65  }

136  static void find_nodes_given_key(
137  const std::string& key,
138  const YAML::Node& node,
139  std::vector<const YAML::Node*>& result)
140  {
141  // recursive depth first
142  YAML::NodeType::value type = node.Type();
143  if (type != YAML::NodeType::Scalar && type != YAML::NodeType::Null) {
144  if (node[key])
145  result.push_back(&node);
146  }
147 
148  switch (type) {
149  case YAML::NodeType::Scalar:
150  break;
151  case YAML::NodeType::Sequence:
152  for (unsigned int i = 0; i < node.size(); ++i) {
153  const YAML::Node& subnode = node[i];
154  find_nodes_given_key(key, subnode, result);
155  }
156  break;
157  case YAML::NodeType::Map:
158  for (YAML::const_iterator i = node.begin(); i != node.end(); ++i) {
159  // const YAML::Node & key1 = i.first();

161  find_nodes_given_key(key, value, result);
