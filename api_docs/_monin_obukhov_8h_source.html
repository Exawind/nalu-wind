40template <typename T>
41KOKKOS_FORCEINLINE_FUNCTION T
42psim_unstable(const T& zeta, const T gamma = 16.0)
43{
44 static constexpr double half_pi = 1.5707963267948966;
45
46 // Actually (1.0 / phim)
47 const T phim = stk::math::pow((1.0 - gamma * zeta), 0.25);
48 const T term1 = (1.0 + phim * phim);
49 const T term2 = (1.0 + phim);
50 const T term3 = term2 * term2;
51 const T term4 = stk::math::log(0.125 * term1 * term3);
52 const T term5 = 2.0 * stk::math::atan(phim);
53
54 return (term4 - term5 + half_pi);
55}

57template <typename T>
58KOKKOS_FORCEINLINE_FUNCTION T
59psih_unstable(const T& zeta, const T gamma = 16.0)
60{
61 // Actually (1.0 / phih)
62 const T phih = stk::math::sqrt(1.0 - gamma * zeta);
63 return (2.0 * stk::math::log(0.5 * (1.0 + phih)));
64}
