60template <typename ScalarType>
61ScalarType
62ddot(const ScalarType* u, const ScalarType* v, int n)
63{
64 ScalarType val = 0.0;
65 for (int i = 0; i < n; ++i) {
66 val += u[i] * v[i];
67 }

71template <typename ScalarType>
72void
73cross3(const ScalarType* u, const ScalarType* v, ScalarType* cross)
74{
75 cross[0] = u[1] * v[2] - u[2] * v[1];
76 cross[1] = -(u[0] * v[2] - u[2] * v[0]);
77 cross[2] = u[0] * v[1] - u[1] * v[0];

80template <typename ScalarType>
81KOKKOS_FORCEINLINE_FUNCTION ScalarType
82determinant22(const ScalarType* mat)
83{
84 enum { XX = 0, XY = 1, YX = 2, YY = 3 };
85 return (mat[XX] * mat[YY] - mat[XY] * mat[YX]);
86}

88template <typename ScalarType>
89KOKKOS_FORCEINLINE_FUNCTION ScalarType
90determinant33(const ScalarType* mat)
91{
92 enum {
93 XX = 0,
94 XY = 1,
95 XZ = 2,
96 YX = 3,
97 YY = 4,
98 YZ = 5,
99 ZX = 6,
100 ZY = 7,
101 ZZ = 8
102 };
103 return (
104 mat[XX] * (mat[YY] * mat[ZZ] - mat[YZ] * mat[ZY]) -
105 mat[XY] * (mat[YX] * mat[ZZ] - mat[YZ] * mat[ZX]) +
106 mat[XZ] * (mat[YX] * mat[ZY] - mat[YY] * mat[ZX]));
107}

109template <typename ScalarType>
110KOKKOS_FORCEINLINE_FUNCTION void
111adjugate_matrix33(const ScalarType jact[3][3], ScalarType adjJac[3][3])
112{
113 adjJac[0][0] = jact[1][1] * jact[2][2] - jact[2][1] * jact[1][2];
114 adjJac[0][1] = jact[1][2] * jact[2][0] - jact[2][2] * jact[1][0];
115 adjJac[0][2] = jact[1][0] * jact[2][1] - jact[2][0] * jact[1][1];
116
117 adjJac[1][0] = jact[0][2] * jact[2][1] - jact[2][2] * jact[0][1];
118 adjJac[1][1] = jact[0][0] * jact[2][2] - jact[2][0] * jact[0][2];
119 adjJac[1][2] = jact[0][1] * jact[2][0] - jact[2][1] * jact[0][0];
120
121 adjJac[2][0] = jact[0][1] * jact[1][2] - jact[1][1] * jact[0][2];
122 adjJac[2][1] = jact[0][2] * jact[1][0] - jact[1][2] * jact[0][0];
123 adjJac[2][2] = jact[0][0] * jact[1][1] - jact[1][0] * jact[0][1];
124}

126template <typename ScalarType>
127KOKKOS_FORCEINLINE_FUNCTION void
128invert_matrix33(const ScalarType A[3][3], ScalarType Ainv[3][3])
129{
130 ScalarType inv_detj = 1.0 / determinant33(&A[0][0]);
131
132 Ainv[0][0] = inv_detj * (A[1][1] * A[2][2] - A[2][1] * A[1][2]);
133 Ainv[0][1] = inv_detj * (A[1][2] * A[2][0] - A[2][2] * A[1][0]);
134 Ainv[0][2] = inv_detj * (A[1][0] * A[2][1] - A[2][0] * A[1][1]);
135
136 Ainv[1][0] = inv_detj * (A[0][2] * A[2][1] - A[2][2] * A[0][1]);
137 Ainv[1][1] = inv_detj * (A[0][0] * A[2][2] - A[2][0] * A[0][2]);
138 Ainv[1][2] = inv_detj * (A[0][1] * A[2][0] - A[2][1] * A[0][0]);
139
140 Ainv[2][0] = inv_detj * (A[0][1] * A[1][2] - A[1][1] * A[0][2]);
141 Ainv[2][1] = inv_detj * (A[0][2] * A[1][0] - A[1][2] * A[0][0]);
142 Ainv[2][2] = inv_detj * (A[0][0] * A[1][1] - A[1][0] * A[0][1]);
143}

145template <typename ScalarType>
146KOKKOS_FORCEINLINE_FUNCTION void
147invert_matrix33(
148 const ScalarType* POINTER_RESTRICT A, ScalarType* POINTER_RESTRICT Ainv)
149{
150 enum {
151 XX = 0,
152 XY = 1,
153 XZ = 2,
154 YX = 3,
155 YY = 4,
156 YZ = 5,
157 ZX = 6,
158 ZY = 7,
159 ZZ = 8
160 };
161 ScalarType inv_detj = 1.0 / determinant33(A);
162
163 Ainv[XX] = inv_detj * (A[YY] * A[ZZ] - A[ZY] * A[YZ]);
164 Ainv[XY] = inv_detj * (A[YZ] * A[ZX] - A[ZZ] * A[YX]);
165 Ainv[XZ] = inv_detj * (A[YX] * A[ZY] - A[ZX] * A[YY]);
166 Ainv[YX] = inv_detj * (A[XZ] * A[ZY] - A[ZZ] * A[XY]);
167 Ainv[YY] = inv_detj * (A[XX] * A[ZZ] - A[ZX] * A[XZ]);
168 Ainv[YZ] = inv_detj * (A[XY] * A[ZX] - A[ZY] * A[XX]);
169 Ainv[ZX] = inv_detj * (A[XY] * A[YZ] - A[YY] * A[XZ]);
170 Ainv[ZY] = inv_detj * (A[XZ] * A[YZ] - A[YZ] * A[XX]);
171 Ainv[ZZ] = inv_detj * (A[XX] * A[YY] - A[YX] * A[XY]);
172}

174enum class SolveStatus { FAILED = 0, SUCCEEDED = 1 };

176template <typename ScalarType>
177KOKKOS_FORCEINLINE_FUNCTION SolveStatus
178solve22(
179 const ScalarType* POINTER_RESTRICT A,
180 const ScalarType* POINTER_RESTRICT b,
181 ScalarType* POINTER_RESTRICT x)
182{
183 if (!stk::simd::are_any(
184 stk::math::abs(determinant22(A)) > tiny_positive_value())) {
185 return SolveStatus::FAILED;
186 }
187
188 enum { XX = 0, XY = 1, YX = 2, YY = 3 };
189 enum { X_RANK1 = 0, Y_RANK1 = 1 };
190
191 const ScalarType inv_detA = 1.0 / determinant22(A);
192 x[X_RANK1] = (A[YY] * b[X_RANK1] - A[XY] * b[Y_RANK1]) * inv_detA;
193 x[Y_RANK1] = -(A[YX] * b[X_RANK1] - A[XX] * b[Y_RANK1]) * inv_detA;
194
195 return SolveStatus::SUCCEEDED;
196}

198// computes b = A^-1 x;
199template <typename ScalarType>
200KOKKOS_FORCEINLINE_FUNCTION SolveStatus
201solve33(
202 const ScalarType* POINTER_RESTRICT A,
203 const ScalarType* POINTER_RESTRICT b,
204 ScalarType* POINTER_RESTRICT x)
205{
206 if (!stk::simd::are_any(
207 stk::math::abs(determinant33(A)) > tiny_positive_value())) {
208 return SolveStatus::FAILED;
209 }
210
211 enum {
212 XX = 0,
213 XY = 1,
214 XZ = 2,
215 YX = 3,
216 YY = 4,
217 YZ = 5,
218 ZX = 6,
219 ZY = 7,
220 ZZ = 8
221 };
222 enum { X_RANK1 = 0, Y_RANK1 = 1, Z_RANK1 = 2 };
223
224 const ScalarType inv_detA = 1.0 / determinant33(A);
225 x[X_RANK1] = ((A[YY] * A[ZZ] - A[YZ] * A[ZY]) * b[X_RANK1] +
226 (A[XZ] * A[ZY] - A[XY] * A[ZZ]) * b[Y_RANK1] +
227 (A[XY] * A[YZ] - A[XZ] * A[YY]) * b[Z_RANK1]) *
228 inv_detA;
229
230 x[Y_RANK1] = ((A[YZ] * A[ZX] - A[YX] * A[ZZ]) * b[X_RANK1] +
231 (A[XX] * A[ZZ] - A[XZ] * A[ZX]) * b[Y_RANK1] +
232 (A[XZ] * A[YX] - A[XX] * A[YZ]) * b[Z_RANK1]) *
233 inv_detA;
234
235 x[Z_RANK1] = ((A[YX] * A[ZY] - A[YY] * A[ZX]) * b[X_RANK1] +
236 (A[XY] * A[ZX] - A[XX] * A[ZY]) * b[Y_RANK1] +
237 (A[XX] * A[YY] - A[XY] * A[YX]) * b[Z_RANK1]) *
238 inv_detA;
239
240 return SolveStatus::SUCCEEDED;
241}

261template <typename ScalarType>
262KOKKOS_FORCEINLINE_FUNCTION void
263mxm33(const ScalarType* A, const ScalarType* B, ScalarType* C)
264{
265 enum {
266 XX = 0,
267 XY = 1,
268 XZ = 2,
269 YX = 3,
270 YY = 4,
271 YZ = 5,
272 ZX = 6,
273 ZY = 7,
274 ZZ = 8

276 C[XX] = A[XX] * B[XX] + A[XY] * B[YX] + A[XZ] * B[ZX];
277 C[XY] = A[XX] * B[XY] + A[XY] * B[YY] + A[XZ] * B[ZY];
278 C[XZ] = A[XX] * B[XZ] + A[XY] * B[YZ] + A[XZ] * B[ZZ];
