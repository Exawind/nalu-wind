22 template <int poly_order, typename Scalar>
23 void
24 scalar_dt_lhs(
25  const CVFEMOperators<poly_order, Scalar>& ops,
26  const nodal_scalar_view<poly_order, Scalar>& vol,
27  double gamma1_div_dt,
28  const nodal_scalar_view<poly_order, Scalar>& rho_p1,
29  matrix_view<poly_order, Scalar>& lhs)
30 {
31  constexpr int n1D = poly_order + 1;
32  const auto& weight = ops.mat_.nodalWeights;
33 
34  for (int n = 0; n < n1D; ++n) {
35  for (int m = 0; m < n1D; ++m) {
36  for (int l = 0; l < n1D; ++l) {
37  auto rowIndex = idx<n1D>(n, m, l);
38  for (int k = 0; k < n1D; ++k) {
39  auto gammaWnk = gamma1_div_dt * weight(n, k);
40  for (int j = 0; j < n1D; ++j) {
41  auto gammWnkWmj = gammaWnk * weight(m, j);
42  for (int i = 0; i < n1D; ++i) {
43  lhs(rowIndex, idx<n1D>(k, j, i)) +=
44  gammWnkWmj * weight(l, i) * vol(k, j, i) * rho_p1(k, j, i);
45  }
46  }
47  }
48  }
49  }
50  }

54 void
55 scalar_dt_rhs(
56  const CVFEMOperators<poly_order, Scalar>& ops,
57  const nodal_scalar_view<poly_order, Scalar>& metric,
58  double gamma_div_dt[3],
59  const nodal_scalar_view<poly_order, Scalar>& rhom1,
60  const nodal_scalar_view<poly_order, Scalar>& rhop0,
61  const nodal_scalar_view<poly_order, Scalar>& rhop1,
62  const nodal_scalar_view<poly_order, Scalar>& phim1,
63  const nodal_scalar_view<poly_order, Scalar>& phip0,
64  const nodal_scalar_view<poly_order, Scalar>& phip1,
65  nodal_scalar_view<poly_order, Scalar>& rhs)
66 {
67  constexpr int n1D = poly_order + 1;
68 
69  nodal_scalar_workview<poly_order, Scalar> work_drhoudt(0);
70  auto& drhoqdt = work_drhoudt.view();
71 
72  for (int k = 0; k < n1D; ++k) {
73  for (int j = 0; j < n1D; ++j) {
74  for (int i = 0; i < n1D; ++i) {
75  drhoqdt(k, j, i) =
76  -(gamma_div_dt[0] * rhop1(k, j, i) * phip1(k, j, i) +
77  gamma_div_dt[1] * rhop0(k, j, i) * phip0(k, j, i) +
78  gamma_div_dt[2] * rhom1(k, j, i) * phim1(k, j, i)) *
79  metric(k, j, i);
80  }
81  }
82  }
83  ops.volume(drhoqdt, rhs);
84 }
