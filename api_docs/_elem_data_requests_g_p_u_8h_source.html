260template <typename T, typename U>
261auto&
262ElemDataRequestsGPU::get_coord_ptr(const T& fieldMgr, const U& iter) const
263{
264 if constexpr (std::is_same_v<T, nalu::FieldManager>)
265 return fieldMgr.template get_ngp_field_ptr<double>(iter.second->name());
266 else
267 return fieldMgr.template get_field<double>(

271template <typename T>
272void
273ElemDataRequestsGPU::fill_host_coords_fields(
274 const ElemDataRequests& dataReq, const T& fieldMgr)
275{
276#if defined(KOKKOS_ENABLE_GPU)
277 coordsFields_ = FieldView(
278 Kokkos::ViewAllocateWithoutInitializing("CoordsFields"),
279 dataReq.get_coordinates_map().size());
280#else
281 coordsFields_ =
282 FieldView("CoordsFields", dataReq.get_coordinates_map().size());
283#endif
284 coordsFieldsTypes_ =
285 CoordsTypesView("CoordsFieldsTypes", dataReq.get_coordinates_map().size());
286
287 hostCoordsFields_ = Kokkos::create_mirror_view(coordsFields_);
288 hostCoordsFieldsTypes_ = Kokkos::create_mirror_view(coordsFieldsTypes_);
289
290 unsigned i = 0;
291 for (auto iter : dataReq.get_coordinates_map()) {
292 hostCoordsFields_(i) = CoordFieldInfo(get_coord_ptr(fieldMgr, iter));
293 hostCoordsFieldsTypes_(i) = iter.first;
294 ++i;
295 }
296}

310template <typename T>
311void
312ElemDataRequestsGPU::fill_host_fields(
313 const ElemDataRequests& dataReq, const T& fieldMgr)
314{
315#if defined(KOKKOS_ENABLE_GPU)
316 fields = FieldInfoView(
317 Kokkos::ViewAllocateWithoutInitializing("Fields"),
318 dataReq.get_fields().size());
319#else
320 fields = FieldInfoView("Fields", dataReq.get_fields().size());
321#endif
322 hostFields = Kokkos::create_mirror_view(fields);
323 unsigned i = 0;
324 for (const FieldInfo& finfo : dataReq.get_fields()) {
325 stk::mesh::NgpField<double>& fld_ptr = get_field_ptr(fieldMgr, finfo);
326 hostFields(i++) =
327 FieldInfoType(fld_ptr, finfo.scalarsDim1, finfo.scalarsDim2);
328 }
329}
