267 template <typename T>
268 void
269 ElemDataRequestsGPU::fill_host_coords_fields(
270  const ElemDataRequests& dataReq, const T& fieldMgr)
271 {
272 #if defined(KOKKOS_ENABLE_GPU)
273  coordsFields_ = FieldView(
274  Kokkos::ViewAllocateWithoutInitializing("CoordsFields"),
275  dataReq.get_coordinates_map().size());
276 #else
277  coordsFields_ =
278  FieldView("CoordsFields", dataReq.get_coordinates_map().size());
279 #endif
280  coordsFieldsTypes_ =
281  CoordsTypesView("CoordsFieldsTypes", dataReq.get_coordinates_map().size());
282 
283  hostCoordsFields_ = Kokkos::create_mirror_view(coordsFields_);
284  hostCoordsFieldsTypes_ = Kokkos::create_mirror_view(coordsFieldsTypes_);
285 
286  unsigned i = 0;
287  for (auto iter : dataReq.get_coordinates_map()) {
288  hostCoordsFields_(i) = CoordFieldInfo(get_coord_ptr(fieldMgr, iter));
289  hostCoordsFieldsTypes_(i) = iter.first;
290  ++i;
291  }
292 }
