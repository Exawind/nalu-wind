60struct Quad42DBasis
61{
62 using traits_t = AlgTraitsQuad4_2D;
63 static constexpr int N1D = 2;
64 static constexpr int DIM = traits_t::nDim_;
65 static constexpr int NNODES = traits_t::nodesPerElement_;
66
67 [[nodiscard]] static constexpr ArrayND<int[2]> to_tensor(int n)
68 {
69 constexpr ArrayND<int[NNODES][2]> map{{{0, 0}, {1, 0}, {1, 1}, {0, 1}}};
70 return {map(n, 0), map(n, 1)};
71 }
72
73 template <typename LocT>
74 [[nodiscard]] static constexpr auto interp_1_2D(int l, const LocT& x)
75 {
76 return 0.5 * (1 + (2 * l - 1) * x);
77 }
78
79 template <typename LocT>
80 [[nodiscard]] static constexpr auto deriv_1_2D(int l, const LocT& /*unused*/)
81 {
82 return 0.5 * (2 * l - 1);
83 }
84
85 template <typename LocT>
86 [[nodiscard]] static constexpr double interpolant(int n, const LocT& x)
87 {
88 const auto ij = to_tensor(n);
89 return interp_1_2D(ij[0], x[0]) * interp_1_2D(ij[1], x[1]);
90 }
91
92 template <typename LocT>
93 [[nodiscard]] static constexpr auto deriv_coeff(int n, const LocT& x, int d)
94 {
95 // for consistency in metric tensor computations, rescale back to-1/2,1/2
96 // range
97 constexpr int iso_range = 2;
98
99 const auto ij = to_tensor(n);
100 if (d == 0) {
101 return iso_range * deriv_1_2D(ij(0), x[0]) * interp_1_2D(ij(1), x[1]);
102 }
103 return iso_range * interp_1_2D(ij(0), x[0]) * deriv_1_2D(ij(1), x[1]);
104 }
105};

147struct Pyr5Basis
148{
149 using traits_t = AlgTraitsPyr5;
150 static constexpr int DIM = traits_t::nDim_;
151 static constexpr int NNODES = traits_t::nodesPerElement_;
152
153 [[nodiscard]] static constexpr int sgn(int n, int d)
154 {
155 constexpr auto map =
156 ArrayND<int[4][2]>{{{-1, -1}, {+1, -1}, {+1, +1}, {-1, +1}}};
157 return map(n % 4, d);
158 }
159
160 template <typename LocT>
161 [[nodiscard]] static constexpr double interpolant(int n, const LocT& x)
162 {
163 if (n >= 4 || x[2] == val_t<LocT>(1.)) {
164 return x[2];
165 }
166 return 0.25 * (1 + sgn(n, 0) * x[0] - x[2]) *
167 (1 + sgn(n, 1) * x[1] - x[2]) / (1 - x[2]);
168 }
169
170 template <typename LocT>
171 [[nodiscard]] static constexpr auto deriv_coeff(int n, const LocT& x, int d)
172 {
173 if (n >= 4 || x[2] == 1.) {
174 return val_t<LocT>(d == 2);
175 }
176 const auto apex = 0.25 / (1 - x[2]);
177 const auto square_x = 1 + sgn(n, 0) * x[0] - x[2];
178 const auto square_y = 1 + sgn(n, 1) * x[1] - x[2];
179 const ArrayND<val_t<LocT>[3]> dsquare{
180 {sgn(n, 0) * square_y, square_x * sgn(n, 1), -(square_x + square_y)}};
181 const ArrayND<val_t<LocT>[3]> dinv_term{0, 0, 4 * apex * apex};
182 return dsquare(d) * apex + square_x * square_y * dinv_term(d);
183 }
184};

186struct Pyr5DegenHexBasis
187{
188 using traits_t = AlgTraitsPyr5;
189 static constexpr int DIM = traits_t::nDim_;
190 static constexpr int NNODES = traits_t::nodesPerElement_;
191
192 [[nodiscard]] static constexpr int sgn(int n, int d)
193 {
194 constexpr auto map =
195 ArrayND<int[4][2]>{{{-1, -1}, {+1, -1}, {+1, +1}, {-1, +1}}};
196 return map(n % 4, d);
197 }
198
199 template <typename LocT>
200 [[nodiscard]] static constexpr auto interpolant(int n, const LocT& x)
201 {
202 if (n >= 4) {
203 return x[2];
204 }
205 return 0.25 * (1 + sgn(n, 0) * x[0]) * (1 + sgn(n, 1) * x[1]) * (1 - x[2]);
206 }
207
208 template <typename LocT>
209 [[nodiscard]] static constexpr auto deriv_coeff(int n, const LocT& x, int d)
210 {
211 if (n >= 4) {
212 return val_t<LocT>(d == 2);
213 }
214 const ArrayND<val_t<LocT>[3]> dsq = {
215 sgn(n, 0) * (1 + sgn(n, 1) * x[1]), (1 + sgn(n, 0) * x[0]) * sgn(n, 1),
216 0};
217 return 0.25 * dsq(d) * (1 - x[2]) -
218 (d == 2) * (1 + sgn(n, 0) * x[0]) * (1 + sgn(n, 1) * x[1]);
219 }
220};

222struct Wed6Basis
223{
224 using traits_t = AlgTraitsWed6;
225 static constexpr int DIM = traits_t::nDim_;
226 static constexpr int NNODES = traits_t::nodesPerElement_;
227
228 template <typename LocT>
229 [[nodiscard]] static constexpr double interpolant(int n, const LocT& x)
230 {
231 const ArrayND<typename LocT::value_type[3]> tri{
232 1 - (x[0] + x[1]), x[0], x[1]};
233 return tri(n % 3) * (0.5 * (1 + (2 * (n > 2) - 1) * x[2]));
234 }
235
236 template <typename LocT>
237 [[nodiscard]] static constexpr auto deriv_coeff(int n, const LocT& x, int d)
238 {
239 const ArrayND<val_t<LocT>[3]> tri{1 - (x[0] + x[1]), x[0], x[1]};
240 constexpr ArrayND<val_t<LocT>[3][3]> dtri{
241 {{-1, -1, 0}, {1, 0, 0}, {0, 1, 0}}};
242 const auto prism = 0.5 * (1 + (2 * (n > 2) - 1) * x[2]);
243 const auto dprism = ArrayND<val_t<LocT>[3]>{0, 0, 0.5 * (2 * (n > 2) - 1)};
244 return dtri(n % 3, d) * prism + tri(n % 3) * dprism(d);
245 }
246};

248struct Hex8Basis
249{
250 using traits_t = AlgTraitsHex8;
251 static constexpr int N1D = 2;
252 static constexpr int DIM = 3;
253 static constexpr int NNODES = traits_t::nodesPerElement_;
254
255 [[nodiscard]] static constexpr int tensor_map(int k, int j, int i)
256 {
257 constexpr ArrayND<int[N1D][N1D][N1D]> map{
258 {{{0, 1}, {3, 2}}, {{4, 5}, {7, 6}}}};
259 return map(k, j, i);
260 }
261
262 [[nodiscard]] static constexpr ArrayND<int[3]> to_tensor(int n)
263 {
264 constexpr ArrayND<int[NNODES][3]> map{
265 {{0, 0, 0},
266 {1, 0, 0},
267 {1, 1, 0},
268 {0, 1, 0},
269 {0, 0, 1},
270 {1, 0, 1},
271 {1, 1, 1},
272 {0, 1, 1}}};
273 return {map(n, 0), map(n, 1), map(n, 2)};
274 }
275
276 template <typename LocT>
277 [[nodiscard]] static constexpr auto interp_1(int l, const LocT& x)
278 {
279 return 0.5 * (1 + (2 * l - 1) * x);
280 }
281
282 template <typename LocT>
283 [[nodiscard]] static constexpr auto deriv_1(int l, const LocT& /*unused*/)
284 {
285 return 0.5 * (2 * l - 1);
286 }
287
288 template <typename LocT>
289 [[nodiscard]] static constexpr double interpolant(int n, const LocT& x)
290 {
291 const auto ijk = to_tensor(n);
292 return interp_1(ijk[0], x[0]) * interp_1(ijk[1], x[1]) *
293 interp_1(ijk[2], x[2]);
294 }
295
296 template <typename LocT>
297 [[nodiscard]] static constexpr auto deriv_coeff(int n, const LocT& x, int d)
298 {
299 constexpr auto iso_range =
300 2; // for historical reasons, rescale to -1/2, 1/2
301
302 const auto ijk = to_tensor(n);
303 const auto xv = (d == 0) ? deriv_1(ijk(0), x[0]) : interp_1(ijk(0), x[0]);
304 const auto yv = (d == 1) ? deriv_1(ijk(1), x[1]) : interp_1(ijk(1), x[1]);
305 const auto zv = (d == 2) ? deriv_1(ijk(2), x[2]) : interp_1(ijk(2), x[2]);
306 return iso_range * xv * yv * zv;
307 }
308};

355template <typename BasisT, typename ParCoordsArrayT, typename ValArrayT>
356[[nodiscard]] KOKKOS_INLINE_FUNCTION constexpr std::enable_if_t<
357 is_tensor_compatible<BasisT>::value,
358 ArrayND<typename ValArrayT::value_type[ParCoordsArrayT::extent_int(0)]
359 [ValArrayT::extent_int(1)]>>
360interpolate(const ParCoordsArrayT& par_coords)
361{
362 constexpr int nparcoords = ParCoordsArrayT::extent_int(0);
363 using res_val_t = typename ValArrayT::value_type;
364 using array_t = res_val_t[nparcoords];
365
366 ArrayND<array_t> result{};
367 for (int l = 0; l < nparcoords; ++l) {
368 for (int d = 0; d < ValArrayT::extent_int(1); ++d) {
369 result(l, d) = 0;
370 }
371 for (int i = 0; i < BasisT::N1D; ++i) {
372 const auto interp_x = BasisT::interp_1(i, par_coords(l, 0));
373 for (int j = 0; j < BasisT::N1D; ++j) {
374 const auto interp_xy = BasisT::interp_1(j, par_coords(l, 1)) * interp_x;
375 for (int k = 0; k < BasisT::N1D; ++k) {
376 const auto interp_xyz =
377 interp_xy * BasisT::interp_1(i, par_coords(l, 2));
378 for (int d = 0; d < ValArrayT::extent_int(1); ++d) {
379 result(l, d) += interp_xyz * vals(BasisT::tensor_map(k, j, i), d);
380 }
381 }
382 }
383 }
384 }
385 return result;
386}

388template <typename BasisT, typename ParCoordsArrayT, typename ValArrayT>
389[[nodiscard]] KOKKOS_INLINE_FUNCTION constexpr std::enable_if_t<
390 !is_tensor_compatible<BasisT>::value,
391 ArrayND<typename ValArrayT::value_type[ParCoordsArrayT::extent_int(0)]
392 [ValArrayT::extent_int(1)]>>
393interpolate(const ParCoordsArrayT& par_coords, const ValArrayT& vals)
394{
395 constexpr int nparcoords = ParCoordsArrayT::extent_int(0);
396 constexpr int nvals = ValArrayT::extent_int(1);
397 using res_val_t = std::decay_t<decltype(par_coords(0, 0) * vals(0, 0))>;
398 using array_t = res_val_t[nparcoords][nvals];
399
400 ArrayND<array_t> result{};
401 for (int l = 0; l < nparcoords; ++l) {
402
403 ArrayND<res_val_t[ParCoordsArrayT::extent_int(1)]> point;
404 for (int d = 0; d < ParCoordsArrayT::extent_int(1); ++d) {
405 point(d) = par_coords(l, d);
406 }
407 for (int d = 0; d < ValArrayT::extent_int(1); ++d) {
408 result(l, d) = 0;
409 }
410
411 for (int n = 0; n < BasisT::NNODES; ++n) {
412 const auto interp_n = BasisT::interp(n, point);
413 for (int d = 0; d < ValArrayT::extent_int(1); ++d) {
414 result(l, d) += interp_n * vals(n, d);
415 }
416 }
417 }
418 return result;
419}

421template <typename BasisT, typename ParCoordsArrayT>
422[[nodiscard]] KOKKOS_INLINE_FUNCTION constexpr ArrayND<
423 typename ParCoordsArrayT::value_type[ParCoordsArrayT::extent_int(0)]
424 [BasisT::NNODES]>
425interpolants(const ParCoordsArrayT& par_coords)
426{
427 constexpr int nparcoords = ParCoordsArrayT::extent_int(0);
428 using res_val_t = std::decay_t<decltype(par_coords(0, 0))>;
429
430 ArrayND<res_val_t[ParCoordsArrayT::extent_int(0)][BasisT::NNODES]> result;
431 for (int l = 0; l < nparcoords; ++l) {
432
433 ArrayND<res_val_t[ParCoordsArrayT::extent_int(1)]> point;
434 for (int d = 0; d < ParCoordsArrayT::extent_int(1); ++d) {
435 point(d) = par_coords(l, d);
436 }
437 for (int n = 0; n < BasisT::NNODES; ++n) {
438 result(l, n) = BasisT::interpolant(n, point);
439 }
440 }
441 return result;
442}
