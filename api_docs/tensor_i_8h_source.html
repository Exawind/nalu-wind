118template <typename T>
119KOKKOS_FORCEINLINE_FUNCTION VectorT<T>
120operator&(const TensorT<T>& t, const VectorT<T>& v)
121{
122 VectorT<T> out;
123 out.x() = t.xx() * v.x() + t.xy() * v.y() + t.xz() * v.z();
124 out.y() = t.yx() * v.x() + t.yy() * v.y() + t.yz() * v.z();
125 out.z() = t.zx() * v.x() + t.zy() * v.y() + t.zz() * v.z();
126 return out;
127}

142operator&(const TensorT<T>& t1, const TensorT<T>& t2)

147 t3.xy() = t1.xx() * t2.xy() + t1.xy() * t2.yy() + t1.xz() * t2.zy();
148 t3.xz() = t1.xx() * t2.xz() + t1.xy() * t2.yz() + t1.xz() * t2.zz();
149

151 t3.yy() = t1.yx() * t2.xy() + t1.yy() * t2.yy() + t1.yz() * t2.zy();
152 t3.yz() = t1.yx() * t2.xz() + t1.yy() * t2.yz() + t1.yz() * t2.zz();

261KOKKOS_FORCEINLINE_FUNCTION Tensor
262quaternion(const Vector& axis, const double angle)
263{
264 const double ang = -1.0 * utils::radians(angle);
265 const double cval = std::cos(0.5 * ang);
266 const double sval = std::sin(0.5 * ang);
267 const auto vmag = mag(axis);
268 const double q0 = cval;
269 const double q1 = sval * axis.x() / vmag;
270 const double q2 = sval * axis.y() / vmag;
271 const double q3 = sval * axis.z() / vmag;
272
273 Tensor t;
274 t.xx() = q0 * q0 + q1 * q1 - q2 * q2 - q3 * q3;
275 t.xy() = 2.0 * (q1 * q2 - q0 * q3);
276 t.xz() = 2.0 * (q0 * q2 + q1 * q3);
277
278 t.yx() = 2.0 * (q1 * q2 + q0 * q3);
279 t.yy() = q0 * q0 - q1 * q1 + q2 * q2 - q3 * q3;
280 t.yz() = 2.0 * (q2 * q3 - q0 * q1);
281
282 t.zx() = 2.0 * (q1 * q3 - q0 * q2);
283 t.zy() = 2.0 * (q0 * q1 + q2 * q3);
284 t.zz() = q0 * q0 - q1 * q1 - q2 * q2 + q3 * q3;
285
286 return t;
287}
