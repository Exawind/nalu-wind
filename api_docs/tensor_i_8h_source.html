236 KOKKOS_FORCEINLINE_FUNCTION Tensor
237 quaternion(const Vector& axis, const double angle)
238 {
239  const double ang = -1.0 * utils::radians(angle);
240  const double cval = std::cos(0.5 * ang);
241  const double sval = std::sin(0.5 * ang);
242  const auto vmag = mag(axis);
243  const double q0 = cval;
244  const double q1 = sval * axis.x() / vmag;
245  const double q2 = sval * axis.y() / vmag;
246  const double q3 = sval * axis.z() / vmag;
247 
248  Tensor t;
249  t.xx() = q0 * q0 + q1 * q1 - q2 * q2 - q3 * q3;
250  t.xy() = 2.0 * (q1 * q2 - q0 * q3);
251  t.xz() = 2.0 * (q0 * q2 + q1 * q3);
252 
253  t.yx() = 2.0 * (q1 * q2 + q0 * q3);
254  t.yy() = q0 * q0 - q1 * q1 + q2 * q2 - q3 * q3;
255  t.yz() = 2.0 * (q2 * q3 - q0 * q1);
256 
257  t.zx() = 2.0 * (q1 * q3 - q0 * q2);
258  t.zy() = 2.0 * (q0 * q1 + q2 * q3);
259  t.zz() = q0 * q0 - q1 * q1 - q2 * q2 + q3 * q3;
260 
261  return t;
262 }
