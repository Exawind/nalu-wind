33 template <typename SimdViewType>
34 KOKKOS_FUNCTION void
35 interleave(SimdViewType& dview, const double* sviews[], int simdElems)
36 {
37  int dim = dview.size();
38  DoubleType* dptr = dview.data();
39  for (int i = 0; i < dim; ++i) {
40  DoubleType& d = dptr[i];
41  for (int simdIndex = 0; simdIndex < simdElems; ++simdIndex) {
42  stk::simd::set_data(d, simdIndex, sviews[simdIndex][i]);
43  }
44  for (int simdIndex = simdElems; simdIndex < simdLen; ++simdIndex) {
45  stk::simd::set_data(d, simdIndex, 0.0);
46  }
47  }

56 {
57  const double* src[stk::simd::ndoubles] = {nullptr};
58  unsigned numViews = simdData.get_num_1D_views();
59  for (unsigned viewIndex = 0; viewIndex < numViews; ++viewIndex) {
60  for (int simdIndex = 0; simdIndex < simdElems; ++simdIndex) {
61  src[simdIndex] = data[simdIndex]->get_1D_view_by_index(viewIndex).data();
62  STK_NGP_ThrowAssert(
63  data[simdIndex]->get_1D_view_by_index(viewIndex).size() ==
64  simdData.get_1D_view_by_index(viewIndex).size());
65  STK_NGP_ThrowAssert(src[simdIndex] != nullptr);
66  STK_NGP_ThrowAssert(
67  src[simdIndex][0] ==
68  data[simdIndex]->get_1D_view_by_index(viewIndex).data()[0]);
69  }
70  interleave(simdData.get_1D_view_by_index(viewIndex), src, simdElems);
71  }
72 
73  numViews = simdData.get_num_2D_views();
