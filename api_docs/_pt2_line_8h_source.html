24KOKKOS_INLINE_FUNCTION
25double
26projectPt2Line(
27 const vs::Vector& pt, const vs::Vector& lStart, const vs::Vector& lEnd)
28{
29 double nonDimCoord = 0.0;
30
31 double num = 0.0;
32 double denom = 0.0;
33
34 for (int i = 0; i < 3; i++) {
35 num += (pt[i] - lStart[i]) * (lEnd[i] - lStart[i]);
36 denom += (lEnd[i] - lStart[i]) * (lEnd[i] - lStart[i]);
37 }
38
39 nonDimCoord = num / denom;
40 return nonDimCoord;
41}

59 double denom = 0.0;
