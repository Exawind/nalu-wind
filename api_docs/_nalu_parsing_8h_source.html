372inline bool
373string_represents_positive_integer(std::string v)
374{
375 return !v.empty() && v.find_first_not_of("0123456789") == std::string::npos;
376}

378template <typename T>
379typename std::enable_if<std::is_integral<T>::value, T>::type
380get_yaml_value(const YAML::Node& v)
381{
382 // yaml will parse inputs with leading zeros as octals if
383 // the number is an octal, e.g. max_itertions: 0010 is
384 // equivalent to max_iterations: 8.
385 // this works around that to have 0010 equivalent to 10
386 if (string_represents_positive_integer(v.template as<std::string>())) {
387 return std::stoi(v.template as<std::string>());
388 } else {
389 return v.template as<T>();
390 }
391}
392
393template <typename T>
394typename std::enable_if<!std::is_integral<T>::value, T>::type
395get_yaml_value(const YAML::Node& v)
396{
397 return v.template as<T>();
398}

402template <typename T>
403void
404get_if_present(
405 const YAML::Node& node,
406 const std::string& key,
407 T& result,
408 const T& default_if_not_present = T())
409{
410 if (node[key]) {
411 result = get_yaml_value<T>(node[key]);
412 } else {
413 result = default_if_not_present;
414 }
415}

419template <typename T>
420void
421get_if_present_no_default(
422 const YAML::Node& node, const std::string& key, T& result)
423{
424 if (node[key]) {
425 result = get_yaml_value<T>(node[key]);
426 }
427}

430template <typename T>
431void
432get_required(const YAML::Node& node, const std::string& key, T& result)
433{
434 if (node[key]) {
435 result = get_yaml_value<T>(node[key]);
436 } else {
437 if (!NaluEnv::self().parallel_rank()) {
438 std::ostringstream err_msg;
439 err_msg << "\n\nError: parsing missing required key: " << key << " at "
440 << NaluParsingHelper::line_info(node)
441 << " for node= " << std::endl;
442 NaluParsingHelper::emit(err_msg, node);
443 std::cout << err_msg.str() << std::endl;
444 }
445 throw std::runtime_error("Error: parsing missing required key: " + key);
446 }
447}
