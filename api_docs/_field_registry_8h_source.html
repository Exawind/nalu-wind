21/* A class that contains definitions for all the available fields that can be
22 * registered in nalu-wind
23 */
24class FieldRegistry
25{
26public:
27 static FieldDefTypes query(int numDim, int numStates, std::string name)
28 {
29 static FieldRegistry instance;
30 const std::map<std::string, FieldDefTypes>* db;
31
32 switch (numDim) {
33 case 2: {
34 switch (numStates) {
35 case 2: {
36 db = &(instance.database_2D_2_state_);
37 break;
38 }
39 case 3: {
40 db = &(instance.database_2D_3_state_);
41 break;
42 }
43 default:
44 throw std::runtime_error(
45 "Unsupported number of reference states for field " + name +
46 " with " + std::to_string(numStates) + " states and " +
47 std::to_string(numDim) + " dims.");
48 }
49 break;
50 }
51 case 3: {
52 switch (numStates) {
53 case 2: {
54 db = &(instance.database_3D_2_state_);
55 break;
56 }
57 case 3: {
58 db = &(instance.database_3D_3_state_);
59 break;
60 }
61 default:
62 throw std::runtime_error(
63 "Unsupported number of reference states for field " + name +
64 " with " + std::to_string(numStates) + " states and " +
65 std::to_string(numDim) + " dims.");
66 }
67 break;
68 }
69 default:
70 throw std::runtime_error(
71 "Only 2 and 3 spatial dimensions are supported. Dim Given was " +
72 std::to_string(numDim));
73 }
74
75 auto fieldDefIter = db->find(name);
76
77 if (fieldDefIter == db->end()) {
78 std::string message = "Attempting to access an undefined field: '" +
79 name + "' with spatial dimension " +
80 std::to_string(numDim) + " and number of states " +
81 std::to_string(numStates);
82 throw std::runtime_error(message);
83 }
84 return fieldDefIter->second;
85 }
86
87 FieldRegistry(const FieldRegistry&) = delete;
88 void operator=(FieldRegistry const&) = delete;
89
90private:
91 FieldRegistry();
92 // In order to accomodate and embed the state and dimenstion information we
93 // ended up creating four separate databases that are templated on the number
94 // of states required by the time integration scheme This was done to preserve
95 // the singelton/refernce lookup only behavior of the FieldRegistry
96 const std::map<std::string, FieldDefTypes>& database_2D_2_state_;
97 const std::map<std::string, FieldDefTypes>& database_2D_3_state_;
98 const std::map<std::string, FieldDefTypes>& database_3D_2_state_;
99 const std::map<std::string, FieldDefTypes>& database_3D_3_state_;
100};
