28 double KOKKOS_FORCEINLINE_FUNCTION
29 linear_ramp_theta(
30  const aero::SixDOF& hub,
31  const vs::Vector& root,
32  const vs::Vector& position,
33  const double rampSpan,
34  const double zeroRampLoc,
35  const bool useRamp = true)
36 {
37  auto v1 = (root - hub.position_).normalize();
38  auto v2 = (position - hub.position_).normalize();
39 
40  // make sure vectors are in the plane of rotation to compute the angle between
41  // them
42  const vs::Vector rotationAxis =
43  wmp::rotate(hub.orientation_, vs::Vector::ihat(), true).normalize();
44  v1 = v1 - vs::project(v1, rotationAxis);
45  v2 = v2 - vs::project(v2, rotationAxis);
46 
47  const auto angle = vs::angle(v1, v2);
48 
49  return stk::math::min(
50  1.0, stk::math::max(
51  static_cast<double>(!useRamp), (zeroRampLoc - angle) / rampSpan));
52 }
