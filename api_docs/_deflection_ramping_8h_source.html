17double KOKKOS_FORCEINLINE_FUNCTION
18linear_ramp_span(
19 const double spanLocation,
20 const double zeroRampDistance,
21 const bool useRamp = true)
22{
23 return 1.0 - stk::math::max(
24 (zeroRampDistance - spanLocation) / zeroRampDistance, 0.0) *

28double KOKKOS_FORCEINLINE_FUNCTION
29linear_ramp_theta(
30 const aero::SixDOF& hub,
31 const vs::Vector& root,
32 const vs::Vector& position,
33 const double rampSpan,
34 const double zeroRampLoc,
35 const bool useRamp = true)
36{
37 auto v1 = (root - hub.position_).normalize();
38 auto v2 = (position - hub.position_).normalize();
39
40 // make sure vectors are in the plane of rotation to compute the angle between
41 // them
42 const vs::Vector rotationAxis =
43 wmp::rotate(hub.orientation_, vs::Vector::ihat(), true).normalize();
44 v1 = v1 - vs::project(v1, rotationAxis);
45 v2 = v2 - vs::project(v2, rotationAxis);
46
47 const auto angle = vs::angle(v1, v2);
48
49 return stk::math::min(
50 1.0, stk::math::max(
51 static_cast<double>(!useRamp), (zeroRampLoc - angle) / rampSpan));
52}

57double KOKKOS_FORCEINLINE_FUNCTION
58temporal_ramp(
59 const double time,
60 const double startRamp,
61 const double endRamp,
62 const bool useRamp = true)
63{
64 const double denom = stk::math::max(endRamp - startRamp, 1e-12);
65 return stk::math::max(
66 static_cast<double>(!useRamp),
67 stk::math::min(1.0, (time - startRamp) / denom));
68}
