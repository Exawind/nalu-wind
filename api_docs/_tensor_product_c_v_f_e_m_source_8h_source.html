22template <int poly_order, typename Scalar>
23void

28 nodal_scalar_view<poly_order, Scalar>& rhs)

42void
43add_volumetric_source(

62template <int poly_order, typename Scalar, typename SourceFunc>
63void
64add_volumetric_source_func(
65 const CVFEMOperators<poly_order, Scalar>& ops,
66 const nodal_scalar_view<poly_order, Scalar>& volume_metric,
67 nodal_scalar_view<poly_order, Scalar>& rhs,
68 SourceFunc func)
69{
70 nodal_scalar_workview<poly_order, Scalar> l_source(0);
71 auto& source = l_source.view();
72
73 for (int k = 0; k < poly_order + 1; ++k) {
74 for (int j = 0; j < poly_order + 1; ++j) {
75 for (int i = 0; i < poly_order + 1; ++i) {
76 source(k, j, i) = func(k, j, i) * volume_metric(k, j, i);
77 }
78 }
79 }
80 ops.volume(source, rhs);
81}
