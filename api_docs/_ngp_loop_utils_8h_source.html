57 template <typename T, typename DataReqType>
58 inline int
59 ngp_calc_thread_shmem_size(
60  int ndim, const DataReqType& dataReq, const ElemReqType reqType)
61 {
62  int preReqSize = get_num_bytes_pre_req_data<T>(dataReq, ndim, reqType);
63  int mdvSize = MultiDimViews<T>::bytes_needed(
64  dataReq.get_total_num_fields(),
65  count_needed_field_views(dataReq.get_host_fields()));
66 
67 #if !defined(KOKKOS_ENABLE_GPU)
68  // On host account for extra data to store the SIMD and non-SIMD versions of
69  // the ScratchViews
70  return (preReqSize + mdvSize) * 2 * simdLen;
71 #else
72  return (preReqSize + mdvSize);
73 #endif
74 }

84 template <typename T, typename DataReqType>
85 inline int
86 ngp_calc_thread_shmem_size(
87  int ndim, const DataReqType& faceDataReq, const DataReqType& elemDataReq)
88 {
89  const int faceMemSize =
90  ngp_calc_thread_shmem_size<T>(ndim, faceDataReq, ElemReqType::FACE);
91  const int elemMemSize =
92  ngp_calc_thread_shmem_size<T>(ndim, elemDataReq, ElemReqType::ELEM);
93 
94  return (faceMemSize + elemMemSize);

317 template <
318  typename Mesh,
319  typename FieldManager,
320  typename DataReqType,
321  typename AlgFunctor>
322 void
323 run_elem_algorithm(
324  const std::string algName,
325  const MeshInfo<Mesh, FieldManager>& meshInfo,
326  const stk::topology::rank_t rank,
327  const DataReqType& dataReqs,
328  const stk::mesh::Selector& sel,
329  const AlgFunctor algorithm)
330 {
331  using Traits = NGPMeshTraits<Mesh>;
332  using TeamPolicy = typename Traits::TeamPolicy;
333  using TeamHandleType = typename Traits::TeamHandleType;
334  using MeshIndex = typename Traits::MeshIndex;
335 
336  const auto& ndim = meshInfo.ndim();
337  const auto& ngpMesh = meshInfo.ngp_mesh();
338  const auto& fieldMgr = meshInfo.ngp_field_manager();
339 
340  ElemDataRequestsGPU dataReqNGP(fieldMgr, dataReqs);
341 
342  const int nodesPerElement = nodes_per_entity(dataReqNGP);
343  STK_NGP_ThrowRequire(nodesPerElement != 0);
344 
345  const auto reqType =
346  (rank == stk::topology::ELEM_RANK) ? ElemReqType::ELEM : ElemReqType::FACE;
347  const int bytes_per_team = 0;
348  const int bytes_per_thread =
349  impl::ngp_calc_thread_shmem_size<sierra::nalu::DoubleType>(
350  ndim, dataReqNGP, reqType);
351 
352  const auto& buckets = ngpMesh.get_bucket_ids(rank, sel);
353  auto team_exec = impl::ngp_mesh_team_policy<TeamPolicy>(
354  buckets.size(), bytes_per_team, bytes_per_thread);
355 
356  Kokkos::parallel_for(
357  algName, team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
358  auto bktId = buckets.device_get(team.league_rank());
359  auto& bkt = ngpMesh.get_bucket(rank, bktId);
360 
361  ElemSimdData<Mesh> elemData(team, ndim, nodesPerElement, dataReqNGP);
362 
363  const size_t bktLen = bkt.size();
364  const size_t simdBktLen = get_num_simd_groups(bktLen);
365 
366  Kokkos::parallel_for(
367  Kokkos::TeamThreadRange(team, simdBktLen), [&](const size_t& bktIndex) {
368  int nSimdElems = get_length_of_next_simd_group(bktIndex, bktLen);
369  elemData.numSimdElems = nSimdElems;
370 
371  for (int is = 0; is < nSimdElems; ++is) {
372  const unsigned bktOrd = bktIndex * simdLen + is;
373  MeshIndex meshIdx{&bkt, bktOrd};
374  const auto& elem = bkt[bktOrd];
375  elemData.elemInfo[is] =
376  EntityInfo<Mesh>{meshIdx, elem, ngpMesh.get_nodes(meshIdx)};
377 
378  fill_pre_req_data(
379  dataReqNGP, ngpMesh, rank, elem, *elemData.scrView[is]);
380  }
381 
382 #if !defined(KOKKOS_ENABLE_GPU)
383  copy_and_interleave(
384  elemData.scrView, nSimdElems, elemData.simdScrView);
385 #endif
386 
387  fill_master_element_views(dataReqNGP, elemData.simdScrView);
388  algorithm(elemData);
389  });
390  });
391 }

497 template <
498  typename Mesh,
499  typename FieldManager,
500  typename DataReqType,
501  typename AlgFunctor>
502 void
503 run_face_elem_algorithm(
504  const std::string& algName,
505  const MeshInfo<Mesh, FieldManager>& meshInfo,
506  const DataReqType& faceDataReqs,
507  const DataReqType& elemDataReqs,
508  const stk::mesh::Selector& sel,
509  const AlgFunctor algorithm)
510 {
511  static constexpr stk::topology::rank_t elemRank = stk::topology::ELEMENT_RANK;
512  const stk::topology::rank_t sideRank = meshInfo.meta().side_rank();
513  using Traits = NGPMeshTraits<Mesh>;
514  using TeamPolicy = typename Traits::TeamPolicy;
515  using TeamHandleType = typename Traits::TeamHandleType;
516  using MeshIndex = typename Traits::MeshIndex;
517 
518  const auto& ndim = meshInfo.ndim();
519  const auto& ngpMesh = meshInfo.ngp_mesh();
520  const auto& fieldMgr = meshInfo.ngp_field_manager();
521 
522  ElemDataRequestsGPU faceDataNGP(fieldMgr, faceDataReqs);
523  ElemDataRequestsGPU elemDataNGP(fieldMgr, elemDataReqs);
524 
525  const int nodesPerElement = nodes_per_entity(elemDataNGP);
526  const int nodesPerFace = nodes_per_entity(faceDataNGP, METype::FACE);
527  STK_NGP_ThrowRequire(nodesPerElement != 0);
528  STK_NGP_ThrowRequire(nodesPerFace != 0);
529 
530  const int bytes_per_team = 0;
531  const int bytes_per_thread =
532  impl::ngp_calc_thread_shmem_size<sierra::nalu::DoubleType>(
533  ndim, faceDataNGP, elemDataNGP);
534 
535  const auto& buckets = ngpMesh.get_bucket_ids(sideRank, sel);
536  auto team_exec = impl::ngp_mesh_team_policy<TeamPolicy>(
537  buckets.size(), bytes_per_team, bytes_per_thread);
538 
539  Kokkos::parallel_for(
540  algName, team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
541  auto bktId = buckets.device_get(team.league_rank());
542  auto& bkt = ngpMesh.get_bucket(sideRank, bktId);
543 
544  FaceElemSimdData<Mesh> faceElemData(
545  team, ndim, nodesPerFace, nodesPerElement, faceDataNGP, elemDataNGP);
546 
547  const size_t bktLen = bkt.size();
548  const size_t simdBktLen = get_num_simd_groups(bktLen);
549 
550  Kokkos::parallel_for(
551  Kokkos::TeamThreadRange(team, simdBktLen), [&](const size_t& bktIndex) {
552  size_t nSimdFaces = get_length_of_next_simd_group(bktIndex, bktLen);
553  size_t nFacesProcessed = 0;
554 
555  // Only group face/elem pairs in SIMD that have the same face ordinals
556  do {
557  int elemFaceOrd = -1;
558  int simdFaceIdx = 0;
559 
560  while ((nFacesProcessed + simdFaceIdx) < nSimdFaces) {
561  const auto& bktOrd =
562  bktIndex * simdLen + nFacesProcessed + simdFaceIdx;
563  const auto& face = bkt[bktOrd];
564  const auto faceIdx = ngpMesh.fast_mesh_index(face);
565  int faceOrd = ngpMesh.get_element_ordinals(sideRank, faceIdx)[0];
566 
567  // If we have one or more faces processed, then does the current
568  // candidate face the same face ordinal as the ones we have
569  // processed so far
570  if ((elemFaceOrd >= 0) && (faceOrd != elemFaceOrd))
571  break;
572 
573  const auto elems = ngpMesh.get_elements(sideRank, faceIdx);
574  MeshIndex meshIdx{&bkt, static_cast<unsigned>(bktOrd)};
575  const auto elem = elems[0];
576  const auto elemIdx = ngpMesh.fast_mesh_index(elem);
577  faceElemData.faceInfo[simdFaceIdx] = BcFaceElemInfo<Mesh>{
578  meshIdx,
579  face,
580  elem,
581  ngpMesh.get_nodes(sideRank, faceIdx),
582  ngpMesh.get_nodes(elemRank, elemIdx),
583  faceOrd};
584 
585  fill_pre_req_data(
586  faceDataNGP, ngpMesh, sideRank, face,
587  *faceElemData.scrFaceView[simdFaceIdx]);
588  fill_pre_req_data(
589  elemDataNGP, ngpMesh, elemRank, elem,
590  *faceElemData.scrElemView[simdFaceIdx]);
591 
592  elemFaceOrd = faceOrd;
593  ++simdFaceIdx;
594  }
595  faceElemData.faceOrd = elemFaceOrd;
596  faceElemData.numSimdElems = simdFaceIdx;
597  nFacesProcessed += simdFaceIdx;
598 
599 #if !defined(KOKKOS_ENABLE_GPU)
600  copy_and_interleave(
601  faceElemData.scrFaceView, faceElemData.numSimdElems,
602  faceElemData.simdFaceView);
603  copy_and_interleave(
604  faceElemData.scrElemView, faceElemData.numSimdElems,
605  faceElemData.simdElemView);
606 #endif
607  fill_master_element_views(
608  faceDataNGP, faceElemData.simdFaceView, elemFaceOrd);
609  fill_master_element_views(
610  elemDataNGP, faceElemData.simdElemView, elemFaceOrd);
611 
612  algorithm(faceElemData);
613  } while (nFacesProcessed < nSimdFaces);
614  });
615  });
616 }

692  int faceOrd = ngpMesh.get_element_ordinals(sideRank, faceIdx)[0];
693 
694  // If we have one or more faces processed, then does the current
695  // candidate face the same face ordinal as the ones we have
696  // processed so far
697  if ((elemFaceOrd >= 0) && (faceOrd != elemFaceOrd))
698  break;
699 
700  const auto elems = ngpMesh.get_elements(sideRank, faceIdx);
701  MeshIndex meshIdx{&bkt, static_cast<unsigned>(bktOrd)};
702  const auto elem = elems[0];
703  const auto elemIdx = ngpMesh.fast_mesh_index(elem);
704  faceElemData.faceInfo[simdFaceIdx] = BcFaceElemInfo<Mesh>{
705  meshIdx,
706  face,
707  elem,
708  ngpMesh.get_nodes(sideRank, faceIdx),
709  ngpMesh.get_nodes(elemRank, elemIdx),
710  faceOrd};
711 
712  fill_pre_req_data(
713  faceDataNGP, ngpMesh, sideRank, face,
714  *faceElemData.scrFaceView[simdFaceIdx]);
715  fill_pre_req_data(
716  elemDataNGP, ngpMesh, elemRank, elem,
717  *faceElemData.scrElemView[simdFaceIdx]);
718 
719  elemFaceOrd = faceOrd;
720  ++simdFaceIdx;
721  }
722  faceElemData.faceOrd = elemFaceOrd;
723  faceElemData.numSimdElems = simdFaceIdx;
724  nFacesProcessed += simdFaceIdx;
725 
726 #if !defined(KOKKOS_ENABLE_GPU)
727  copy_and_interleave(
728  faceElemData.scrFaceView, faceElemData.numSimdElems,
729  faceElemData.simdFaceView);
730  copy_and_interleave(
731  faceElemData.scrElemView, faceElemData.numSimdElems,
732  faceElemData.simdElemView);
733 #endif
734  fill_master_element_views(
735  faceDataNGP, faceElemData.simdFaceView, elemFaceOrd);
736  fill_master_element_views(
737  elemDataNGP, faceElemData.simdElemView, elemFaceOrd);
738 
739  algorithm(faceElemData, threadVal);
740  } while (nFacesProcessed < nSimdFaces);
741  },
742  ReducerType(bktVal));
743 
744  Kokkos::single(
745  Kokkos::PerTeam(team), [&]() { reduceVal.join(teamVal, bktVal); });
746  },
747  reduceVal);
748 }
