40template <typename TeamPolicy>
41inline TeamPolicy
42ngp_mesh_team_policy(
43 const size_t sz, const size_t bytes_per_team, const size_t bytes_per_thread)
44{
45 TeamPolicy policy(sz, NTHREADS_PER_DEVICE_TEAM);
46 return policy.set_scratch_size(
47 1, Kokkos::PerTeam(bytes_per_team), Kokkos::PerThread(bytes_per_thread));
48}

57template <typename T, typename DataReqType>
58inline int
59ngp_calc_thread_shmem_size(
60 int ndim, const DataReqType& dataReq, const ElemReqType reqType)
61{
62 int preReqSize = get_num_bytes_pre_req_data<T>(dataReq, ndim, reqType);
63 int mdvSize = MultiDimViews<T>::bytes_needed(
64 dataReq.get_total_num_fields(),
65 count_needed_field_views(dataReq.get_host_fields()));
66
67#if !defined(KOKKOS_ENABLE_GPU)
68 // On host account for extra data to store the SIMD and non-SIMD versions of
69 // the ScratchViews
70 return (preReqSize + mdvSize) * 2 * simdLen;
71#else
72 return (preReqSize + mdvSize);
73#endif
74}

84template <typename T, typename DataReqType>
85inline int
86ngp_calc_thread_shmem_size(
87 int ndim, const DataReqType& faceDataReq, const DataReqType& elemDataReq)
88{
89 const int faceMemSize =
90 ngp_calc_thread_shmem_size<T>(ndim, faceDataReq, ElemReqType::FACE);
91 const int elemMemSize =
92 ngp_calc_thread_shmem_size<T>(ndim, elemDataReq, ElemReqType::ELEM);
93
94 return (faceMemSize + elemMemSize);
95}

111template <typename Mesh, typename AlgFunctor>
112void
113run_entity_algorithm(
114 const std::string& algName,
115 const Mesh& mesh,
116 const stk::topology::rank_t rank,
117 const stk::mesh::Selector& sel,
118 const AlgFunctor algorithm)
119{
120 using Traits = NGPMeshTraits<Mesh>;
121 using TeamPolicy = typename Traits::TeamPolicy;
122 using TeamHandleType = typename Traits::TeamHandleType;
123 using MeshIndex = typename Traits::MeshIndex;
124
125 const auto& buckets = mesh.get_bucket_ids(rank, sel);
126 auto team_exec = TeamPolicy(buckets.size(), NTHREADS_PER_DEVICE_TEAM);
127 Kokkos::parallel_for(
128 algName, team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
129 auto bktId = buckets.device_get(team.league_rank());
130 auto& bkt = mesh.get_bucket(rank, bktId);
131
132 const size_t bktLen = bkt.size();
133 Kokkos::parallel_for(
134 Kokkos::TeamThreadRange(team, bktLen), [&](const size_t& bktIndex) {
135 MeshIndex meshIdx{bkt.bucket_id(), static_cast<unsigned>(bktIndex)};
136 algorithm(meshIdx);
137 });
138 });
139}

319template <
320 typename Mesh,
321 typename FieldManager,
322 typename DataReqType,
323 typename AlgFunctor>
324void
325run_elem_algorithm(
326 const std::string algName,
327 const MeshInfo<Mesh, FieldManager>& meshInfo,
328 const stk::topology::rank_t rank,
329 const DataReqType& dataReqs,
330 const stk::mesh::Selector& sel,
331 const AlgFunctor algorithm)
332{
333 using Traits = NGPMeshTraits<Mesh>;
334 using TeamPolicy = typename Traits::TeamPolicy;
335 using TeamHandleType = typename Traits::TeamHandleType;
336 using MeshIndex = typename Traits::MeshIndex;
337
338 const auto& ndim = meshInfo.ndim();
339 const auto& ngpMesh = meshInfo.ngp_mesh();
340 const auto& fieldMgr = meshInfo.ngp_field_manager();
341
342 ElemDataRequestsGPU dataReqNGP(fieldMgr, dataReqs);
343
344 const int nodesPerElement = nodes_per_entity(dataReqNGP);
345 STK_NGP_ThrowRequire(nodesPerElement != 0);
346
347 const auto reqType =
348 (rank == stk::topology::ELEM_RANK) ? ElemReqType::ELEM : ElemReqType::FACE;
349 const int bytes_per_team = 0;
350 const int bytes_per_thread =
351 impl::ngp_calc_thread_shmem_size<sierra::nalu::DoubleType>(
352 ndim, dataReqNGP, reqType);
353
354 const auto& buckets = ngpMesh.get_bucket_ids(rank, sel);
355 auto team_exec = impl::ngp_mesh_team_policy<TeamPolicy>(
356 buckets.size(), bytes_per_team, bytes_per_thread);
357
358 Kokkos::parallel_for(
359 algName, team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
360 auto bktId = buckets.device_get(team.league_rank());
361 auto& bkt = ngpMesh.get_bucket(rank, bktId);
362
363 ElemSimdData<Mesh> elemData(team, ndim, nodesPerElement, dataReqNGP);
364
365 const size_t bktLen = bkt.size();
366 const size_t simdBktLen = get_num_simd_groups(bktLen);
367
368 Kokkos::parallel_for(
369 Kokkos::TeamThreadRange(team, simdBktLen), [&](const size_t& bktIndex) {
370 int nSimdElems = get_length_of_next_simd_group(bktIndex, bktLen);
371 elemData.numSimdElems = nSimdElems;
372
373 for (int is = 0; is < nSimdElems; ++is) {
374 const unsigned bktOrd = bktIndex * simdLen + is;
375 MeshIndex meshIdx{bkt.bucket_id(), bktOrd};
376 const auto& elem = bkt[bktOrd];
377 elemData.elemInfo[is] =
378 EntityInfo<Mesh>{meshIdx, elem, ngpMesh.get_nodes(rank, meshIdx)};
379
380 fill_pre_req_data(
381 dataReqNGP, ngpMesh, rank, elem, *elemData.scrView[is]);
382 }
383
384#if !defined(KOKKOS_ENABLE_GPU)
385 copy_and_interleave(
386 elemData.scrView, nSimdElems, elemData.simdScrView);
387#endif
388
389 fill_master_element_views(dataReqNGP, elemData.simdScrView);
390 algorithm(elemData);
391 });
392 });
393}

499template <
500 typename Mesh,
501 typename FieldManager,
502 typename DataReqType,
503 typename AlgFunctor>
504void
505run_face_elem_algorithm(
506 const std::string& algName,
507 const MeshInfo<Mesh, FieldManager>& meshInfo,
508 const DataReqType& faceDataReqs,
509 const DataReqType& elemDataReqs,
510 const stk::mesh::Selector& sel,
511 const AlgFunctor algorithm)
512{
513 static constexpr stk::topology::rank_t elemRank = stk::topology::ELEMENT_RANK;
514 const stk::topology::rank_t sideRank = meshInfo.meta().side_rank();
515 using Traits = NGPMeshTraits<Mesh>;
516 using TeamPolicy = typename Traits::TeamPolicy;
517 using TeamHandleType = typename Traits::TeamHandleType;
518 using MeshIndex = typename Traits::MeshIndex;
519
520 const auto& ndim = meshInfo.ndim();
521 const auto& ngpMesh = meshInfo.ngp_mesh();
522 const auto& fieldMgr = meshInfo.ngp_field_manager();
523
524 ElemDataRequestsGPU faceDataNGP(fieldMgr, faceDataReqs);
525 ElemDataRequestsGPU elemDataNGP(fieldMgr, elemDataReqs);
526
527 const int nodesPerElement = nodes_per_entity(elemDataNGP);
528 const int nodesPerFace = nodes_per_entity(faceDataNGP, METype::FACE);
529 STK_NGP_ThrowRequire(nodesPerElement != 0);
530 STK_NGP_ThrowRequire(nodesPerFace != 0);
531
532 const int bytes_per_team = 0;
533 const int bytes_per_thread =
534 impl::ngp_calc_thread_shmem_size<sierra::nalu::DoubleType>(
535 ndim, faceDataNGP, elemDataNGP);
536
537 const auto& buckets = ngpMesh.get_bucket_ids(sideRank, sel);
538 auto team_exec = impl::ngp_mesh_team_policy<TeamPolicy>(
539 buckets.size(), bytes_per_team, bytes_per_thread);
540
541 Kokkos::parallel_for(
542 algName, team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
543 auto bktId = buckets.device_get(team.league_rank());
544 auto& bkt = ngpMesh.get_bucket(sideRank, bktId);
545
546 FaceElemSimdData<Mesh> faceElemData(
547 team, ndim, nodesPerFace, nodesPerElement, faceDataNGP, elemDataNGP);
548
549 const size_t bktLen = bkt.size();
550 const size_t simdBktLen = get_num_simd_groups(bktLen);
551
552 Kokkos::parallel_for(
553 Kokkos::TeamThreadRange(team, simdBktLen), [&](const size_t& bktIndex) {
554 size_t nSimdFaces = get_length_of_next_simd_group(bktIndex, bktLen);
555 size_t nFacesProcessed = 0;
556
557 // Only group face/elem pairs in SIMD that have the same face ordinals
558 do {
559 int elemFaceOrd = -1;
560 int simdFaceIdx = 0;
561
562 while ((nFacesProcessed + simdFaceIdx) < nSimdFaces) {
563 const auto& bktOrd =
564 bktIndex * simdLen + nFacesProcessed + simdFaceIdx;
565 const auto& face = bkt[bktOrd];
566 const auto faceIdx = ngpMesh.fast_mesh_index(face);
567 int faceOrd = ngpMesh.get_element_ordinals(sideRank, faceIdx)[0];
568
569 // If we have one or more faces processed, then does the current
570 // candidate face the same face ordinal as the ones we have
571 // processed so far
572 if ((elemFaceOrd >= 0) && (faceOrd != elemFaceOrd))
573 break;
574
575 const auto elems = ngpMesh.get_elements(sideRank, faceIdx);
576 MeshIndex meshIdx{bkt.bucket_id(), static_cast<unsigned>(bktOrd)};
577 const auto elem = elems[0];
578 const auto elemIdx = ngpMesh.fast_mesh_index(elem);
579 faceElemData.faceInfo[simdFaceIdx] = BcFaceElemInfo<Mesh>{
580 meshIdx,
581 face,
582 elem,
583 ngpMesh.get_nodes(sideRank, faceIdx),
584 ngpMesh.get_nodes(elemRank, elemIdx),
585 faceOrd};
586
587 fill_pre_req_data(
588 faceDataNGP, ngpMesh, sideRank, face,
589 *faceElemData.scrFaceView[simdFaceIdx]);
590 fill_pre_req_data(
591 elemDataNGP, ngpMesh, elemRank, elem,
592 *faceElemData.scrElemView[simdFaceIdx]);
593
594 elemFaceOrd = faceOrd;
595 ++simdFaceIdx;
596 }
597 faceElemData.faceOrd = elemFaceOrd;
598 faceElemData.numSimdElems = simdFaceIdx;
599 nFacesProcessed += simdFaceIdx;
600
601#if !defined(KOKKOS_ENABLE_GPU)
602 copy_and_interleave(
603 faceElemData.scrFaceView, faceElemData.numSimdElems,
604 faceElemData.simdFaceView);
605 copy_and_interleave(
606 faceElemData.scrElemView, faceElemData.numSimdElems,
607 faceElemData.simdElemView);
608#endif
609 fill_master_element_views(
610 faceDataNGP, faceElemData.simdFaceView, elemFaceOrd);
611 fill_master_element_views(
612 elemDataNGP, faceElemData.simdElemView, elemFaceOrd);
613
614 algorithm(faceElemData);
615 } while (nFacesProcessed < nSimdFaces);
616 });
617 });
618}

773 const auto& ngpMesh = meshInfo.ngp_mesh();
774 const auto& fieldMgr = meshInfo.ngp_field_manager();

776 ElemDataRequestsGPU faceDataNGP(fieldMgr, faceDataReqs);
777 ElemDataRequestsGPU elemDataNGP(fieldMgr, elemDataReqs);
778
779 const int nodesPerElement = nodes_per_entity(elemDataNGP);
780 const int nodesPerFace = nodes_per_entity(faceDataNGP, METype::FACE);
781 STK_NGP_ThrowRequire(nodesPerElement != 0);
782 STK_NGP_ThrowRequire(nodesPerFace != 0);
783
784 const int bytes_per_team = 0;
785 const int bytes_per_thread =
786 impl::ngp_calc_thread_shmem_size<double>(ndim, faceDataNGP, elemDataNGP);
787
788 const auto& buckets = ngpMesh.get_bucket_ids(sideRank, sel);
789 auto team_exec = impl::ngp_mesh_team_policy<TeamPolicy>(
790 buckets.size(), bytes_per_team, bytes_per_thread);
791
792 Kokkos::parallel_for(
793 team_exec, KOKKOS_LAMBDA(const TeamHandleType& team) {
794 auto bktId = buckets.device_get(team.league_rank());
795 auto& bkt = ngpMesh.get_bucket(sideRank, bktId);
796
797 typename Traits::ScratchViewsType faceViews(
798 team, ndim, nodesPerFace, faceDataNGP);
799 typename Traits::ScratchViewsType elemViews(
800 team, ndim, nodesPerElement, elemDataNGP);
801
802 const size_t bktLen = bkt.size();
803 Kokkos::parallel_for(
804 Kokkos::TeamThreadRange(team, bktLen), [&](const size_t& bktIndex) {
805 MeshIndex meshIdx{bkt.bucket_id(), static_cast<unsigned>(bktIndex)};
806 const auto face = bkt[bktIndex];
807 const auto faceIdx = ngpMesh.fast_mesh_index(face);
808 const auto elements = ngpMesh.get_elements(sideRank, faceIdx);
809
810 STK_NGP_ThrowAssert(elements.size() == 1);
811 const auto faceOrd =
812 ngpMesh.get_element_ordinals(sideRank, faceIdx)[0];
813 const auto elem = elements[0];
814 const auto elemIdx = ngpMesh.fast_mesh_index(elem);
815
816 // Fill up face data
817 fill_pre_req_data(faceDataNGP, ngpMesh, sideRank, face, faceViews);
818 fill_master_element_views(faceDataNGP, faceViews);
819 // Fill up element data
820 fill_pre_req_data(elemDataNGP, ngpMesh, elemRank, elem, elemViews);
821 fill_master_element_views(elemDataNGP, elemViews);
822
823 const BcFaceElemInfo<Mesh> faceinfo{
824 meshIdx, face, elem, ngpMesh.get_nodes(elemRank, elemIdx), faceOrd};
825
826 algorithm(faceinfo, elemViews, faceViews);
827 });
828 });
829}
