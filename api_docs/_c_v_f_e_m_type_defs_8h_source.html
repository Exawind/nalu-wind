29template <int p, typename Scalar = DoubleType>
30struct CVFEMViews
31{
32 static constexpr int n1D = p + 1;
33 static constexpr int nscs = p;
34 static constexpr int dim = 3;
35 static constexpr int npe = n1D * n1D * n1D;
36
37 using nodal_scalar_array = Scalar[n1D][n1D][n1D];
38 using nodal_scalar_view = ViewType<nodal_scalar_array>;
39 using nodal_scalar_wsv = ScratchWorkView<npe, nodal_scalar_view>;
40 //--------------------------------------------------------------------------
41 using nodal_vector_array = Scalar[n1D][n1D][n1D][dim];
42 using nodal_vector_view = ViewType<nodal_vector_array>;
43 using nodal_vector_wsv =
44 ScratchWorkView<dim * n1D * n1D * n1D, nodal_vector_view>;
45 //--------------------------------------------------------------------------
46 using nodal_tensor_array = Scalar[n1D][n1D][n1D][dim][dim];
47 using nodal_tensor_view = ViewType<nodal_tensor_array>;
48 using nodal_tensor_wsv =
49 ScratchWorkView<dim * dim * n1D * n1D * n1D, nodal_tensor_view>;
50 //--------------------------------------------------------------------------
51 using scs_scalar_array = Scalar[dim][n1D][n1D][n1D];
52 using scs_scalar_view = ViewType<scs_scalar_array>;
53 using scs_scalar_wsv =
54 ScratchWorkView<dim * n1D * n1D * n1D, scs_scalar_view>;
55 //--------------------------------------------------------------------------
56 using scs_vector_array = Scalar[dim][n1D][n1D][n1D][dim];
57 using scs_vector_view = ViewType<scs_vector_array>;
58 using scs_vector_wsv =
59 ScratchWorkView<dim * dim * n1D * n1D * n1D, scs_vector_view>;
60 //--------------------------------------------------------------------------
61 using scs_symmtensor_array = Scalar[dim][n1D][n1D][n1D][dim][dim];
62 using scs_symmtensor_view = ViewType<scs_symmtensor_array>;
63 using scs_symmtensor_wsv =
64 ScratchWorkView<dim * dim * dim * n1D * n1D * n1D, scs_symmtensor_view>;
65 //--------------------------------------------------------------------------
66 using scs_tensor_array = Scalar[dim][n1D][n1D][n1D][dim][dim];
67 using scs_tensor_view = ViewType<scs_tensor_array>;
68 using scs_tensor_wsv =
69 ScratchWorkView<dim * dim * dim * n1D * n1D * n1D, scs_tensor_view>;
70 //--------------------------------------------------------------------------
71 using matrix_array = Scalar[npe][npe];
72 using matrix_view = ViewType<matrix_array>;
73 //--------------------------------------------------------------------------
74 using matrix_vector_array = Scalar[dim * npe][dim * npe];
75 using matrix_vector_view = ViewType<matrix_vector_array>;
76};
