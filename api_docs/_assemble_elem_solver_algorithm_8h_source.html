49 template <typename LambdaFunction>
50 void run_algorithm(stk::mesh::BulkData& bulk_data, LambdaFunction lambdaFunc)
51 {
52 stk::mesh::MetaData& meta_data = bulk_data.mesh_meta_data();
53 const int nDim = meta_data.spatial_dimension();
54 const int lhsSize = rhsSize_ * rhsSize_;
55 const int scratchIdsSize = rhsSize_;
56
57 const stk::mesh::NgpMesh& ngpMesh = realm_.ngp_mesh();
58 const nalu_ngp::FieldManager& fieldMgr = realm_.ngp_field_manager();
59 ElemDataRequestsGPU dataNeededNGP(fieldMgr, dataNeededByKernels_);
60
61 const auto reqType = (entityRank_ == stk::topology::ELEM_RANK)
62 ? ElemReqType::ELEM
63 : ElemReqType::FACE;
64
65 const int bytes_per_team = 0;
66 const int bytes_per_thread = calculate_shared_mem_bytes_per_thread(
67 lhsSize, rhsSize_, scratchIdsSize, meta_data.spatial_dimension(),
68 dataNeededNGP, reqType);
69
70 stk::mesh::Selector elemSelector = meta_data.locally_owned_part() &
71 stk::mesh::selectUnion(partVec_) &
72 !realm_.get_inactive_selector();
73
74 const auto& elem_buckets =
75 stk::mesh::get_bucket_ids(bulk_data, entityRank_, elemSelector);
76
77 // Create local copies of class data
78 const auto entityRank = entityRank_;
79 const auto nodesPerEntity = nodesPerEntity_;
80 const auto rhsSize = rhsSize_;
81
82 auto team_exec = sierra::nalu::get_device_team_policy(
83 elem_buckets.size(), bytes_per_team, bytes_per_thread);
84 Kokkos::parallel_for(
85 team_exec, KOKKOS_LAMBDA(const sierra::nalu::DeviceTeamHandleType& team) {
86 auto bktId = elem_buckets.device_get(team.league_rank());
87 auto& b = ngpMesh.get_bucket(entityRank, bktId);
88
89#if !defined(KOKKOS_ENABLE_GPU)
90 STK_ThrowAssertMsg(
91 b.topology().num_nodes() == (unsigned)nodesPerEntity_,
92 "AssembleElemSolverAlgorithm expected nodesPerEntity_ = "
93 << nodesPerEntity_
94 << ", but b.topology().num_nodes() = " << b.topology().num_nodes());
95#endif
96
97 SharedMemData<DeviceTeamHandleType, DeviceShmem> smdata(
98 team, nDim, dataNeededNGP, nodesPerEntity, rhsSize);
99
100 const size_t bucketLen = b.size();
101 const size_t simdBucketLen = get_num_simd_groups(bucketLen);
102
103 Kokkos::parallel_for(
104 Kokkos::TeamThreadRange(team, simdBucketLen),
105 [&](const size_t& bktIndex) {
106 int numSimdElems =
107 get_length_of_next_simd_group(bktIndex, bucketLen);
108 smdata.numSimdElems = numSimdElems;
109
110 for (int simdElemIndex = 0; simdElemIndex < numSimdElems;
111 ++simdElemIndex) {
112 stk::mesh::Entity element = b[bktIndex * simdLen + simdElemIndex];
113 const auto elemIndex = ngpMesh.fast_mesh_index(element);
114 smdata.ngpElemNodes[simdElemIndex] =
115 ngpMesh.get_nodes(entityRank, elemIndex);
116 fill_pre_req_data(
117 dataNeededNGP, ngpMesh, entityRank, element,
118 *smdata.prereqData[simdElemIndex]);
119 }
120
121#if !defined(KOKKOS_ENABLE_GPU)
122 // No need to interleave on GPUs
123 copy_and_interleave(
124 smdata.prereqData, numSimdElems, smdata.simdPrereqData);
125#endif
126
127 fill_master_element_views(dataNeededNGP, smdata.simdPrereqData);
128 lambdaFunc(smdata);
129 });
130 });
131 }

137 double diagRelaxFactor_{1.0};
138 unsigned nodesPerEntity_;
139 int rhsSize_;
140};
