99inline HostTeamPolicy
100get_host_team_policy(
101 const size_t sz, const size_t bytes_per_team, const size_t bytes_per_thread)
102{
103 HostTeamPolicy policy(sz, Kokkos::AUTO);
104 return policy.set_scratch_size(
105 1, Kokkos::PerTeam(bytes_per_team), Kokkos::PerThread(bytes_per_thread));
106}

133KOKKOS_FUNCTION SharedMemView<T*, TeamShmemType>

246KOKKOS_FUNCTION void
247set_vals(ViewType& view, const T& val)
248{
249 unsigned length = view.size();
250 auto* viewData = view.data();
