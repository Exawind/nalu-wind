382 template <typename T>
383 PecletFunction<T>*
384 EquationSystem::create_peclet_function(const std::string dofName)
385 {
386  PecletFunction<T>* pecletFunction = NULL;
387  if ("classic" == realm_.get_tanh_functional_form(dofName)) {
388  const T hybridFactor = realm_.get_hybrid_factor(dofName);
389  const T A = 5.0;
390  pecletFunction = new ClassicPecletFunction<T>(A, hybridFactor);
391  } else {
392  const T c1 = realm_.get_tanh_trans(dofName);
393  const T c2 = realm_.get_tanh_width(dofName);
394  pecletFunction = new TanhFunction<T>(c1, c2);
395  }
396  return pecletFunction;
397 }
398 
399 template <typename T>
400 PecletFunction<T>*
401 EquationSystem::ngp_create_peclet_function(const std::string& dofName)
402 {
403  PecletFunction<T>* pecletFunction = nullptr;
404  if ("classic" == realm_.get_tanh_functional_form(dofName)) {
405  const T hybridFactor = realm_.get_hybrid_factor(dofName);
406  const T A = 5.0;
407  pecletFunction =
408  nalu_ngp::create<ClassicPecletFunction<T>>(A, hybridFactor);
409  } else {
410  const T c1 = realm_.get_tanh_trans(dofName);
411  const T c2 = realm_.get_tanh_width(dofName);
412  pecletFunction = nalu_ngp::create<TanhFunction<T>>(c1, c2);
413  }
414 
415  ngpPecletFunctions_.push_back(pecletFunction);
416  return pecletFunction;
417 }
