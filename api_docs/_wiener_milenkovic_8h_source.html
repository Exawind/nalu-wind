19// Wiener-Milenkovic Parameters (WMP)
20//
21/* Wiener-Milenkovic parameters are a way of specifing rotation operations as
22 * three parameters. They can be composed into a single parameter whose net
23 * rotation will be the same as all the individual rotations. The assembled or
24 * 'composed' parameters behave sort of like a stack. They can be decomposed
25 * following first in last out ordering.
26 *
27 * References:
28 *
29 * Wang, Qi, and Wenbin Yu. "Geometrically nonlinear analysis of composite beams
30 * using Wiener-MilenkoviÄ‡ parameters." Journal of Renewable and Sustainable
31 * Energy 9.3 (2017): 033306.
32 *
33 * Bauchau, Olivier Andre. Flexible multibody dynamics. Vol. 176. Dordrecht:
34 * Springer, 2011. Section 13.7.4
35 */

73KOKKOS_FORCEINLINE_FUNCTION
74vs::Vector
75rotate(const vs::Vector wmP, const vs::Vector vec, const bool transpose = false)
76{
77 const double trans = bool_sign(transpose);
78 const double wm0 = compute_coeff_zero(wmP);
79 const double nu = 2.0 / (4.0 - wm0);
80 const double cosPhiO2 = 0.5 * wm0 * nu;
81 const vs::Vector crossWmVec = wmP ^ vec;
82
83 return vec + trans * nu * cosPhiO2 * crossWmVec +
84 0.5 * nu * nu * (wmP ^ crossWmVec);
85}

88KOKKOS_FORCEINLINE_FUNCTION
89vs::Tensor
90rotation_tensor(const vs::Vector wmP)
91{
92 const double wm0 = compute_coeff_zero(wmP);
93 const double nu = 2.0 / (4.0 - wm0);
94 const double cosPhiO2 = 0.5 * wm0 * nu;
95 auto wmp_ss = vs::skew_sym(wmP);
96 return vs::Tensor::I() + nu * cosPhiO2 * wmp_ss +
97 0.5 * nu * nu * (wmp_ss & wmp_ss);
98}

101KOKKOS_FORCEINLINE_FUNCTION
102vs::Vector
103compose(
104 const vs::Vector wmP,
105 const vs::Vector wmQ,
106 bool transposeP = false,
107 bool transposeQ = false)
108{
109 const double tP = bool_sign(transposeP);
110 const double tQ = bool_sign(transposeQ);
111
112 const double p0 = compute_coeff_zero(wmP);
113 const double q0 = compute_coeff_zero(wmQ);
114
115 const double delta1 = (4.0 - p0) * (4.0 - q0);
116 const double delta2 = p0 * q0 - tP * (wmP & wmQ);
117
118 const double sign = bool_sign(delta2 < 0.0);
119 const double preFac = sign * 4.0 / (delta1 + sign * delta2);
120
121 return preFac * (tQ * p0 * wmQ + tP * q0 * wmP + ((tP * wmP) ^ (tQ * wmQ)));
122}

142/* Linearly interpolate the Wiener-Milenkovic parameters between 'qStart' and
143 'qEnd' with an interpolating factor 'interpFac' see
144 O.A.Bauchau, 2011, Flexible Multibody Dynamics p. 649, section 17.2,
145 Algorithm 1'
146*/
147KOKKOS_FORCEINLINE_FUNCTION
148vs::Vector
149linear_interp_rotation(
150 const vs::Vector qStart, const vs::Vector qEnd, const double interpFac)
151{
152 // remove rigid body rotation
153 auto qIntermediate = pop(qStart, qEnd);
154 qIntermediate *= interpFac;
155 return push(qStart, qIntermediate);
156}
157
158KOKKOS_FORCEINLINE_FUNCTION
159vs::Vector
160linear_interp_translation(
161 const vs::Vector qStart, const vs::Vector qEnd, const double interpFac)
162{
163 return qStart + interpFac * (qEnd - qStart);
164}
