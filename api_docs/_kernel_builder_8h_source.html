35 inline std::pair<AssembleElemSolverAlgorithm*, bool>
36 build_or_add_part_to_solver_alg(
37  EquationSystem& eqSys,
38  stk::mesh::Part& part,
39  std::map<std::string, SolverAlgorithm*>& solverAlgs)
40 {
41  const stk::topology topo = part.topology();
42  const std::string algName =
43  eqSys.name_ + "_AssembleElemSolverAlg_" + topo.name();
44 
45  bool isNotNGP =
46  !(topo == stk::topology::HEXAHEDRON_8 ||
47  topo == stk::topology::HEXAHEDRON_27 ||
48  topo == stk::topology::QUADRILATERAL_4_2D ||
49  topo == stk::topology::TRIANGLE_3_2D || topo == stk::topology::WEDGE_6 ||
50  topo == stk::topology::TETRAHEDRON_4 || topo == stk::topology::PYRAMID_5);
51  STK_ThrowRequireMsg(
52  !isNotNGP, "Consolidated algorithm called on non-NGP MasterElement");
53 
54  auto itc = solverAlgs.find(algName);
55  bool createNewAlg = itc == solverAlgs.end();
56  if (createNewAlg) {
57  auto* theSolverAlg = new AssembleElemSolverAlgorithm(
58  eqSys.realm_, &part, &eqSys, stk::topology::ELEMENT_RANK,
59  topo.num_nodes());
60  STK_ThrowRequire(theSolverAlg != nullptr);
61 
62  NaluEnv::self().naluOutputP0()
63  << "Created the following interior elem alg: " << algName << std::endl;
64  solverAlgs.insert({algName, theSolverAlg});
65  } else {
66  auto& partVec = itc->second->partVec_;
67  if (std::find(partVec.begin(), partVec.end(), &part) == partVec.end()) {
68  partVec.push_back(&part);
69  }
70  }
71 
72  auto* theSolverAlg =
73  dynamic_cast<AssembleElemSolverAlgorithm*>(solverAlgs.at(algName));
74  STK_ThrowRequire(theSolverAlg != nullptr);
75 
76  return {theSolverAlg, createNewAlg};
77 }

179 template <template <typename> class T, typename... Args>
180 Kernel*
181 build_face_elem_topo_kernel(
182  stk::topology faceTopo, stk::topology elemTopo, Args&&... args)
183 {
184  switch (faceTopo.value()) {
185  case stk::topology::QUAD_4:
186  switch (elemTopo) {
187  case stk::topology::HEX_8:
188  return new T<AlgTraitsQuad4Hex8>(std::forward<Args>(args)...);
189  case stk::topology::PYRAMID_5:
190  return new T<AlgTraitsQuad4Pyr5>(std::forward<Args>(args)...);
191  case stk::topology::WEDGE_6:
192  return new T<AlgTraitsQuad4Wed6>(std::forward<Args>(args)...);
193  default:
194  STK_ThrowRequireMsg(
195  false, "Quad4 exposed face is not attached to either a hex8, pyr5, or "
196  "wedge6.");
197  return nullptr;
198  }
199  case stk::topology::TRI_3:
200  switch (elemTopo) {
201  case stk::topology::TET_4:
202  return new T<AlgTraitsTri3Tet4>(std::forward<Args>(args)...);
203  case stk::topology::PYRAMID_5:
204  return new T<AlgTraitsTri3Pyr5>(std::forward<Args>(args)...);
205  case stk::topology::WEDGE_6:
206  return new T<AlgTraitsTri3Wed6>(std::forward<Args>(args)...);
207  default:
208  STK_ThrowRequireMsg(
209  false,
210  "Tri3 exposed face is not attached to either a tet4, pyr5, or wedge6.");
211  return nullptr;
212  }
213  case stk::topology::LINE_2:
214  switch (elemTopo) {
215  case stk::topology::TRI_3_2D:
216  return new T<AlgTraitsEdge2DTri32D>(std::forward<Args>(args)...);
217  default:
218  return new T<AlgTraitsEdge2DQuad42D>(std::forward<Args>(args)...);
219  }
220  default:
221  return nullptr;
222  }
223 }
