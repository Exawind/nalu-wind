80 class LocalGraphArrays

83  template <typename ViewType>
84  LocalGraphArrays(const ViewType& rowLengths)
85  : rowPointers(), rowPointersData(nullptr), colIndices()
86  {
87  RowPointers rowPtrs("rowPtrs", rowLengths.size() + 1);
88  rowPointers = rowPtrs;

91  size_t nnz = compute_row_pointers(rowPointers, rowLengths);
92  colIndices = Kokkos::View<
93  LocalOrdinal*, typename LinSys::HostRowLengths::memory_space>(
94  Kokkos::ViewAllocateWithoutInitializing("colIndices"), nnz);
95  Kokkos::deep_copy(colIndices, INVALID);
96  }

98  size_t get_row_length(size_t localRow) const
99  {
100  return rowPointersData[localRow + 1] - rowPointersData[localRow];
101  }

103  void insertIndices(
104  size_t localRow, size_t numInds, const LocalOrdinal* inds, int numDof)
105  {
106  LocalOrdinal* row = &colIndices((int)rowPointersData[localRow]);
107  size_t rowLen = get_row_length(localRow);
108  LocalOrdinal* rowEnd = std::find(row, row + rowLen, INVALID);
109  for (size_t i = 0; i < numInds; ++i) {
110  LocalOrdinal* insertPoint = std::lower_bound(row, rowEnd, inds[i]);
111  if (insertPoint <= rowEnd && *insertPoint != inds[i]) {
112  insert(inds[i], numDof, insertPoint, rowEnd + numDof);
113  rowEnd += numDof;
114  }
115  }
116  }
117 
118  template <typename ViewType1, typename ViewType2>
119  static size_t
120  compute_row_pointers(ViewType1& rowPtrs, const ViewType2& rowLengths)
121  {
122  size_t nnz = 0;
123  auto rowPtrData = rowPtrs.data();
124  auto rowLens = rowLengths.data();
125  for (unsigned i = 0, iend = rowLengths.size(); i < iend; ++i) {
126  rowPtrData[i] = nnz;
127  nnz += rowLens[i];
128  }
129  rowPtrData[rowLengths.size()] = nnz;
130  return nnz;
131  }

133  RowPointers rowPointers;
134  typename RowPointers::traits::data_type rowPointersData;
135  ColumnIndices colIndices;

137 private:
138  void insert(
139  LocalOrdinal ind,
140  int numDof,
141  LocalOrdinal* insertPoint,
142  LocalOrdinal* rowEnd)
143  {
144  for (LocalOrdinal* ptr = rowEnd - 1; ptr != insertPoint; --ptr) {
145  *ptr = *(ptr - numDof);
146  }
147  for (int i = 0; i < numDof; ++i) {
148  *insertPoint++ = ind + i;
149  }
150  }
