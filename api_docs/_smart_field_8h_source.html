35 /* SmartField is a type that is designed to automatically handle field syncs
36  * and modifies
37  * It adheres to the pattern of sync before you use, and modify after
38  *
39  * There are 3 Template params:
40  * - FieldType: Data type for the underlying stk field
41  *
42  * - MEMSPACE: where the field is valid
43  * - DEVICE: is self explanatory
44  * - HOST: is using the same modern stk syntax based on
45  * kokkos::parallel_for but for host data
46  * - LEGACY: for the stk::mesh::field_data type access (bucket loops)
47  *
48  * - ACCESS: how the data can/should be used
49  * - READ: read only/const data accessors (only syncs data)
50  * - READ_WRITE: read and write data to field (syncs then marks modified)
51  * - WRITE_ALL: should only be used when overwritting all the field_data
52  * (clears sync state)
53  *
54  * NOTE: this implementation makes heavy use of SFINAE
55  */
56 template <typename FieldType, typename MEMSPACE, typename ACCESS>
57 class SmartField
58 {
59 };
