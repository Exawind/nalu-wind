71  void operator()(int index) const
72  {
73  auto pointId = actBulk_.coarseSearchPointIds_.h_view(index);
74  auto elemId = actBulk_.coarseSearchElemIds_.h_view(index);
75 
76  const stk::mesh::Entity elem =
77  stkBulk_.get_entity(stk::topology::ELEMENT_RANK, elemId);
78  const stk::topology& elemTopo = stkBulk_.bucket(elem).topology();
79  MasterElement* meSCV =
80  MasterElementRepo::get_volume_master_element_on_host(elemTopo);
81 
82  const unsigned numNodes = stkBulk_.num_nodes(elem);
83  const int numIp = meSCV->num_integration_points();
84 
85  // just allocate for largest expected size (hex27)
86  STK_ThrowAssert(numIp <= 216);
87  STK_ThrowAssert(numNodes <= 27);
88 
89  double scvip[216];
90  double elemcoords[27 * 3];
91  sierra::nalu::SharedMemView<double*> scvIp(&scvip[0], 216);
92  sierra::nalu::SharedMemView<double**> elemCoords(&elemcoords[0], 27, 3);
93 
94  stk::mesh::Entity const* elem_nod_rels = stkBulk_.begin_nodes(elem);
95 
96  for (unsigned i = 0; i < numNodes; i++) {
97  const double* coords =
98  stk::mesh::field_data(*coordinates_, elem_nod_rels[i]);
99  for (int j = 0; j < 3; j++) {
100  elemCoords(i, j) = coords[j];
101  }
102  }
103 
104  meSCV->determinant(elemCoords, scvIp);
105 
106  const auto* ipNodeMap = meSCV->ipNodeMap();
107 
108  for (int nIp = 0; nIp < numIp; nIp++) {
109  const int nodeIndex = ipNodeMap[nIp];
110  stk::mesh::Entity node = elem_nod_rels[nodeIndex];
111  const double* nodeCoords = stk::mesh::field_data(*coordinates_, node);
112  const double dual_vol = *stk::mesh::field_data(*dualNodalVolume_, node);
113  double* sourceTerm = stk::mesh::field_data(*actuatorSource_, node);
114 
115  // anything else that is required should be stashed on the functor
116  // during functor construction i.e. ActuatorBulk, flags, ActuatorMeta,
117  // etc.
118  innerLoopFunctor_(pointId, nodeCoords, sourceTerm, dual_vol, scvIp[nIp]);
119  }
120  }
