26template <int p, typename VolumeArray, typename DeltaArray, typename OutArray>
27KOKKOS_FORCEINLINE_FUNCTION void
28apply_mass(
29 int index,
30 double gamma,
31 const VolumeArray& vol,
32 const DeltaArray& delta,
33 OutArray& out)
34{
35 static constexpr auto vandermonde = Coeffs<p>::W;
36 for (int k = 0; k < p + 1; ++k) {
37 for (int j = 0; j < p + 1; ++j) {
38 ArrayND<ftype[p + 1]> scratch;
39 for (int i = 0; i < p + 1; ++i) {
40 scratch(i) = -gamma * vol(index, k, j, i) * delta(k, j, i);
41 }
42 for (int i = 0; i < p + 1; ++i) {
43 ftype acc(0);
44 for (int q = 0; q < p + 1; ++q) {
45 acc += vandermonde(i, q) * scratch(q);
46 }
47 out(k, j, i) = acc;
48 }
49 }
50 }
51
52 for (int i = 0; i < p + 1; ++i) {
53 ArrayND<ftype[p + 1][p + 1]> scratch;
54 for (int k = 0; k < p + 1; ++k) {
55 for (int j = 0; j < p + 1; ++j) {
56 ftype acc(0);
57 for (int q = 0; q < p + 1; ++q) {
58 acc += vandermonde(j, q) * out(k, q, i);
59 }
60 scratch(k, j) = acc;
61 }
62 }
63
64 for (int k = 0; k < p + 1; ++k) {
65 for (int j = 0; j < p + 1; ++j) {
66 ftype acc(0);
67 for (int q = 0; q < p + 1; ++q) {
68 acc += vandermonde(k, q) * scratch(q, j);
69 }
70 out(k, j, i) = acc;
71 }
72 }
73 }

76template <int p, int dir, typename InArray, typename OutArray>
77KOKKOS_FORCEINLINE_FUNCTION void
78edge_integral(const InArray& in, OutArray& out)
79{
80 for (int k = 0; k < p + 1; ++k) {
81 for (int j = 0; j < p + 1; ++j) {
82 for (int i = 0; i < p + 1; ++i) {
83 const int l = impl::active_index<dir>::index_0(k, j, i);
84 const int m = impl::active_index<dir>::index_1(k, j, i);
85 const int n = impl::active_index<dir>::index_2(k, j, i);
86 ftype acc(0);
87 for (int q = 0; q < p + 1; ++q) {
88 static constexpr auto vandermonde = Coeffs<p>::W;
89 acc += vandermonde(l, q) * shuffled_access<dir>(in, n, m, q);
90 }
91 out(k, j, i) = acc;
92 }
93 }
94 }

106template <int p, typename VolumeArray, typename InArray, typename OutArray>
107KOKKOS_FUNCTION void
108consistent_mass_time_derivative(
109 int index,
110 Kokkos::Array<double, 3> gammas,
111 const VolumeArray& vol,
112 const InArray& qm1,
113 const InArray& qp0,
114 const InArray& qp1,
115 OutArray& out)
116{
117 static constexpr auto vandermonde = Coeffs<p>::W;
118
119 for (int k = 0; k < p + 1; ++k) {
120 for (int j = 0; j < p + 1; ++j) {
121 ArrayND<ftype[p + 1]> scratch_1d;
122 for (int i = 0; i < p + 1; ++i) {
123 scratch_1d(i) =
124 -vol(index, k, j, i) *
125 (gammas[0] * qp1(index, k, j, i) + gammas[1] * qp0(index, k, j, i) +
126 gammas[2] * qm1(index, k, j, i));
127 }
128
129 for (int i = 0; i < p + 1; ++i) {
130 ftype integral_for_subcv_i = 0;
131 for (int q = 0; q < p + 1; ++q) {
132 integral_for_subcv_i += vandermonde(i, q) * scratch_1d(q);
133 }
134 out(k, j, i) = integral_for_subcv_i;
135 }
136 }
137 }
138
139 for (int i = 0; i < p + 1; ++i) {
140 ArrayND<ftype[p + 1][p + 1]> scratch;
141 for (int k = 0; k < p + 1; ++k) {
142 for (int j = 0; j < p + 1; ++j) {
143 scratch(k, j) = 0;
144 for (int q = 0; q < p + 1; ++q) {
145 scratch(k, j) += vandermonde(j, q) * out(k, q, i);
146 }
147 }
148 }
149
150 for (int j = 0; j < p + 1; ++j) {
151 for (int k = 0; k < p + 1; ++k) {
152 ftype integral_for_subcv_k = 0;
153 for (int q = 0; q < p + 1; ++q) {
154 integral_for_subcv_k += vandermonde(k, q) * scratch(q, j);
155 }
156 out(k, j, i) = integral_for_subcv_k;
157 }
158 }
159 }
160}

162template <int p, typename VolumeArray, typename InArray, typename OutArray>
163KOKKOS_FUNCTION void
164lumped_time_derivative(
165 int index,
166 Kokkos::Array<double, 3> gammas,
167 const VolumeArray& vol,
168 const InArray& qm1,
169 const InArray& qp0,
170 const InArray& qp1,
171 OutArray& out)
172{
173 static constexpr auto lumped = Coeffs<p>::Wl;
174 for (int k = 0; k < p + 1; ++k) {
175 const auto Wk = lumped[k];
176 for (int j = 0; j < p + 1; ++j) {
177 const auto WkWj = Wk * lumped[j];
178 for (int i = 0; i < p + 1; ++i) {
179 out(k, j, i) =
180 -WkWj * lumped[i] * vol(index, k, j, i) *
181 (gammas[0] * qp1(index, k, j, i) + gammas[1] * qp0(index, k, j, i) +
182 gammas[2] * qm1(index, k, j, i));
183 }
184 }
185 }

188template <

190 typename VolumeArray,
191 typename DeltaArray,
192 typename ScratchArray,
193 typename OutArray>
194KOKKOS_FORCEINLINE_FUNCTION void
195mass_term(
196 int index,
197 double gamma,
198 const VolumeArray& vol,
199 const DeltaArray& delta,
200 ScratchArray& scratch,
201 OutArray& out)
202{
203 for (int k = 0; k < p + 1; ++k) {
204 for (int j = 0; j < p + 1; ++j) {
205 for (int i = 0; i < p + 1; ++i) {
206 scratch(k, j, i) = -gamma * vol(index, k, j, i) * delta(k, j, i);
207 }
208 }
209 }
210 edge_integral<p, 0>(scratch, out);
211 edge_integral<p, 1>(out, scratch);
212 edge_integral<p, 2>(scratch, out);

215template <int p, typename VolumeArray, typename DeltaArray, typename OutArray>
216KOKKOS_FORCEINLINE_FUNCTION void
217lumped_mass_term(
218 int index,
219 double gamma,
220 const VolumeArray& vol,
221 const DeltaArray& delta,
222 OutArray& out)
223{
224 static constexpr auto lumped = Coeffs<p>::Wl;
225
226 for (int k = 0; k < p + 1; ++k) {
227 const auto gammaWk = -gamma * lumped(k);
228 for (int j = 0; j < p + 1; ++j) {
229 const auto gammaWkWj = lumped(j) * gammaWk;
230 for (int i = 0; i < p + 1; ++i) {
231 out(k, j, i) =
232 gammaWkWj * lumped(i) * vol(index, k, j, i) * delta(k, j, i);
233 }
234 }
235 }
