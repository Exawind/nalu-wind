109template <template <int> class PhysicsUpdate, typename... Args>
110std::unique_ptr<typename PhysicsUpdate<inst::P1>::update_type>
111make_updater(int p, Args&&... args)
112{
113 switch (p) {
114 case inst::P2:
115 return std::unique_ptr<PhysicsUpdate<inst::P2>>(
116 new PhysicsUpdate<inst::P2>(std::forward<Args>(args)...));
117 case inst::P3:
118 return std::unique_ptr<PhysicsUpdate<inst::P3>>(
119 new PhysicsUpdate<inst::P3>(std::forward<Args>(args)...));
120 case inst::P4:
121 return std::unique_ptr<PhysicsUpdate<inst::P4>>(
122 new PhysicsUpdate<inst::P4>(std::forward<Args>(args)...));
123 default:
124 return std::unique_ptr<PhysicsUpdate<inst::P1>>(
125 new PhysicsUpdate<inst::P1>(std::forward<Args>(args)...));
126 }
127}

129inline bool
130part_is_valid_for_matrix_free(int order, const stk::mesh::Part& part)
131{
132 if (
133 part.topology() == stk::topology::HEX_8 ||
134 part.topology() == stk::topology::QUAD_4) {
135 return order == 1;
136 }
137
138 if (
139 part.topology() == stk::topology::HEX_27 ||
140 part.topology() == stk::topology::QUAD_9) {
141 return order == 2;
142 }
143
144 if (part.topology().is_superelement()) {
145 return order == floor(std::cbrt(part.topology().num_nodes() + 1) - 1);
146 }
147
148 if (part.topology().is_superface()) {
149 return order == floor(std::sqrt(part.topology().num_nodes() + 1) - 1);
150 }
151
152 for (const auto* subpart : part.subsets()) {
153 if (subpart == nullptr) {
154 return false;
155 }
156 return part_is_valid_for_matrix_free(order, *subpart);
157 }
158 return false;
159}
