129 inline bool
130 part_is_valid_for_matrix_free(int order, const stk::mesh::Part& part)
131 {
132  if (
133  part.topology() == stk::topology::HEX_8 ||
134  part.topology() == stk::topology::QUAD_4) {
135  return order == 1;
136  }
137 
138  if (
139  part.topology() == stk::topology::HEX_27 ||
140  part.topology() == stk::topology::QUAD_9) {
141  return order == 2;
142  }
143 
144  if (part.topology().is_superelement()) {
145  return order == floor(std::cbrt(part.topology().num_nodes() + 1) - 1);
146  }
147 
148  if (part.topology().is_superface()) {
149  return order == floor(std::sqrt(part.topology().num_nodes() + 1) - 1);
150  }
151 
152  for (const auto* subpart : part.subsets()) {
153  if (subpart == nullptr) {
154  return false;
155  }
156  return part_is_valid_for_matrix_free(order, *subpart);
157  }
158  return false;
159 }
