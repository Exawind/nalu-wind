110 template <typename T, typename Container>
111 Kokkos::View<NGPCopyHolder<T>*, Kokkos::LayoutRight, MemSpace>
112 create_ngp_view(const Container& hostVec)
113 {
114  using NGPInfo = NGPCopyHolder<T>;
115  using NGPInfoView = Kokkos::View<NGPInfo*, Kokkos::LayoutRight, MemSpace>;
116 
117  const size_t numObjects = hostVec.size();
118  const std::string clsName(typeid(T).name());
119  const std::string debuggingName = "NGP" + clsName + "View";
120  NGPInfoView ngpVec(debuggingName, numObjects);
121 
122  typename NGPInfoView::HostMirror hostNgpView =
123  Kokkos::create_mirror_view(ngpVec);
124 
125  for (size_t i = 0; i < numObjects; ++i)
126  hostNgpView(i) = NGPInfo(hostVec[i]->create_on_device());
127 
128  Kokkos::deep_copy(ngpVec, hostNgpView);
129 
130  return ngpVec;
131 }
