24using Array2D = Kokkos::View<double* [3], MemSpace>;

26KOKKOS_FORCEINLINE_FUNCTION
27unsigned
28abl_find_index(const Array1D& xinp, const double& xout)
29{
30 const unsigned npts = xinp.extent(0);
31 for (unsigned i = 1; i < npts; ++i) {
32 if (xout <= xinp(i)) {
33 return (i - 1);
34 }
35 }
36 return (npts - 2);
37}

43class ABLScalarInterpolator
44{
45public:
46 using Array1D = abl_impl::Array1D;
47 KOKKOS_DEFAULTED_FUNCTION ABLScalarInterpolator() = default;
48 KOKKOS_DEFAULTED_FUNCTION ~ABLScalarInterpolator() = default;
49
50 ABLScalarInterpolator(
51 const std::vector<double>& xinp, const std::vector<double>& yinp)
52 : xinp_("ABLScalarX", xinp.size()),
53 yinp_("ABLScalarY", yinp.size()),
54 xinpHost_(Kokkos::create_mirror_view(xinp_)),
55 yinpHost_(Kokkos::create_mirror_view(yinp_)),
56 numPts_(xinp_.size())
57 {
58 for (unsigned i = 0; i < numPts_; ++i) {
59 xinpHost_(i) = xinp[i];
60 yinpHost_(i) = yinp[i];
61 }
62 Kokkos::deep_copy(xinp_, xinpHost_);
63 Kokkos::deep_copy(yinp_, yinpHost_);
64 }
65
68 void update_view_on_device(const std::vector<double>& yinp)
69 {
70 for (unsigned i = 0; i < numPts_; ++i) {
71 yinpHost_(i) = yinp[i];
72 }
73 Kokkos::deep_copy(yinp_, yinpHost_);
74 }
75
76 KOKKOS_FORCEINLINE_FUNCTION
77 void operator()(const double& xout, double& yout) const
78 {
79 // Forcing only at hub-height
80 if (numPts_ == 1)
81 yout = yinp_(0);
82
83 if (xout <= xinp_(0))
84 // Constant forcing below first specified height
85 yout = yinp_(0);
86 else if (xout >= xinp_(numPts_ - 1))
87 // Constant forcing above last specified height
88 yout = yinp_(numPts_ - 1);
89 else {
90 // Linearly interpolate source term within user-specified heights
91 auto idx = abl_impl::abl_find_index(xinp_, xout);
92 double fac = (xout - xinp_(idx)) / (xinp_(idx + 1) - xinp_(idx));
93 yout = (1.0 - fac) * yinp_(idx) + fac * yinp_(idx + 1);
94 }
95 }
96
97private:
99 Array1D xinp_;
101 Array1D yinp_;
102
104 Array1D::HostMirror xinpHost_;
106 Array1D::HostMirror yinpHost_;
107
109 unsigned numPts_;
110};

115class ABLVectorInterpolator
116{
117public:
118 using Array1D = abl_impl::Array1D;

123 ABLVectorInterpolator(
124 const std::vector<double>& xinp,
125 const std::vector<std::vector<double>>& yinp)
126 : xinp_("ABLVectorX", xinp.size()),
127 yinp_("ABLVectorY", yinp.size()),
128 xinpHost_(Kokkos::create_mirror_view(xinp_)),
129 yinpHost_(Kokkos::create_mirror_view(yinp_)),
130 numPts_(xinp.size())
131 {
132 for (unsigned i = 0; i < numPts_; ++i) {
133 xinpHost_(i) = xinp[i];
134 for (int d = 0; d < ndim; ++d)
135 yinpHost_(i, d) = yinp[d][i];
136 }
137 Kokkos::deep_copy(xinp_, xinpHost_);
138 Kokkos::deep_copy(yinp_, yinpHost_);
139 }
140
143 void update_view_on_device(const std::vector<std::vector<double>>& yinp)
144 {
145 for (unsigned i = 0; i < numPts_; ++i) {
146 for (int d = 0; d < ndim; ++d)
147 yinpHost_(i, d) = yinp[d][i];
148 }
149 Kokkos::deep_copy(yinp_, yinpHost_);
150 }
151
152 KOKKOS_FORCEINLINE_FUNCTION
153 void operator()(const double& xout, double* yout) const
154 {
155 // Forcing only at hub-height
156 if (numPts_ == 1)
157 for (int d = 0; d < ndim; d++)
158 yout[d] = yinp_(0, d);
159
160 if (xout <= xinp_(0))
161 // constant forcing below first specified height
162 for (int d = 0; d < ndim; d++)
163 yout[d] = yinp_(0, d);
164 else if (xout >= xinp_(numPts_ - 1)) {
165 // constant forcing above last specified height
166 const int idx = numPts_ - 1;
167 for (int d = 0; d < ndim; d++)
168 yout[d] = yinp_(idx, d);
169 } else {
170 // interpolate within user-provided range of inputs
171 auto idx = abl_impl::abl_find_index(xinp_, xout);
172 double fac = (xout - xinp_(idx)) / (xinp_(idx + 1) - xinp_(idx));
173 for (int d = 0; d < ndim; d++)
174 yout[d] = (1.0 - fac) * yinp_(idx, d) + fac * yinp_(idx + 1, d);
175 }
176 }
177
178private:
179 static constexpr int ndim = 3;
180
182 Array1D xinp_;
184 Array2D yinp_;
185
186 Array1D::HostMirror xinpHost_;
187 Array2D::HostMirror yinpHost_;
188
190 unsigned numPts_;
191};
