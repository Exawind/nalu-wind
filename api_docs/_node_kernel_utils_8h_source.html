24template <typename LambdaGeneral, typename LambdaUserSrc>
25void
26process_ngp_node_kernels(
27 std::map<AlgorithmType, SolverAlgorithm*>& solverAlgMap,
28 Realm& realm,
29 stk::mesh::Part* part,
30 EquationSystem* eqSystem,
31 LambdaGeneral lambdaGeneral,
32 LambdaUserSrc lambdaUsrSrc)
33{
34 const auto algMass = AlgorithmType::MASS;
35 const auto it = solverAlgMap.find(algMass);
36
37 if (it == solverAlgMap.end()) {
38 auto* nodeAlg = new AssembleNGPNodeSolverAlgorithm(realm, part, eqSystem);
39 solverAlgMap[algMass] = nodeAlg;
40
41 // Custom node src kernels for this equation system
42 lambdaGeneral(*nodeAlg);
43
44 const auto it =
45 realm.solutionOptions_->srcTermsMap_.find(eqSystem->eqnTypeName_);
46 if (it != realm.solutionOptions_->srcTermsMap_.end()) {
47 NaluEnv::self().naluOutputP0() << "Processing user source terms for "
48 << eqSystem->eqnTypeName_ << std::endl;
49 for (auto& srcName : it->second) {
50 lambdaUsrSrc(*nodeAlg, srcName);
51 }
52 }
53 } else {
54 it->second->partVec_.push_back(part);
55 }
56}
