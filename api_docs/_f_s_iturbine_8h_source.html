55template <typename T>
56inline vs::VectorT<T>
57vector_from_field(
58 const stk::mesh::Field<T>& field, const stk::mesh::Entity& node)
59{
60 // debug only check for optimization
61 assert(field.max_size() == 3);
62 assert(field.template type_is<T>());
63 T* ptr = stk::mesh::field_data(field, node);
64 return {ptr[0], ptr[1], ptr[2]};
65}
66
69template <typename T>
70inline void
71vector_to_field(
72 vs::VectorT<T> vec, stk::mesh::Field<T>& field, const stk::mesh::Entity& node)
73{
74 // debug only check for optimization
75 assert(field.max_size() == 3);
76 assert(field.template type_is<T>());
77 T* ptr = stk::mesh::field_data(field, node);
78 for (int i = 0; i < 3; ++i) {
79 ptr[i] = vec[i];
80 }
81}
