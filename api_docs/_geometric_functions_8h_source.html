23template <int p, int dk, int dj, int di, typename BoxArray>
24KOKKOS_FUNCTION typename BoxArray::value_type
25hex_jacobian_component_scs(const BoxArray& box, int l, int s, int r)
26{
27 enum { LN = 0, RN = 1 };
28 enum { XH = 0, YH = 1, ZH = 2 };
29
30 static constexpr auto nlin = Coeffs<p>::Nlin;
31 static constexpr auto ntlin = Coeffs<p>::Ntlin;
32 typename BoxArray::value_type jac(0);
33 switch (dj) {
34 case XH: {
35 const double lj = (dk == YH) ? ntlin(LN, l)
36 : (dk == XH) ? nlin(LN, r)
37 : nlin(LN, s);
38 const double rj = (dk == YH) ? ntlin(RN, l)
39 : (dk == XH) ? nlin(RN, r)
40 : nlin(RN, s);
41
42 const double lk = (dk == ZH) ? ntlin(LN, l) : nlin(LN, s);
43 const double rk = (dk == ZH) ? ntlin(RN, l) : nlin(RN, s);
44
45 jac = -lj * lk * box(di, 0) + lj * lk * box(di, 1) + rj * lk * box(di, 2) -
46 rj * lk * box(di, 3) - lj * rk * box(di, 4) + lj * rk * box(di, 5) +
47 rj * rk * box(di, 6) - rj * rk * box(di, 7);
48 break;
49 }
50 case YH: {
51 const double li = (dk == XH) ? ntlin(LN, l) : nlin(LN, r);
52 const double ri = (dk == XH) ? ntlin(RN, l) : nlin(RN, r);
53
54 const double lk = (dk == ZH) ? ntlin(LN, l) : nlin(LN, s);
55 const double rk = (dk == ZH) ? ntlin(RN, l) : nlin(RN, s);
56
57 jac = -li * lk * box(di, 0) - ri * lk * box(di, 1) + ri * lk * box(di, 2) +
58 li * lk * box(di, 3) - li * rk * box(di, 4) - ri * rk * box(di, 5) +
59 ri * rk * box(di, 6) + li * rk * box(di, 7);
60 break;
61 }
62 default: {
63 const double li = (dk == XH) ? ntlin(LN, l) : nlin(LN, r);
64 const double ri = (dk == XH) ? ntlin(RN, l) : nlin(RN, r);
65
66 const double lj = (dk == YH) ? ntlin(LN, l)
67 : (dk == XH) ? nlin(LN, r)
68 : nlin(LN, s);
69 const double rj = (dk == YH) ? ntlin(RN, l)
70 : (dk == XH) ? nlin(RN, r)
71 : nlin(RN, s);
72
73 jac = -li * lj * box(di, 0) - ri * lj * box(di, 1) - ri * rj * box(di, 2) -
74 li * rj * box(di, 3) + li * lj * box(di, 4) + ri * lj * box(di, 5) +
75 ri * rj * box(di, 6) + li * rj * box(di, 7);
76 break;
77 }
78 }
79 constexpr double isoParametricFactor = 0.5;
80 return jac * isoParametricFactor;
81}

112template <int p, int dk, typename BoxArrayType>
113KOKKOS_FUNCTION ArrayND<ftype[3]>
114linear_area(const BoxArrayType& box, int k, int j, int i)
115{
116 enum { XH = 0, YH = 1, ZH = 2 };
117 static constexpr int ds1 = (dk == XH) ? ZH : (dk == YH) ? XH : YH;
118 static constexpr int ds2 = (dk == XH) ? YH : (dk == YH) ? ZH : XH;
119 const auto dx_ds1 = hex_jacobian_component_scs<p, dk, ds1, XH>(box, k, j, i);
120 const auto dx_ds2 = hex_jacobian_component_scs<p, dk, ds2, XH>(box, k, j, i);
121 const auto dy_ds1 = hex_jacobian_component_scs<p, dk, ds1, YH>(box, k, j, i);
122 const auto dy_ds2 = hex_jacobian_component_scs<p, dk, ds2, YH>(box, k, j, i);
123 const auto dz_ds1 = hex_jacobian_component_scs<p, dk, ds1, ZH>(box, k, j, i);
124 const auto dz_ds2 = hex_jacobian_component_scs<p, dk, ds2, ZH>(box, k, j, i);
125 return ArrayND<ftype[3]>{
126 {dy_ds1 * dz_ds2 - dz_ds1 * dy_ds2, dz_ds1 * dx_ds2 - dx_ds1 * dz_ds2,
127 dx_ds1 * dy_ds2 - dy_ds1 * dx_ds2}};
128}

130template <int p, int dk, typename BoxArray>
131KOKKOS_FUNCTION ArrayND<ftype[3]>
132laplacian_metric(const BoxArray& box, int k, int j, int i)
133{
134 enum { XH = 0, YH = 1, ZH = 2 };
135
136 const auto jac = linear_hex_jacobian_scs<p, dk>(box, k, j, i);
137 const auto adj_jac = adjugate_matrix(jac);
138 const auto inv_detj =
139 1.0 / (jac(XH, XH) * adj_jac(XH, XH) + jac(YH, XH) * adj_jac(YH, XH) +
140 jac(ZH, XH) * adj_jac(ZH, XH));
141
142 switch (dk) {
143 case XH: {
144 return ArrayND<ftype[3]>{
145 {-inv_detj * (adj_jac(XH, XH) * adj_jac(XH, XH) +
146 adj_jac(XH, YH) * adj_jac(XH, YH) +
147 adj_jac(XH, ZH) * adj_jac(XH, ZH)),
148 -inv_detj * (adj_jac(XH, XH) * adj_jac(YH, XH) +
149 adj_jac(XH, YH) * adj_jac(YH, YH) +
150 adj_jac(XH, ZH) * adj_jac(YH, ZH)),
151 -inv_detj * (adj_jac(XH, XH) * adj_jac(ZH, XH) +
152 adj_jac(XH, YH) * adj_jac(ZH, YH) +
153 adj_jac(XH, ZH) * adj_jac(ZH, ZH))}};
154 }
155 case YH: {
156 return ArrayND<ftype[3]>{
157 {-inv_detj * (adj_jac(YH, XH) * adj_jac(YH, XH) +
158 adj_jac(YH, YH) * adj_jac(YH, YH) +
159 adj_jac(YH, ZH) * adj_jac(YH, ZH)),
160 -inv_detj * (adj_jac(YH, XH) * adj_jac(XH, XH) +
161 adj_jac(YH, YH) * adj_jac(XH, YH) +
162 adj_jac(YH, ZH) * adj_jac(XH, ZH)),
163 -inv_detj * (adj_jac(YH, XH) * adj_jac(ZH, XH) +
164 adj_jac(YH, YH) * adj_jac(ZH, YH) +
165 adj_jac(YH, ZH) * adj_jac(ZH, ZH))}};
166 }
167 default:
168 return ArrayND<ftype[3]>{
169 {-inv_detj * (adj_jac(ZH, XH) * adj_jac(ZH, XH) +
170 adj_jac(ZH, YH) * adj_jac(ZH, YH) +
171 adj_jac(ZH, ZH) * adj_jac(ZH, ZH)),
172 -inv_detj * (adj_jac(ZH, XH) * adj_jac(XH, XH) +
173 adj_jac(ZH, YH) * adj_jac(XH, YH) +
174 adj_jac(ZH, ZH) * adj_jac(XH, ZH)),
175 -inv_detj * (adj_jac(ZH, XH) * adj_jac(YH, XH) +
176 adj_jac(ZH, YH) * adj_jac(YH, YH) +
177 adj_jac(ZH, ZH) * adj_jac(YH, ZH))}};
178 }
179}

181template <int p, int dj, int di, typename CoeffArray, typename BoxArray>
182KOKKOS_FUNCTION typename BoxArray::value_type
183hex_jacobian_component(
184 const CoeffArray& Nlin, const BoxArray& box, int k, int j, int i)
185{
186 enum { LN = 0, RN = 1 };
187 enum { XH = 0, YH = 1, ZH = 2 };
188 if (dj == XH) {
189 return (-Nlin(LN, j) * Nlin(LN, k) * box(di, 0) +
190 Nlin(LN, j) * Nlin(LN, k) * box(di, 1) +
191 Nlin(RN, j) * Nlin(LN, k) * box(di, 2) -
192 Nlin(RN, j) * Nlin(LN, k) * box(di, 3) -
193 Nlin(LN, j) * Nlin(RN, k) * box(di, 4) +
194 Nlin(LN, j) * Nlin(RN, k) * box(di, 5) +
195 Nlin(RN, j) * Nlin(RN, k) * box(di, 6) -
196 Nlin(RN, j) * Nlin(RN, k) * box(di, 7)) *
197 0.5;
198 } else if (dj == YH) {
199 return (-Nlin(LN, i) * Nlin(LN, k) * box(di, 0) -
200 Nlin(RN, i) * Nlin(LN, k) * box(di, 1) +
201 Nlin(RN, i) * Nlin(LN, k) * box(di, 2) +
202 Nlin(LN, i) * Nlin(LN, k) * box(di, 3) -
203 Nlin(LN, i) * Nlin(RN, k) * box(di, 4) -
204 Nlin(RN, i) * Nlin(RN, k) * box(di, 5) +
205 Nlin(RN, i) * Nlin(RN, k) * box(di, 6) +
206 Nlin(LN, i) * Nlin(RN, k) * box(di, 7)) *
207 0.5;
208 } else {
209 return (-Nlin(LN, i) * Nlin(LN, j) * box(di, 0) -
210 Nlin(RN, i) * Nlin(LN, j) * box(di, 1) -
211 Nlin(RN, i) * Nlin(RN, j) * box(di, 2) -
212 Nlin(LN, i) * Nlin(RN, j) * box(di, 3) +
213 Nlin(LN, i) * Nlin(LN, j) * box(di, 4) +
214 Nlin(RN, i) * Nlin(LN, j) * box(di, 5) +
215 Nlin(RN, i) * Nlin(RN, j) * box(di, 6) +
216 Nlin(LN, i) * Nlin(RN, j) * box(di, 7)) *
217 0.5;
218 }
219}

221template <int p, typename BoxArray>
222KOKKOS_FUNCTION ArrayND<typename BoxArray::value_type[3][3]>
223linear_hex_jacobian(const BoxArray& box, int k, int j, int i)
224{
225 static constexpr auto coeff = Coeffs<p>::Nlin;
226 enum { XH = 0, YH = 1, ZH = 2 };
227 ArrayND<typename BoxArray::value_type[3][3]> jac;
228 jac(0, 0) = hex_jacobian_component<p, XH, XH>(coeff, box, k, j, i);
229 jac(0, 1) = hex_jacobian_component<p, XH, YH>(coeff, box, k, j, i);
230 jac(0, 2) = hex_jacobian_component<p, XH, ZH>(coeff, box, k, j, i);
231 jac(1, 0) = hex_jacobian_component<p, YH, XH>(coeff, box, k, j, i);
232 jac(1, 1) = hex_jacobian_component<p, YH, YH>(coeff, box, k, j, i);
233 jac(1, 2) = hex_jacobian_component<p, YH, ZH>(coeff, box, k, j, i);
234 jac(2, 0) = hex_jacobian_component<p, ZH, XH>(coeff, box, k, j, i);
235 jac(2, 1) = hex_jacobian_component<p, ZH, YH>(coeff, box, k, j, i);
236 jac(2, 2) = hex_jacobian_component<p, ZH, ZH>(coeff, box, k, j, i);
237 return jac;
238}
