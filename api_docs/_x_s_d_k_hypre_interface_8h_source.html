196 int CallFunction(HYPRE_Solver solver, HYPRE_Solver precond)
197 {
198 if (chooser_ == Hypre::Solver) {
199 if (option_ == 0) {
200 return int_func_(solver, int_param1_);
201 } else if (option_ == 1) {
202 return double_func_(solver, double_param1_);
203 } else if (option_ == 2) {
204 return double_int_func_(solver, double_param1_, int_param1_);
205 } else if (option_ == 3) {
206 return int_int_func_(solver, int_param1_, int_param2_);
207 } else if (option_ == 4) {
208 return int_star_func_(solver, int_star_param_);
209 } else if (option_ == 5) {
210 return double_star_func_(solver, double_star_param_);
211 } else if (option_ == 6) {
212 return char_star_func_(solver, char_star_param_);
213 }
214 } else {
215 if (option_ == 0) {
216 return int_func_(precond, int_param1_);
217 } else if (option_ == 1) {
218 return double_func_(precond, double_param1_);
219 } else if (option_ == 2) {
220 return double_int_func_(precond, double_param1_, int_param1_);
221 } else if (option_ == 3) {
222 return int_int_func_(precond, int_param1_, int_param2_);
223 } else if (option_ == 4) {
224 return int_star_func_(precond, int_star_param_);
225 } else if (option_ == 5) {
226 return double_star_func_(precond, double_star_param_);
227 } else if (option_ == 6) {
228 return char_star_func_(solver, char_star_param_);
229 }
230 }
231 // Should never reach here, but quell warning
232 return 0;
233 }
