20 template <class T, int dim = 3>
21 KOKKOS_FUNCTION T
22 get_M43_constant(T D[dim][dim], const double CMdeg)
23 {
24  STK_NGP_ThrowRequireMsg(
25  dim == 3, "Compute of M43 constant requires 3D problem");
26 
27  // Coefficients for the polynomial
28  double c[15] = {0.971903113666644, 0.065591700544879, 0.071103489538998,
29  0.049918716158500, -0.056904657182031, 0.097974249406576,
30  -0.015589487087603, 0.002003723064733, 0.002177318950949,
31  0.034227247973836, 0.001219656091495, 0.000417947294931,
32  0.000421085902741, 0.001223678414510, 0.003695127828465};
33 
34  T smallestEV = stk::math::min(D[0][0], stk::math::min(D[1][1], D[2][2]));
35  T largestEV = stk::math::max(D[0][0], stk::math::max(D[1][1], D[2][2]));
36  T middleEV = stk::math::if_then_else(
37  D[0][0] == smallestEV, stk::math::min(D[1][1], D[2][2]),
38  stk::math::if_then_else(
39  D[1][1] == smallestEV, stk::math::min(D[0][0], D[2][2]),
40  stk::math::min(D[0][0], D[1][1])));
41 
42  // Scale the EVs
43  middleEV = middleEV / smallestEV;
44  largestEV = largestEV / smallestEV;
45 
46  T r =
47  stk::math::sqrt(stk::math::pow(middleEV, 2) + stk::math::pow(largestEV, 2));
48  T theta = stk::math::acos(largestEV / r);
49 
50  T x = stk::math::log(r);
51  T y = stk::math::log(stk::math::sin(2.0 * theta));
52 
53  T poly = c[0] + c[1] * x + c[2] * y + c[3] * x * x + c[4] * x * y +
54  c[5] * y * y + c[6] * x * x * x + c[7] * x * x * y +
55  c[8] * x * y * y + c[9] * y * y * y + c[10] * x * x * x * x +
56  c[11] * x * x * x * y + c[12] * x * x * y * y +
57  c[13] * x * y * y * y + c[14] * y * y * y * y;
58 
59  return poly * CMdeg;
60 }
