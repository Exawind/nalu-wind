99 template <template <typename> class ElemAlg, class... Args>
100 void register_elem_algorithm(
101 AlgorithmType algType,
102 stk::mesh::Part* part,
103 const std::string& algSuffix,
104 Args&&... args)
105 {
106 const auto topo = part->topology();
107 const std::string entityName = "elem_" + topo.name();
108
109 const std::string algName = unique_name(algType, entityName, algSuffix);
110
111 const auto it = algMap_.find(algName);
112 if (it == algMap_.end()) {
113 algMap_[algName].reset(
114 nalu_ngp::create_elem_algorithm<Algorithm, ElemAlg>(
115 topo, realm_, part, std::forward<Args>(args)...));
116 NaluEnv::self().naluOutputP0()
117 << "Created algorithm = " << algName << std::endl;
118 } else {
119 it->second->partVec_.push_back(part);
120 }
121 }
