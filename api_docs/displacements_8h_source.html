109KOKKOS_FORCEINLINE_FUNCTION
110vs::Vector
111compute_translational_displacements(
112 const SixDOF deflections, const SixDOF referencePos, const vs::Vector cfdPos)
113{
114 const auto localPos = local_aero_coordinates(cfdPos, referencePos);
115 const auto delta = cfdPos - referencePos.position_;
116 // deflection roations need to be applied from the aerodynamic local frame of
117 // reference
118 const vs::Vector rotation =
119 wmp::rotate(deflections.orientation_, localPos, true);
120 return deflections.position_ + rotation - delta;
121}

136 const double ramp = 1.0)

146KOKKOS_FORCEINLINE_FUNCTION
147vs::Vector
148compute_mesh_velocity(
149 const SixDOF totalVel,
150 const SixDOF totalDis,
151 const SixDOF referencePos,
152 const vs::Vector cfdPos)
153{
154 const auto pointLocal = local_aero_coordinates(cfdPos, referencePos);
155 const auto pointRotate = wmp::rotate(totalDis.orientation_, pointLocal, true);
156 return totalVel.position_ + (totalVel.orientation_ ^ pointRotate);
157}

159KOKKOS_FORCEINLINE_FUNCTION
160vs::Vector
161compute_mesh_velocity(
162 const SixDOF totalVel,
163 const SixDOF totalDis,
164 const SixDOF referencePos,
165 const vs::Vector cfdPos,
166 const vs::Vector stiffVel,
167 const double ramp = 1.0)
168{
169 const auto pointLocal = local_aero_coordinates(cfdPos, referencePos);
170 const auto pointRotate = wmp::rotate(totalDis.orientation_, pointLocal, true);
171 const auto totalMeshVel =
172 totalVel.position_ + (totalVel.orientation_ ^ pointRotate);
173 return stiffVel + ramp * (totalMeshVel - stiffVel);
174}
