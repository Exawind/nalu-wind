39KOKKOS_INLINE_FUNCTION
40size_t
41get_num_simd_groups(size_t length)
42{
43 size_t numSimdGroups = length / simdLen;
44 const size_t remainder = length % simdLen;

46 numSimdGroups += 1;
47 }

51KOKKOS_INLINE_FUNCTION
52int
53get_length_of_next_simd_group(int index, int length)
54{
55 int nextLength = simdLen;
56 if (length - index * simdLen < simdLen) {
57 nextLength = length - index * simdLen;
58 }
59 if (nextLength < 0 || nextLength > simdLen) {
60 nextLength = 0;
61 }
62 return nextLength;
63}
