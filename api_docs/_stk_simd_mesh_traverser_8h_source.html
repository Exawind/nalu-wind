111 template <typename ValidFunc, typename RemainderFunc>
112 void
113 simd_traverse(
114  const stk::mesh::NgpMesh& mesh,
115  stk::topology::rank_t rank,
116  const stk::mesh::Selector& active,
117  ValidFunc func,
118  RemainderFunc rem)
119 {
120 
121  auto buckets = mesh.get_bucket_ids(rank, active);
122  const auto bucket_offsets = impl::simd_bucket_offsets(mesh, rank, buckets);
123  Kokkos::parallel_for(
124  DeviceTeamPolicy(buckets.size(), Kokkos::AUTO),
125  KOKKOS_LAMBDA(const typename DeviceTeamPolicy::member_type& team) {
126  const auto bucket_id = buckets.device_get(team.league_rank());
127  const auto& b = mesh.get_bucket(rank, bucket_id);
128  const auto bucket_len = b.size();
129  Kokkos::parallel_for(
130  Kokkos::TeamThreadRange(team, impl::get_num_simd_groups(bucket_len)),
131  [&](int e) {
132  const int num_simd_elems =
133  impl::get_length_of_next_simd_group(e, bucket_len);
134  const int simd_elem_index =
135  bucket_offsets.device_get(team.league_rank()) + e;
136  for (int ne = 0; ne < num_simd_elems; ++ne) {
137  func(simd_elem_index, ne, b[impl::bucket_index(e, ne)]);
138  }
139  for (int ne = num_simd_elems; ne < simd_len; ++ne) {
140  rem(simd_elem_index, ne, b[impl::bucket_index(e, 0)]);
141  }
142  });
143  });
144 }
