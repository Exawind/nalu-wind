38KOKKOS_INLINE_FUNCTION int
39get_num_simd_groups(int length)
40{
41 return (length % simd_len > 0) ? length / simd_len + 1 : length / simd_len;
42}

44KOKKOS_INLINE_FUNCTION int
45get_length_of_next_simd_group(int index, int length)
46{
47 int nextLength = simd_len;
48 if (length - index * simd_len < simd_len) {
49 nextLength = length - index * simd_len;
50 }
51
52 if (nextLength < 0 || nextLength > simd_len) {
53 nextLength = 0;
54 }
55 return nextLength;
56}

58inline int
59num_simd_elements(
60 const stk::mesh::NgpMesh& mesh,
61 stk::topology::rank_t rank,
62 const stk::mesh::Selector& selector)
63{
64 const auto buckets = mesh.get_bucket_ids(rank, selector);
65 int mesh_index = 0;
66 for (unsigned id = 0u; id < buckets.size(); ++id) {
67 mesh_index +=
68 get_num_simd_groups(mesh.get_bucket(rank, buckets[id]).size());
69 }
70 return mesh_index;
71}

73inline stk::NgpVector<int>
74simd_bucket_offsets(
75 const stk::mesh::NgpMesh& mesh,
76 stk::topology::rank_t rank,
77 stk::NgpVector<unsigned> buckets)
78{
79 stk::NgpVector<int> simd_lengths(buckets.size());
80 for (unsigned id = 0u; id < buckets.size(); ++id) {
81 simd_lengths[id] =
82 get_num_simd_groups(mesh.get_bucket(rank, buckets[id]).size());
83 }
84 stk::NgpVector<int> simd_offset(buckets.size());
85 int prev_sum = 0;
86 for (unsigned k = 0u; k < buckets.size(); ++k) {
87 simd_offset[k] = prev_sum;
88 prev_sum += simd_lengths[k];
89 }
90 simd_offset.copy_host_to_device();
91 return simd_offset;
92}

111template <typename ValidFunc, typename RemainderFunc>
112void
113simd_traverse(
114 const stk::mesh::NgpMesh& mesh,
115 stk::topology::rank_t rank,
116 const stk::mesh::Selector& active,
117 ValidFunc func,
118 RemainderFunc rem)
119{
120
121 auto buckets = mesh.get_bucket_ids(rank, active);
122 const auto bucket_offsets = impl::simd_bucket_offsets(mesh, rank, buckets);
123 Kokkos::parallel_for(
124 DeviceTeamPolicy(buckets.size(), Kokkos::AUTO),
125 KOKKOS_LAMBDA(const typename DeviceTeamPolicy::member_type& team) {
126 const auto bucket_id = buckets.device_get(team.league_rank());
127 const auto& b = mesh.get_bucket(rank, bucket_id);
128 const auto bucket_len = b.size();
129 Kokkos::parallel_for(
130 Kokkos::TeamThreadRange(team, impl::get_num_simd_groups(bucket_len)),
131 [&](int e) {
132 const int num_simd_elems =
133 impl::get_length_of_next_simd_group(e, bucket_len);
134 const int simd_elem_index =
135 bucket_offsets.device_get(team.league_rank()) + e;
136 for (int ne = 0; ne < num_simd_elems; ++ne) {
137 func(simd_elem_index, ne, b[impl::bucket_index(e, ne)]);
138 }
139 for (int ne = num_simd_elems; ne < simd_len; ++ne) {
140 rem(simd_elem_index, ne, b[impl::bucket_index(e, 0)]);
141 }
142 });
143 });
144}
