168template <typename AlgTraitsFace, typename AlgTraitsElem>
169struct AlgTraitsFaceElem
170{
171 using FaceTraits = AlgTraitsFace;
172 using ElemTraits = AlgTraitsElem;
173
174 static constexpr int nDim_ = ElemTraits::nDim_;
175 static_assert(
176 nDim_ == FaceTraits::nDim_, "inconsistent dimension specification");
177
178 static constexpr int nodesPerElement_ = ElemTraits::nodesPerElement_;
179 static constexpr int nodesPerFace_ = FaceTraits::nodesPerElement_;
180
181 static constexpr int numScsIp_ = ElemTraits::numScsIp_;
182 static constexpr int numScvIp_ = ElemTraits::numScvIp_;
183
184 static constexpr int numFaceIp_ = FaceTraits::numScsIp_;
185
186 static constexpr stk::topology::topology_t elemTopo_ = ElemTraits::topo_;
187 static constexpr stk::topology::topology_t faceTopo_ = FaceTraits::topo_;
188};
