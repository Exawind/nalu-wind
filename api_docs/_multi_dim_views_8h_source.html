26 KOKKOS_FUNCTION
27 inline size_t
28 adjust_up_to_alignment_boundary(size_t input, size_t alignment)
29 {
30  size_t remainder = input % alignment;
31  if (remainder > 0) {
32  return input + alignment - remainder;
33  }
34  return input;
35 }
