26KOKKOS_FUNCTION
27inline size_t
28adjust_up_to_alignment_boundary(size_t input, size_t alignment)
29{
30 size_t remainder = input % alignment;
31 if (remainder > 0) {
32 return input + alignment - remainder;
33 }
34 return input;
35}

51 KOKKOS_FUNCTION
52 MultiDimViews(
53 const TEAMHANDLETYPE& /*team*/,
54 unsigned /* maxOrdinal */,
55 const NumNeededViews& numNeededViews)
56 : indices(),
57 views_1D(),
58 views_2D(),
59 views_3D(),
60 views_4D(),
61 views_1D_size(0),
62 views_2D_size(0),
63 views_3D_size(0),
64 views_4D_size(0)
65 {
66 STK_NGP_ThrowRequireMsg(
67 (numNeededViews.num1DViews <= maxViewsPerDim ||
68 numNeededViews.num2DViews <= maxViewsPerDim ||
69 numNeededViews.num3DViews <= maxViewsPerDim ||
70 numNeededViews.num4DViews <= maxViewsPerDim),
71 "Number of requested views exceed maxViewsPerDim");
72#if !defined(KOKKOS_ENABLE_GPU)
73 for (unsigned i = 0; i < numNeededViews.num1DViews; ++i) {
74 views_1D[i] = nullptr;
75 }
76 for (unsigned i = 0; i < numNeededViews.num2DViews; ++i) {
77 views_2D[i] = nullptr;
78 }
79 for (unsigned i = 0; i < numNeededViews.num3DViews; ++i) {
80 views_3D[i] = nullptr;
81 }
82 for (unsigned i = 0; i < numNeededViews.num4DViews; ++i) {
83 views_4D[i] = nullptr;
84 }
85#else
86 for (unsigned i = 0; i < numNeededViews.num1DViews; ++i) {
87 new (&views_1D[i]) SharedMemView1D;
88 }
89 for (unsigned i = 0; i < numNeededViews.num2DViews; ++i) {
90 new (&views_2D[i]) SharedMemView2D;
91 }
92 for (unsigned i = 0; i < numNeededViews.num3DViews; ++i) {
93 new (&views_3D[i]) SharedMemView3D;
94 }
95 for (unsigned i = 0; i < numNeededViews.num4DViews; ++i) {
96 new (&views_4D[i]) SharedMemView4D;
97 }
98#endif
99 }
