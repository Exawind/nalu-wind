131template <typename T>
132KOKKOS_FORCEINLINE_FUNCTION VectorT<T>&
133VectorT<T>::normalize()
134{
135 T vmag = mag(*this);
136
137 if (vmag < Traits::eps()) {
138 *this = VectorT<T>::zero();
139 } else {
140 *this /= vmag;
141 }
142 return *this;
143}

145template <typename T>
146KOKKOS_FORCEINLINE_FUNCTION double
147angle(const VectorT<T>& v1, const VectorT<T>& v2)
148{
149 const double denom = stk::math::max(mag(v1) * mag(v2), vs::DTraits<T>::eps());
150 const double val = (v1 & v2) / denom;
151 return stk::math::acos(stk::math::max(-1.0, stk::math::min(1.0, val)));
152}
