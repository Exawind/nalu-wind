46 template <typename LambdaFunction>
47 void
48 run_face_elem_algorithm(stk::mesh::BulkData& bulk, LambdaFunction lamdbaFunc)
49 {
50 int nDim = bulk.mesh_meta_data().spatial_dimension();
51
52 // Register face ME instance in elemdata also to obtain face integration
53 // points
54 if (elemDataNeeded_.get_cvfem_face_me() == nullptr)
55 elemDataNeeded_.add_cvfem_face_me(faceDataNeeded_.get_cvfem_face_me());
56
57 int rhsSize = nodesPerElem_ * numDof_, lhsSize = rhsSize * rhsSize,
58 scratchIdsSize = rhsSize;
59
60 const stk::mesh::NgpMesh& ngpMesh = realm_.ngp_mesh();
61 const nalu_ngp::FieldManager& fieldMgr = realm_.ngp_field_manager();
62 ElemDataRequestsGPU faceDataNGP(fieldMgr, faceDataNeeded_);
63 ElemDataRequestsGPU elemDataNGP(fieldMgr, elemDataNeeded_);
64
65 const int bytes_per_team = 0;
66 const int bytes_per_thread = calculate_shared_mem_bytes_per_thread(
67 lhsSize, rhsSize, scratchIdsSize, nDim, faceDataNGP, elemDataNGP);
68
69 const auto nodesPerFace = nodesPerFace_;
70 const auto nodesPerElem = nodesPerElem_;
71 stk::mesh::Selector s_locally_owned_union =
72 bulk.mesh_meta_data().locally_owned_part() &
73 stk::mesh::selectUnion(partVec_);
74 stk::mesh::EntityRank sideRank = bulk.mesh_meta_data().side_rank();
75 const auto& buckets =
76 stk::mesh::get_bucket_ids(bulk, sideRank, s_locally_owned_union);
77
78 auto team_exec = sierra::nalu::get_device_team_policy(
79 buckets.size(), bytes_per_team, bytes_per_thread);
80 Kokkos::parallel_for(
81 team_exec, KOKKOS_LAMBDA(const sierra::nalu::DeviceTeamHandleType& team) {
82 auto bktId = buckets.device_get(team.league_rank());
83 auto& b = ngpMesh.get_bucket(sideRank, bktId);
84
85#if !defined(KOKKOS_ENABLE_GPU)
86 STK_ThrowAssertMsg(
87 b.topology().num_nodes() == (unsigned)nodesPerFace_,
88 "AssembleFaceElemSolverAlgorithm expected nodesPerEntity_ = "
89 << nodesPerFace_
90 << ", but b.topology().num_nodes() = " << b.topology().num_nodes());
91#endif
92
93 SharedMemData_FaceElem<DeviceTeamHandleType, DeviceShmem> smdata(
94 team, nDim, faceDataNGP, elemDataNGP, nodesPerFace, nodesPerElem,
95 rhsSize);
96
97 const size_t bucketLen = b.size();
98 const size_t simdBucketLen =
99 sierra::nalu::get_num_simd_groups(bucketLen);
100
101 Kokkos::parallel_for(
102 Kokkos::TeamThreadRange(team, simdBucketLen),
103 [&](const size_t& bktIndex) {
104 size_t simdGroupLen =
105 sierra::nalu::get_length_of_next_simd_group(bktIndex, bucketLen);
106 size_t numFacesProcessed = 0;
107 do {
108 int elemFaceOrdinal = -1;
109 int simdFaceIndex = 0;
110 while ((numFacesProcessed + simdFaceIndex) < simdGroupLen) {
111 stk::mesh::Entity face =
112 b[bktIndex * simdLen + numFacesProcessed + simdFaceIndex];
113 const auto ngpFaceIndex = ngpMesh.fast_mesh_index(face);
114 // STK_ThrowAssertMsg(
115 // bulk.num_elements(face) == 1,
116 // "Expecting just 1 element attached to face!");
117 int thisElemFaceOrdinal =
118 ngpMesh.get_element_ordinals(sideRank, ngpFaceIndex)[0];
119
120 if (
121 elemFaceOrdinal >= 0 &&
122 thisElemFaceOrdinal != elemFaceOrdinal) {
123 break;
124 }
125
126 const auto& elems =
127 ngpMesh.get_elements(sideRank, ngpFaceIndex);
128 const auto elemIndex = ngpMesh.fast_mesh_index(elems[0]);
129
130 smdata.ngpConnectedNodes[simdFaceIndex] =
131 ngpMesh.get_nodes(stk::topology::ELEMENT_RANK, elemIndex);
132 smdata.elemFaceOrdinal = thisElemFaceOrdinal;
133 elemFaceOrdinal = thisElemFaceOrdinal;
134 sierra::nalu::fill_pre_req_data(
135 faceDataNGP, ngpMesh, sideRank, face,
136 *smdata.faceViews[simdFaceIndex]);
137 sierra::nalu::fill_pre_req_data(
138 elemDataNGP, ngpMesh, stk::topology::ELEMENT_RANK, elems[0],
139 *smdata.elemViews[simdFaceIndex]);
140 ++simdFaceIndex;
141 }
142 smdata.numSimdFaces = simdFaceIndex;
143 numFacesProcessed += simdFaceIndex;
144
145#if !defined(KOKKOS_ENABLE_GPU)
146 // No need to interleave on GPUs
147 copy_and_interleave(
148 smdata.faceViews, smdata.numSimdFaces, smdata.simdFaceViews);
149 copy_and_interleave(
150 smdata.elemViews, smdata.numSimdFaces, smdata.simdElemViews);
151#endif
152
153 fill_master_element_views(
154 faceDataNGP, smdata.simdFaceViews, smdata.elemFaceOrdinal);
155 fill_master_element_views(
156 elemDataNGP, smdata.simdElemViews, smdata.elemFaceOrdinal);
157
158 lamdbaFunc(smdata);
159 } while (numFacesProcessed < simdGroupLen);
160 });
161 });
162 }
163
164 ElemDataRequests faceDataNeeded_;
165 ElemDataRequests elemDataNeeded_;
166 double diagRelaxFactor_{1.0};
167 unsigned numDof_;
168 unsigned nodesPerFace_;
169 unsigned nodesPerElem_;
170 int rhsSize_;
171};
