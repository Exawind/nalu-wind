92 // Return a field by the given name and of type T, the template parameter.
93 // This function will throw if the named field is not of the type
94 // specified by the template parameter: ScalarFieldType, VectorFieldType,
95 // ScalarIntFieldType, GlobalIdFieldType,....
96 template <typename T>
97 stk::mesh::Field<T>* get_field_ptr(
98 const std::string& name,
99 stk::mesh::FieldState state = stk::mesh::FieldState::StateNone) const
100 {
101 FieldDefTypes fieldDef =
102 FieldRegistry::query(numDimensions_, numStates_, name);
103 FieldPointerTypes pointerSet = std::visit(
104 [&](auto def) -> FieldPointerTypes {
105 return &meta_.get_field<T>(def.rank, name)->field_of_state(state);
106 },
107 fieldDef);
108 return std::get<stk::mesh::Field<T>*>(pointerSet);
109 }
