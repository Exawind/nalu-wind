32 template <typename FieldInfoViewType>
33 KOKKOS_INLINE_FUNCTION NumNeededViews
34 count_needed_field_views(const FieldInfoViewType& neededFields)
35 {
36  NumNeededViews numNeededViews = {0, 0, 0, 0};
37 
38  for (unsigned i = 0; i < neededFields.size(); ++i) {
39  const auto& fieldInfo = neededFields(i);
40  stk::mesh::EntityRank fieldEntityRank = get_entity_rank(fieldInfo);
41  unsigned scalarsDim1 = fieldInfo.scalarsDim1;
42  unsigned scalarsDim2 = fieldInfo.scalarsDim2;
43 
44  if (
45  fieldEntityRank == stk::topology::EDGE_RANK ||
46  fieldEntityRank == stk::topology::FACE_RANK ||
47  fieldEntityRank == stk::topology::ELEM_RANK) {
48  if (scalarsDim2 == 0) {
49  numNeededViews.num1DViews++;
50  } else {
51  numNeededViews.num2DViews++;
52  }
53  } else if (fieldEntityRank == stk::topology::NODE_RANK) {
54  if (scalarsDim2 == 0) {
55  if (scalarsDim1 == 1) {
56  numNeededViews.num1DViews++;
57  } else {
58  numNeededViews.num2DViews++;
59  }
60  } else {
61  numNeededViews.num3DViews++;
62  }
63  } else {
64  STK_NGP_ThrowRequireMsg(false, "Unknown stk-rank");
65  }
66  }
67 
68  return numNeededViews;
69 }

144 template <
145  typename T,
146  typename SHMEM,
147  typename TEAMHANDLETYPE,
148  typename ELEMDATAREQUESTSTYPE,
149  typename MULTIDIMVIEWSTYPE>
150 KOKKOS_FUNCTION int
151 create_needed_field_views(
152  const TEAMHANDLETYPE& team,
153  const ELEMDATAREQUESTSTYPE& dataNeeded,
154  int nodesPerEntity,
155  MULTIDIMVIEWSTYPE& fieldViews)
156 {
157  int numScalars = 0;
158 
159  const typename ELEMDATAREQUESTSTYPE::FieldInfoView& neededFields =
160  dataNeeded.get_fields();
161  for (unsigned i = 0; i < neededFields.size(); ++i) {
162  const typename ELEMDATAREQUESTSTYPE::FieldInfoType& fieldInfo =
163  neededFields(i);
164  stk::mesh::EntityRank fieldEntityRank = get_entity_rank(fieldInfo);
165  unsigned scalarsDim1 = fieldInfo.scalarsDim1;
166  unsigned scalarsDim2 = fieldInfo.scalarsDim2;
167 
168  if (
169  fieldEntityRank == stk::topology::EDGE_RANK ||
170  fieldEntityRank == stk::topology::FACE_RANK ||
171  fieldEntityRank == stk::topology::ELEM_RANK) {
172  if (scalarsDim2 == 0) {
173  fieldViews.add_1D_view(
174  get_field_ordinal(fieldInfo),
175  get_shmem_view_1D<T, TEAMHANDLETYPE, SHMEM>(team, scalarsDim1));
176  numScalars += scalarsDim1;
177  } else {
178  fieldViews.add_2D_view(
179  get_field_ordinal(fieldInfo),
180  get_shmem_view_2D<T, TEAMHANDLETYPE, SHMEM>(
181  team, scalarsDim1, scalarsDim2));
182  numScalars += scalarsDim1 * scalarsDim2;
183  }
184  } else if (fieldEntityRank == stk::topology::NODE_RANK) {
185  if (scalarsDim2 == 0) {
186  if (scalarsDim1 == 1) {
187  fieldViews.add_1D_view(
188  get_field_ordinal(fieldInfo),
189  get_shmem_view_1D<T, TEAMHANDLETYPE, SHMEM>(team, nodesPerEntity));
190  numScalars += nodesPerEntity;
191  } else {
192  fieldViews.add_2D_view(
193  get_field_ordinal(fieldInfo),
194  get_shmem_view_2D<T, TEAMHANDLETYPE, SHMEM>(
195  team, nodesPerEntity, scalarsDim1));
196  numScalars += nodesPerEntity * scalarsDim1;
197  }
198  } else {
199  fieldViews.add_3D_view(
200  get_field_ordinal(fieldInfo),
201  get_shmem_view_3D<T, TEAMHANDLETYPE, SHMEM>(
202  team, nodesPerEntity, scalarsDim1, scalarsDim2));
203  numScalars += nodesPerEntity * scalarsDim1 * scalarsDim2;
204  }
205  } else {
206  STK_NGP_ThrowRequireMsg(false, "Unknown stk-rank");
207  }
208  }
209 
210  return numScalars;
211 }
