123template <typename Mesh, typename Field, typename SimdDataType>

148 struct Ops
149 {
150 KOKKOS_INLINE_FUNCTION
151 Ops(
152 const NodeFieldOp<Mesh, Field, SimdDataType>& obj,
153 const SimdDataType& edata_in,
154 const unsigned ni_in,
155 const unsigned ic_in)
156 : obj_(obj), edata_(edata_in), ni(ni_in), ic(ic_in)
157 {
158 }
159
160 KOKKOS_DEFAULTED_FUNCTION ~Ops() = default;
161
162 KOKKOS_INLINE_FUNCTION
163 void operator=(const DoubleType& val) const
164 {
165 const auto& msh = obj_.ngpMesh_;
166 const auto& fld = obj_.ngpField_;
167 const auto* einfo = edata_.info();
168#ifdef USE_STK_SIMD_NONE
169 fld.get(msh, einfo[0].entityNodes[ni], ic) = stk::simd::get_data(val, 0);
170#else
171 for (int is = 0; is < edata_.numSimdElems; ++is) {
172 fld.get(msh, einfo[is].entityNodes[ni], ic) =
173 stk::simd::get_data(val, is);
174 }
175#endif
176 }
177
178 KOKKOS_INLINE_FUNCTION
179 void operator+=(const DoubleType& val) const
180 {
181 const auto& msh = obj_.ngpMesh_;
182 const auto& fld = obj_.ngpField_;
183 const auto* einfo = edata_.info();
184#ifdef USE_STK_SIMD_NONE
185 Kokkos::atomic_add(
186 &fld.get(msh, einfo[0].entityNodes[ni], ic),
187 stk::simd::get_data(val, 0));
188#else
189 for (int is = 0; is < edata_.numSimdElems; ++is) {
190 Kokkos::atomic_add(
191 &fld.get(msh, einfo[is].entityNodes[ni], ic),
192 stk::simd::get_data(val, is));
193 }
194#endif
195 }
196
197 KOKKOS_INLINE_FUNCTION
198 void operator=(const double& val) const

201 const auto& fld = obj_.ngpField_;
202 const auto* einfo = edata_.info();
203#ifdef USE_STK_SIMD_NONE

205#else

209#endif
210 }

213 void operator+=(const double& val) const

216 const auto& fld = obj_.ngpField_;
217 const auto* einfo = edata_.info();

265template <typename Mesh, typename Field, typename SimdDataType>
266struct ElemFieldOp
267{
268 static_assert(
269 std::is_floating_point<typename Field::value_type>::value,
270 "NGP field must have a floating type");
271
272 KOKKOS_INLINE_FUNCTION
273 ElemFieldOp(const Field& ngpField) : ngpField_(ngpField) {}
274
275 KOKKOS_DEFAULTED_FUNCTION ~ElemFieldOp() = default;
276
279 struct Ops
280 {
281 KOKKOS_INLINE_FUNCTION
282 Ops(
283 const ElemFieldOp<Mesh, Field, SimdDataType>& obj,
284 const SimdDataType& edata_in,
285 unsigned ic_in)
286 : obj_(obj), edata_(edata_in), ic(ic_in)
287 {
288 }
289
290 KOKKOS_DEFAULTED_FUNCTION ~Ops() = default;
291
292 KOKKOS_INLINE_FUNCTION
293 void operator=(const DoubleType& val) const
294 {
295 const auto& fld = obj_.ngpField_;
296 const auto* einfo = edata_.info();
297#ifdef USE_STK_SIMD_NONE
298 fld.get(einfo[0].meshIdx, ic) = stk::simd::get_data(val, 0);
299#else
300 for (int is = 0; is < edata_.numSimdElems; ++is) {
301 fld.get(einfo[is].meshIdx, ic) = stk::simd::get_data(val, is);
302 }
303#endif
304 }
305
306 KOKKOS_INLINE_FUNCTION
307 void operator+=(const DoubleType& val) const
308 {
309 const auto& fld = obj_.ngpField_;
310 const auto* einfo = edata_.info();
311
312 // No atomic_add here as only one element active per thread
313#ifdef USE_STK_SIMD_NONE
314 fld.get(einfo[0].meshIdx, ic) += stk::simd::get_data(val, 0);
315#else
316 for (int is = 0; is < edata_.numSimdElems; ++is) {
317 fld.get(einfo[is].meshIdx, ic) += stk::simd::get_data(val, is);
318 }
319#endif
320 }
321
322 KOKKOS_INLINE_FUNCTION
323 void operator=(const double& val) const
324 {
325 const auto& fld = obj_.ngpField_;
326 const auto* einfo = edata_.info();
327#ifdef USE_STK_SIMD_NONE
328 fld.get(einfo[0].meshIdx, ic) = val;
329#else
330 for (int is = 0; is < edata_.numSimdElems; ++is) {
331 fld.get(einfo[is].meshIdx, ic) = val;
332 }
333#endif
334 }
335
336 KOKKOS_INLINE_FUNCTION
337 void operator+=(const double& val) const
338 {
339 const auto& fld = obj_.ngpField_;
340 const auto* einfo = edata_.info();
341#ifdef USE_STK_SIMD_NONE
342 fld.get(einfo[0].meshIdx, ic) += val;
343#else
344 for (int is = 0; is < edata_.numSimdElems; ++is) {
345 fld.get(einfo[is].meshIdx, ic) += val;
346 }
347#endif
348 }
349
350 KOKKOS_INLINE_FUNCTION
351 void operator-=(const DoubleType& val) const { Ops::operator+=(-val); }
