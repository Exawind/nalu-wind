58 Ops() = default;
59
60 KOKKOS_DEFAULTED_FUNCTION ~Ops() = default;

64 {

129 NodeFieldOp(const Mesh& ngpMesh, const Field& ngpField)

142 {
143 KOKKOS_DEFAULTED_FUNCTION
144 Ops() = default;
145
146 KOKKOS_DEFAULTED_FUNCTION ~Ops() = default;
147
148 KOKKOS_INLINE_FUNCTION
149 void operator=(const DoubleType& val) const
150 {
151 const auto& msh = obj_.ngpMesh_;
152 const auto& fld = obj_.ngpField_;
153 const auto* einfo = edata_.info();
154#ifdef USE_STK_SIMD_NONE
155 fld.get(msh, einfo[0].entityNodes[ni], ic) = stk::simd::get_data(val, 0);
156#else
157 for (int is = 0; is < edata_.numSimdElems; ++is) {
158 fld.get(msh, einfo[is].entityNodes[ni], ic) =
159 stk::simd::get_data(val, is);
160 }
161#endif
162 }
163

166 {
167 const auto& msh = obj_.ngpMesh_;
168 const auto& fld = obj_.ngpField_;

170#ifdef USE_STK_SIMD_NONE
171 Kokkos::atomic_add(
172 &fld.get(msh, einfo[0].entityNodes[ni], ic),
173 stk::simd::get_data(val, 0));
174#else
175 for (int is = 0; is < edata_.numSimdElems; ++is) {
176 Kokkos::atomic_add(
177 &fld.get(msh, einfo[is].entityNodes[ni], ic),
178 stk::simd::get_data(val, is));
179 }
180#endif
181 }

188 const auto* einfo = edata_.info();

251template <typename Mesh, typename Field, typename SimdDataType>
252struct ElemFieldOp
253{
254 static_assert(
255 std::is_floating_point<typename Field::value_type>::value,
256 "NGP field must have a floating type");
257
258 KOKKOS_INLINE_FUNCTION
259 ElemFieldOp(const Field& ngpField) : ngpField_(ngpField) {}
260
261 KOKKOS_DEFAULTED_FUNCTION ~ElemFieldOp() = default;
262
265 struct Ops
266 {
267 KOKKOS_DEFAULTED_FUNCTION
268 Ops() = default;
269
270 KOKKOS_DEFAULTED_FUNCTION ~Ops() = default;
271
272 KOKKOS_INLINE_FUNCTION
273 void operator=(const DoubleType& val) const
274 {
275 const auto& fld = obj_.ngpField_;
276 const auto* einfo = edata_.info();
277#ifdef USE_STK_SIMD_NONE
278 fld.get(einfo[0].meshIdx, ic) = stk::simd::get_data(val, 0);
279#else
280 for (int is = 0; is < edata_.numSimdElems; ++is) {
281 fld.get(einfo[is].meshIdx, ic) = stk::simd::get_data(val, is);
282 }
283#endif
284 }
285
286 KOKKOS_INLINE_FUNCTION
287 void operator+=(const DoubleType& val) const
288 {
289 const auto& fld = obj_.ngpField_;
290 const auto* einfo = edata_.info();
291
292 // No atomic_add here as only one element active per thread
293#ifdef USE_STK_SIMD_NONE
294 fld.get(einfo[0].meshIdx, ic) += stk::simd::get_data(val, 0);
295#else
296 for (int is = 0; is < edata_.numSimdElems; ++is) {
297 fld.get(einfo[is].meshIdx, ic) += stk::simd::get_data(val, is);
298 }
299#endif
300 }
301
302 KOKKOS_INLINE_FUNCTION
303 void operator=(const double& val) const
304 {
305 const auto& fld = obj_.ngpField_;
306 const auto* einfo = edata_.info();
307#ifdef USE_STK_SIMD_NONE
308 fld.get(einfo[0].meshIdx, ic) = val;
309#else
310 for (int is = 0; is < edata_.numSimdElems; ++is) {
311 fld.get(einfo[is].meshIdx, ic) = val;
312 }
313#endif
314 }
315
316 KOKKOS_INLINE_FUNCTION
317 void operator+=(const double& val) const
318 {
319 const auto& fld = obj_.ngpField_;
320 const auto* einfo = edata_.info();
321#ifdef USE_STK_SIMD_NONE
322 fld.get(einfo[0].meshIdx, ic) += val;
323#else
324 for (int is = 0; is < edata_.numSimdElems; ++is) {
325 fld.get(einfo[is].meshIdx, ic) += val;
326 }
327#endif
328 }
329
330 KOKKOS_INLINE_FUNCTION
331 void operator-=(const DoubleType& val) const { Ops::operator+=(-val); }
332
333 KOKKOS_INLINE_FUNCTION
334 void operator-=(const double& val) const { Ops::operator+=(-val); }
335
336 const ElemFieldOp<Mesh, Field, SimdDataType>& obj_;
337
339 const SimdDataType& edata_;
340
342 unsigned ic;
343 };

348 {

350 }
351
353 const Field ngpField_;
354};
